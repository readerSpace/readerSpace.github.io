<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

function resize(){
  canvas.width = innerWidth;
  canvas.height = innerHeight;
}
resize();
addEventListener("resize", resize);

const refractEl = document.getElementById("refract");
const lightEl = document.getElementById("light");
const dispersionEl = document.getElementById("dispersion");

const refractText = document.getElementById("refractText");
const lightText = document.getElementById("lightText");
const dispersionText = document.getElementById("dispersionText");

const rotateBtn = document.getElementById("rotateBtn");

let angle = 0;
let autoRotate = true;

rotateBtn.onclick = () => {
  autoRotate = !autoRotate;
};

function gemVertices(cx, cy, r, rot){
  const pts = [];
  const n = 8;

  for(let i=0;i<n;i++){
    const a = rot + i/n*Math.PI*2;
    const rr = i%2===0 ? r : r*0.65;

    pts.push({
      x: cx + Math.cos(a)*rr,
      y: cy + Math.sin(a)*rr
    });
  }

  return pts;
}

function drawGem(pts){
  ctx.save();

  ctx.beginPath();
  ctx.moveTo(pts[0].x, pts[0].y);

  for(let p of pts){
    ctx.lineTo(p.x, p.y);
  }

  ctx.closePath();

  const grad = ctx.createRadialGradient(
    canvas.width*0.55,
    canvas.height*0.42,
    30,
    canvas.width*0.55,
    canvas.height*0.42,
    260
  );

  grad.addColorStop(0, "rgba(255,255,255,0.95)");
  grad.addColorStop(0.25, "rgba(180,220,255,0.8)");
  grad.addColorStop(0.7, "rgba(70,120,255,0.3)");
  grad.addColorStop(1, "rgba(20,40,120,0.15)");

  ctx.fillStyle = grad;
  ctx.fill();

  ctx.strokeStyle = "rgba(255,255,255,0.9)";
  ctx.lineWidth = 2;
  ctx.stroke();

  ctx.clip();

  drawInternalReflections(pts);

  ctx.restore();
}

function drawInternalReflections(pts){

  for(let i=0;i<24;i++){

    const p1 = pts[Math.floor(Math.random()*pts.length)];
    const p2 = pts[Math.floor(Math.random()*pts.length)];

    const grad = ctx.createLinearGradient(
      p1.x,p1.y,p2.x,p2.y
    );

    grad.addColorStop(0,"rgba(255,255,255,0)");
    grad.addColorStop(0.5,"rgba(255,255,255,0.5)");
    grad.addColorStop(1,"rgba(255,255,255,0)");

    ctx.strokeStyle = grad;
    ctx.lineWidth = Math.random()*3;

    ctx.beginPath();
    ctx.moveTo(p1.x,p1.y);
    ctx.lineTo(p2.x,p2.y);
    ctx.stroke();
  }
}

function drawIncomingLight(cx, cy){

  const power = Number(lightEl.value);

  ctx.strokeStyle = `rgba(255,255,255,${0.35+power/180})`;
  ctx.lineWidth = 4;

  ctx.beginPath();
  ctx.moveTo(120, cy-150);
  ctx.lineTo(cx-90, cy-40);
  ctx.stroke();

  const dispersion = Number(dispersionEl.value);

  const colors = [
    "#ff0000",
    "#ff8800",
    "#ffee00",
    "#00ff88",
    "#00aaff",
    "#6644ff"
  ];

  for(let i=0;i<colors.length;i++){

    const spread = (i - colors.length/2) * dispersion * 0.22;

    ctx.strokeStyle = colors[i];
    ctx.lineWidth = 3;

    ctx.beginPath();
    ctx.moveTo(cx+40, cy+10);

    ctx.lineTo(
      cx + 260,
      cy + spread
    );

    ctx.stroke();
  }
}

function drawGlow(cx, cy){

  const glow = ctx.createRadialGradient(
    cx, cy, 20,
    cx, cy, 260
  );

  glow.addColorStop(0,"rgba(255,255,255,0.32)");
  glow.addColorStop(0.5,"rgba(80,160,255,0.12)");
  glow.addColorStop(1,"rgba(0,0,0,0)");

  ctx.fillStyle = glow;

  ctx.beginPath();
  ctx.arc(cx,cy,260,0,Math.PI*2);
  ctx.fill();
}

function updateTexts(){
  refractText.textContent = Number(refractEl.value).toFixed(2);
  lightText.textContent = lightEl.value;
  dispersionText.textContent = dispersionEl.value;
}

function drawBackground(){

  const g = ctx.createLinearGradient(
    0,0,0,canvas.height
  );

  g.addColorStop(0,"#020617");
  g.addColorStop(1,"#0f172a");

  ctx.fillStyle = g;
  ctx.fillRect(0,0,canvas.width,canvas.height);

  for(let i=0;i<120;i++){

    const x = (i*137)%canvas.width;
    const y = (i*97)%canvas.height;

    ctx.fillStyle = "rgba(255,255,255,0.08)";

    ctx.beginPath();
    ctx.arc(x,y,Math.random()*2,0,Math.PI*2);
    ctx.fill();
  }
}

function loop(){

  updateTexts();

  drawBackground();

  if(autoRotate){
    angle += 0.003 * Number(refractEl.value);
  }

  const cx = canvas.width*0.58;
  const cy = canvas.height*0.5;

  drawGlow(cx,cy);

  const pts = gemVertices(
    cx,
    cy,
    170,
    angle
  );

  drawIncomingLight(cx,cy);

  drawGem(pts);

  requestAnimationFrame(loop);
}

loop();
