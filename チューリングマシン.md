const samples = {
  appendOne: {
    name: "1の列の末尾に1を追加",
    input: "111",
    start: "q0",
    rules: `# 1が続く間は右へ進む
q0 1 1 R q0
# 空白を見つけたら1を書いて停止
q0 _ 1 N HALT`
  },
  invertBits: {
    name: "0と1を反転する",
    input: "1011001",
    start: "q0",
    rules: `# 0なら1にする
q0 0 1 R q0
# 1なら0にする
q0 1 0 R q0
# 空白で停止
q0 _ _ N HALT`
  },
  unaryAdd: {
    name: "単項足し算 111+11 → 11111",
    input: "111+11",
    start: "q0",
    rules: `# 左の1を飛ばす
q0 1 1 R q0
# + を見つけたら空白にして右へ
q0 + _ R q1
# 右側の1を飛ばす
q1 1 1 R q1
# 末尾の空白を見つけたら左へ戻る
q1 _ _ L q2
# 最後の1を消して停止
q2 1 _ N HALT`
  },
  binaryIncrement: {
    name: "2進数に1を足す",
    input: "1011",
    start: "q0",
    rules: `# まず右端へ移動
q0 0 0 R q0
q0 1 1 R q0
q0 _ _ L q1
# 右端から繰り上がり処理
q1 0 1 N HALT
q1 1 0 L q1
q1 _ 1 N HALT`
  }
};

const sampleSelect = document.getElementById("sampleSelect");
const inputTapeEl = document.getElementById("inputTape");
const rulesTextEl = document.getElementById("rulesText");
const speedEl = document.getElementById("speed");
const speedTextEl = document.getElementById("speedText");
const tapeEl = document.getElementById("tape");
const stateBadge = document.getElementById("stateBadge");
const messageEl = document.getElementById("message");
const stepCountEl = document.getElementById("stepCount");
const headPosEl = document.getElementById("headPos");
const readSymbolEl = document.getElementById("readSymbol");
const outputTapeEl = document.getElementById("outputTape");
const currentRuleBox = document.getElementById("currentRuleBox");
const ruleTable = document.getElementById("ruleTable");
const loadBtn = document.getElementById("loadBtn");
const stepBtn = document.getElementById("stepBtn");
const runBtn = document.getElementById("runBtn");
const resetBtn = document.getElementById("resetBtn");

let tape = new Map();
let head = 0;
let state = "q0";
let startState = "q0";
let steps = 0;
let rules = new Map();
let halted = false;
let running = false;
let timer = null;
let lastRule = null;

for (const [key, sample] of Object.entries(samples)) {
  const opt = document.createElement("option");
  opt.value = key;
  opt.textContent = sample.name;
  sampleSelect.appendChild(opt);
}

function loadSample(key) {
  const s = samples[key];
  inputTapeEl.value = s.input;
  rulesTextEl.value = s.rules;
  startState = s.start;
  resetMachine();
}

function parseRules() {
  const map = new Map();
  const lines = rulesTextEl.value.split("\n");
  const parsed = [];

  for (let line of lines) {
    line = line.trim();
    if (!line || line.startsWith("#")) continue;
    const parts = line.split(/\s+/);
    if (parts.length !== 5) continue;
    const [from, read, write, move, to] = parts;
    const key = `${from}|${read}`;
    const rule = { from, read, write, move: move.toUpperCase(), to };
    map.set(key, rule);
    parsed.push(rule);
  }
  rules = map;
  renderRuleTable(parsed);
}

function renderRuleTable(parsed) {
  ruleTable.innerHTML = `
    <thead>
      <tr>
        <th>状態</th><th>読む</th><th>書く</th><th>移動</th><th>次状態</th>
      </tr>
    </thead>
    <tbody>
      ${parsed.map(r => `
        <tr>
          <td>${escapeHtml(r.from)}</td>
          <td>${escapeHtml(r.read)}</td>
          <td>${escapeHtml(r.write)}</td>
          <td>${escapeHtml(r.move)}</td>
          <td>${escapeHtml(r.to)}</td>
        </tr>`).join("")}
    </tbody>
  `;
}

function escapeHtml(s) {
  return String(s).replace(/[&<>"]/g, c => ({"&":"&amp;","<":"&lt;",">":"&gt;","\"":"&quot;"}[c]));
}

function resetMachine() {
  stopRun();
  parseRules();
  tape = new Map();
  const input = inputTapeEl.value || "";
  for (let i = 0; i < input.length; i++) {
    const ch = input[i] === " " ? "_" : input[i];
    if (ch !== "_") tape.set(i, ch);
  }
  head = 0;
  state = startState;
  steps = 0;
  halted = false;
  lastRule = null;
  messageEl.textContent = "準備完了";
  render();
}

function readTape(pos) {
  return tape.get(pos) || "_";
}

function writeTape(pos, symbol) {
  if (symbol === "_") tape.delete(pos);
  else tape.set(pos, symbol);
}

function stepMachine() {
  if (halted) return;
  parseRules();

  const symbol = readTape(head);
  const key = `${state}|${symbol}`;
  const rule = rules.get(key);

  if (!rule) {
    halted = true;
    lastRule = null;
    messageEl.textContent = `規則がないため停止: (${state}, ${symbol})`;
    render();
    stopRun();
    return;
  }

  writeTape(head, rule.write);
  if (rule.move === "R") head++;
  else if (rule.move === "L") head--;
  else if (rule.move === "N") {}
  else {
    halted = true;
    messageEl.textContent = `移動方向が不正です: ${rule.move}`;
    render();
    stopRun();
    return;
  }

  state = rule.to;
  steps++;
  lastRule = rule;

  if (state === "HALT") {
    halted = true;
    messageEl.textContent = "HALT状態に到達しました";
    stopRun();
  } else {
    messageEl.textContent = "実行中";
  }
  render();
}

function render() {
  renderTape();
  stateBadge.textContent = state;
  stateBadge.className = "stateBadge";
  if (halted && state === "HALT") stateBadge.classList.add("haltBadge");
  else if (halted) stateBadge.classList.add("errorBadge");

  stepCountEl.textContent = steps;
  headPosEl.textContent = head;
  readSymbolEl.textContent = readTape(head);
  outputTapeEl.textContent = getOutputTape();

  if (lastRule) {
    currentRuleBox.innerHTML = `
      <strong>直前に適用した規則</strong><br>
      状態 <code>${escapeHtml(lastRule.from)}</code> で <code>${escapeHtml(lastRule.read)}</code> を読んだので、
      <code>${escapeHtml(lastRule.write)}</code> を書き、<code>${escapeHtml(lastRule.move)}</code> に移動し、
      状態 <code>${escapeHtml(lastRule.to)}</code> へ移りました。
    `;
  } else {
    const symbol = readTape(head);
    const next = rules.get(`${state}|${symbol}`);
    if (next && !halted) {
      currentRuleBox.innerHTML = `
        <strong>次に適用される規則</strong><br>
        <code>${escapeHtml(next.from)} ${escapeHtml(next.read)} ${escapeHtml(next.write)} ${escapeHtml(next.move)} ${escapeHtml(next.to)}</code>
      `;
    } else {
      currentRuleBox.textContent = "現在適用できる規則はありません。";
    }
  }
}

function renderTape() {
  tapeEl.innerHTML = "";
  const min = head - 12;
  const max = head + 12;
  for (let i = min; i <= max; i++) {
    const box = document.createElement("div");
    box.className = "cellBox";
    if (i === head) {
      const mark = document.createElement("div");
      mark.className = "headMark";
      mark.textContent = "▼";
      box.appendChild(mark);
    }
    const cell = document.createElement("div");
    cell.className = "cell" + (i === head ? " head" : "");
    cell.textContent = readTape(i);
    const idx = document.createElement("div");
    idx.className = "index";
    idx.textContent = i;
    box.appendChild(cell);
    box.appendChild(idx);
    tapeEl.appendChild(box);
  }
}

function getOutputTape() {
  if (tape.size === 0) return "_";
  const keys = [...tape.keys()];
  const min = Math.min(...keys);
  const max = Math.max(...keys);
  let s = "";
  for (let i = min; i <= max; i++) s += readTape(i);
  return s.replace(/^_+|_+$/g, "") || "_";
}

function run() {
  if (running) {
    stopRun();
    return;
  }
  running = true;
  runBtn.textContent = "停止";
  runBtn.classList.remove("secondary");
  timer = setInterval(() => {
    if (!halted) stepMachine();
    else stopRun();
  }, Number(speedEl.value));
}

function stopRun() {
  running = false;
  clearInterval(timer);
  runBtn.textContent = "自動実行";
  runBtn.classList.add("secondary");
}

sampleSelect.addEventListener("change", () => loadSample(sampleSelect.value));
loadBtn.addEventListener("click", resetMachine);
stepBtn.addEventListener("click", stepMachine);
runBtn.addEventListener("click", run);
resetBtn.addEventListener("click", resetMachine);
speedEl.addEventListener("input", () => {
  speedTextEl.textContent = speedEl.value;
  if (running) {
    stopRun();
    run();
  }
});

loadSample("appendOne");
