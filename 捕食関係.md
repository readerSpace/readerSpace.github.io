  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const grassRate = document.getElementById("grassRate");
    const rabbitBirth = document.getElementById("rabbitBirth");
    const huntPower = document.getElementById("huntPower");
    const foxCost = document.getElementById("foxCost");
    const grassValue = document.getElementById("grassValue");
    const rabbitBirthValue = document.getElementById("rabbitBirthValue");
    const huntValue = document.getElementById("huntValue");
    const foxCostValue = document.getElementById("foxCostValue");
    const rabbitCountEl = document.getElementById("rabbitCount");
    const foxCountEl = document.getElementById("foxCount");
    const pauseBtn = document.getElementById("pauseBtn");
    const resetBtn = document.getElementById("resetBtn");

    let W = 900;
    let H = 600;
    let worldH = 430;
    let graphH = 150;
    let paused = false;
    let rabbits = [];
    let foxes = [];
    let grass = [];
    let history = [];
    let tick = 0;

    const GRID = 42;

    function rand(a, b) {
      return a + Math.random() * (b - a);
    }

    function clamp(v, a, b) {
      return Math.max(a, Math.min(b, v));
    }

    function dist2(a, b) {
      const dx = a.x - b.x;
      const dy = a.y - b.y;
      return dx * dx + dy * dy;
    }

    function resize() {
      const rect = canvas.getBoundingClientRect();
      const dpr = window.devicePixelRatio || 1;
      canvas.width = Math.floor(rect.width * dpr);
      canvas.height = Math.floor(rect.height * dpr);
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      W = rect.width;
      H = rect.height;
      graphH = 155;
      worldH = H - graphH - 18;
      initGrass();
    }

    function initGrass() {
      grass = [];
      for (let y = 0; y < GRID; y++) {
        grass[y] = [];
        for (let x = 0; x < GRID; x++) {
          grass[y][x] = Math.random();
        }
      }
    }

    function createRabbit(x = rand(30, W - 30), y = rand(30, worldH - 30)) {
      return {
        x, y,
        vx: rand(-1.2, 1.2),
        vy: rand(-1.2, 1.2),
        energy: rand(35, 70),
        age: 0
      };
    }

    function createFox(x = rand(30, W - 30), y = rand(30, worldH - 30)) {
      return {
        x, y,
        vx: rand(-1.5, 1.5),
        vy: rand(-1.5, 1.5),
        energy: rand(80, 130),
        age: 0
      };
    }

    function reset() {
      rabbits = [];
      foxes = [];
      history = [];
      tick = 0;
      initGrass();
      for (let i = 0; i < 70; i++) rabbits.push(createRabbit());
      for (let i = 0; i < 14; i++) foxes.push(createFox());
    }

    function grassAt(x, y) {
      const gx = clamp(Math.floor((x / W) * GRID), 0, GRID - 1);
      const gy = clamp(Math.floor((y / worldH) * GRID), 0, GRID - 1);
      return { gx, gy, value: grass[gy][gx] };
    }

    function updateGrass() {
      const rate = Number(grassRate.value);
      for (let y = 0; y < GRID; y++) {
        for (let x = 0; x < GRID; x++) {
          grass[y][x] = clamp(grass[y][x] + rate * (1 - grass[y][x]), 0, 1);
        }
      }
    }

    function moveAnimal(a, speed, fearTarget = null, chaseTarget = null) {
      if (fearTarget) {
        const dx = a.x - fearTarget.x;
        const dy = a.y - fearTarget.y;
        const d = Math.hypot(dx, dy) + 0.001;
        a.vx += (dx / d) * 0.35;
        a.vy += (dy / d) * 0.35;
      }
      if (chaseTarget) {
        const dx = chaseTarget.x - a.x;
        const dy = chaseTarget.y - a.y;
        const d = Math.hypot(dx, dy) + 0.001;
        a.vx += (dx / d) * 0.42;
        a.vy += (dy / d) * 0.42;
      }

      a.vx += rand(-0.22, 0.22);
      a.vy += rand(-0.22, 0.22);
      const v = Math.hypot(a.vx, a.vy) || 1;
      a.vx = (a.vx / v) * speed;
      a.vy = (a.vy / v) * speed;
      a.x += a.vx;
      a.y += a.vy;

      if (a.x < 8 || a.x > W - 8) a.vx *= -1;
      if (a.y < 8 || a.y > worldH - 8) a.vy *= -1;
      a.x = clamp(a.x, 8, W - 8);
      a.y = clamp(a.y, 8, worldH - 8);
    }

    function nearest(from, list, maxD) {
      let best = null;
      let bestD = maxD * maxD;
      for (const item of list) {
        const d = dist2(from, item);
        if (d < bestD) {
          bestD = d;
          best = item;
        }
      }
      return best;
    }

    function updateRabbits() {
      const birth = Number(rabbitBirth.value);
      const newRabbits = [];

      for (const r of rabbits) {
        r.age++;
        const predator = nearest(r, foxes, 70);
        moveAnimal(r, 1.45, predator, null);

        const g = grassAt(r.x, r.y);
        if (g.value > 0.12) {
          const eat = Math.min(g.value, 0.12);
          grass[g.gy][g.gx] -= eat;
          r.energy += eat * 40;
        }

        r.energy -= 0.09;

        if (r.energy > 65 && Math.random() < birth && rabbits.length + newRabbits.length < 260) {
          r.energy *= 0.72;
          newRabbits.push(createRabbit(r.x + rand(-10, 10), r.y + rand(-10, 10)));
        }
      }

      rabbits = rabbits.filter(r => r.energy > 0 && r.age < 2800);
      rabbits.push(...newRabbits);
    }

    function updateFoxes() {
      const cost = Number(foxCost.value);
      const power = Number(huntPower.value);
      const newFoxes = [];
      const eaten = new Set();

      for (const f of foxes) {
        f.age++;
        const prey = nearest(f, rabbits, 120);
        moveAnimal(f, 1.75, null, prey);
        f.energy -= cost;

        for (let i = 0; i < rabbits.length; i++) {
          if (eaten.has(i)) continue;
          const r = rabbits[i];
          const catchRadius = 9 + power * 5;
          if (dist2(f, r) < catchRadius * catchRadius) {
            eaten.add(i);
            f.energy += 46;
            break;
          }
        }

        if (f.energy > 155 && Math.random() < 0.018 && foxes.length + newFoxes.length < 90) {
          f.energy *= 0.62;
          newFoxes.push(createFox(f.x + rand(-12, 12), f.y + rand(-12, 12)));
        }
      }

      rabbits = rabbits.filter((_, i) => !eaten.has(i));
      foxes = foxes.filter(f => f.energy > 0 && f.age < 3600);
      foxes.push(...newFoxes);
    }

    function updateHistory() {
      if (tick % 5 !== 0) return;
      history.push({ r: rabbits.length, f: foxes.length });
      if (history.length > 360) history.shift();
    }

    function drawGrass() {
      const cellW = W / GRID;
      const cellH = worldH / GRID;
      for (let y = 0; y < GRID; y++) {
        for (let x = 0; x < GRID; x++) {
          const v = grass[y][x];
          ctx.fillStyle = `rgba(34, 197, 94, ${0.08 + v * 0.38})`;
          ctx.fillRect(x * cellW, y * cellH, cellW + 0.5, cellH + 0.5);
        }
      }
    }

    function drawRabbit(r) {
      ctx.save();
      ctx.translate(r.x, r.y);
      const ang = Math.atan2(r.vy, r.vx);
      ctx.rotate(ang);
      ctx.fillStyle = "rgba(255,255,255,0.95)";
      ctx.strokeStyle = "rgba(71,85,105,0.8)";
      ctx.lineWidth = 1.2;
      ctx.beginPath();
      ctx.ellipse(0, 0, 7, 5, 0, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      ctx.beginPath();
      ctx.ellipse(5, -4, 2, 6, 0.4, 0, Math.PI * 2);
      ctx.ellipse(5, 4, 2, 6, -0.4, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      ctx.restore();
    }

    function drawFox(f) {
      ctx.save();
      ctx.translate(f.x, f.y);
      const ang = Math.atan2(f.vy, f.vx);
      ctx.rotate(ang);
      ctx.fillStyle = "#ef4444";
      ctx.strokeStyle = "#7f1d1d";
      ctx.lineWidth = 1.2;
      ctx.beginPath();
      ctx.moveTo(10, 0);
      ctx.lineTo(-7, -7);
      ctx.lineTo(-4, 0);
      ctx.lineTo(-7, 7);
      ctx.closePath();
      ctx.fill();
      ctx.stroke();
      ctx.fillStyle = "#fecaca";
      ctx.beginPath();
      ctx.moveTo(-7, -5);
      ctx.lineTo(-16, 0);
      ctx.lineTo(-7, 5);
      ctx.closePath();
      ctx.fill();
      ctx.restore();
    }

    function drawGraph() {
      const x0 = 18;
      const y0 = worldH + 18;
      const gw = W - 36;
      const gh = graphH - 28;

      ctx.save();
      ctx.fillStyle = "rgba(255,255,255,0.84)";
      ctx.strokeStyle = "rgba(148,163,184,0.65)";
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.roundRect(x0, y0, gw, gh, 12);
      ctx.fill();
      ctx.stroke();

      ctx.fillStyle = "#334155";
      ctx.font = "13px system-ui";
      ctx.fillText("個体数の時間変化", x0 + 12, y0 + 22);
      ctx.fillStyle = "#64748b";
      ctx.fillText("白線：ウサギ　赤線：キツネ", x0 + 140, y0 + 22);

      if (history.length > 2) {
        const maxR = Math.max(80, ...history.map(h => h.r));
        const maxF = Math.max(20, ...history.map(h => h.f));

        function plot(key, color, maxVal) {
          ctx.strokeStyle = color;
          ctx.lineWidth = 2;
          ctx.beginPath();
          history.forEach((h, i) => {
            const x = x0 + 12 + (i / (history.length - 1)) * (gw - 24);
            const y = y0 + gh - 12 - (h[key] / maxVal) * (gh - 44);
            if (i === 0) ctx.moveTo(x, y);
            else ctx.lineTo(x, y);
          });
          ctx.stroke();
        }

        plot("r", "rgba(255,255,255,0.95)", maxR);
        plot("f", "rgba(239,68,68,0.95)", maxF);
      }
      ctx.restore();
    }

    function drawInfo() {
      rabbitCountEl.textContent = rabbits.length;
      foxCountEl.textContent = foxes.length;
      grassValue.textContent = Number(grassRate.value).toFixed(3);
      rabbitBirthValue.textContent = Number(rabbitBirth.value).toFixed(3);
      huntValue.textContent = Number(huntPower.value).toFixed(1);
      foxCostValue.textContent = Number(foxCost.value).toFixed(2);

      ctx.save();
      ctx.fillStyle = "rgba(15,23,42,0.72)";
      ctx.font = "13px system-ui";
      ctx.fillText("草が多い → ウサギが増える → キツネが増える → ウサギが減る → キツネも減る", 18, 24);
      ctx.restore();
    }

    function step() {
      if (!paused) {
        updateGrass();
        updateRabbits();
        updateFoxes();
        updateHistory();
        tick++;

        // 全滅したときの自動復旧。観察しやすくするため。
        if (rabbits.length < 3 && Math.random() < 0.03) {
          for (let i = 0; i < 20; i++) rabbits.push(createRabbit());
        }
        if (foxes.length < 1 && rabbits.length > 45 && Math.random() < 0.02) {
          for (let i = 0; i < 4; i++) foxes.push(createFox());
        }
      }

      ctx.clearRect(0, 0, W, H);
      drawGrass();
      rabbits.forEach(drawRabbit);
      foxes.forEach(drawFox);
      drawGraph();
      drawInfo();

      requestAnimationFrame(step);
    }

    pauseBtn.addEventListener("click", () => {
      paused = !paused;
      pauseBtn.textContent = paused ? "再開" : "一時停止";
    });

    resetBtn.addEventListener("click", reset);
    window.addEventListener("resize", resize);

    resize();
    reset();
    step();
  </script>
