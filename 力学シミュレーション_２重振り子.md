# 🔵 二重振り子とは？

image

👉 二重振り子とは
「振り子の先に、さらに振り子がついたもの」です。

* 上の棒：1つ目の振り子
* 下の棒：2つ目の振り子

---

# 🔵 普通の振り子との違い

### ① 単振り子（普通）

* 動きは周期的（同じ動きを繰り返す）
* 未来は予測できる

### ② 二重振り子

* 動きがめちゃくちゃになる
* 少しの違いで全然違う動きになる

👉 これが「カオス」です

---

# 🔵 なぜ難しくなるの？

ポイントはこれ👇

## ① 自由度が増える

* 単振り子 → 角度1つ（θ）
* 二重振り子 → 角度2つ（θ₁, θ₂）

👉 状態が一気に複雑になる

---

## ② 相互作用がある

* 上の振り子が下を引っ張る
* 下の振り子も上に影響する

👉 「非線形」になる（足し算じゃない）

---

## ③ エネルギーのやり取り

* 上 ↔ 下 でエネルギー交換
* 回転・揺れが混ざる

---

# 🔵 一番重要：初期値敏感性

例えば：

* 角度をほんの0.001°変える
* → 数秒後には全く違う動き

👉 これを

👉 **「初期条件への鋭敏な依存」**
（バタフライ効果）

と言います

これはカオスの本質で、文献でも強調されている特徴です 

---

# 🔵 実際の運動の特徴

二重振り子はこんな動きをします：

* 最初は普通に揺れる
* 途中で回転し始める
* 突然逆方向に動く
* パターンが読めない

👉 でも「完全なランダム」ではない
👉 法則はちゃんとある

---

# 🔵 数式的にはどうなってる？

実は運動方程式はこういう感じ：

* sin(θ)
* θ̇²（角速度の2乗）
* 分母に複雑な関数

👉 非線形微分方程式

つまり：

❌ 解けない（解析解がない）
⭕ 数値シミュレーションが必要

---

# 🔵 シミュレーションの意味

二重振り子は

👉 「理論では解けない → コンピュータで再現」

の代表例

---

## シミュレーションでやること

1. 初期角度を決める
2. 微分方程式を数値的に解く
3. 時間ごとの位置を計算

---

## よく使う方法

* オイラー法（簡単）
* ルンゲクッタ法（高精度）

# 🔵 なぜ重要なの？

二重振り子は実はすごく重要👇

---

## ① カオスの入門モデル

* ロジスティック写像と並ぶ基本例

---

## ② 天気予報と同じ構造

* 小さな誤差 → 大きな違い

---

## ③ 宇宙にも関係

## ④ 情報・エントロピーとも関係

<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");
const lambdaText = document.getElementById("lambda");

const originX = canvas.width / 2;
const originY = 180;

// 物理パラメータ
const g = 9.8;
const L1 = 140;
const L2 = 140;
const m1 = 1.0;
const m2 = 1.0;

// 数値積分の時間刻み
const dt = 0.01;

// 本体の状態 [theta1, omega1, theta2, omega2]
let state;

// ほんの少しだけ違う影の振り子
let shadowState;

// 初期距離
let delta0 = 1e-8;

// Lyapunov指数計算用
let lyapunovSum = 0;
let time = 0;

// 軌跡保存
let trail = [];

function reset() {
  state = [
    Math.PI / 2,
    0,
    Math.PI / 2 + 0.01,
    0
  ];

  shadowState = [
    state[0] + delta0,
    state[1],
    state[2],
    state[3]
  ];

  lyapunovSum = 0;
  time = 0;
  trail = [];
  lambdaText.textContent = "0.0000";
}

reset();

function derivatives(s) {
  const [theta1, omega1, theta2, omega2] = s;

  const delta = theta2 - theta1;

  const den1 = (m1 + m2) * L1 - m2 * L1 * Math.cos(delta) * Math.cos(delta);
  const den2 = (L2 / L1) * den1;

  const domega1 =
    (
      m2 * L1 * omega1 * omega1 * Math.sin(delta) * Math.cos(delta)
      + m2 * g * Math.sin(theta2) * Math.cos(delta)
      + m2 * L2 * omega2 * omega2 * Math.sin(delta)
      - (m1 + m2) * g * Math.sin(theta1)
    ) / den1;

  const domega2 =
    (
      -m2 * L2 * omega2 * omega2 * Math.sin(delta) * Math.cos(delta)
      + (m1 + m2) * g * Math.sin(theta1) * Math.cos(delta)
      - (m1 + m2) * L1 * omega1 * omega1 * Math.sin(delta)
      - (m1 + m2) * g * Math.sin(theta2)
    ) / den2;

  return [
    omega1,
    domega1,
    omega2,
    domega2
  ];
}

// 4次のRunge-Kutta法
function rk4Step(s, dt) {
  const k1 = derivatives(s);
  const k2 = derivatives(s.map((v, i) => v + 0.5 * dt * k1[i]));
  const k3 = derivatives(s.map((v, i) => v + 0.5 * dt * k2[i]));
  const k4 = derivatives(s.map((v, i) => v + dt * k3[i]));

  return s.map((v, i) =>
    v + dt / 6 * (k1[i] + 2 * k2[i] + 2 * k3[i] + k4[i])
  );
}

function distance(a, b) {
  let sum = 0;
  for (let i = 0; i < a.length; i++) {
    let d = a[i] - b[i];

    // 角度差を -π〜π に補正
    if (i === 0 || i === 2) {
      d = Math.atan2(Math.sin(d), Math.cos(d));
    }

    sum += d * d;
  }
  return Math.sqrt(sum);
}

function normalizeShadow() {
  const d = distance(state, shadowState);

  if (d === 0) return;

  lyapunovSum += Math.log(d / delta0);

  for (let i = 0; i < state.length; i++) {
    let diff = shadowState[i] - state[i];

    if (i === 0 || i === 2) {
      diff = Math.atan2(Math.sin(diff), Math.cos(diff));
    }

    shadowState[i] = state[i] + diff * (delta0 / d);
  }
}

function position(s) {
  const [theta1, , theta2] = s;

  const x1 = originX + L1 * Math.sin(theta1);
  const y1 = originY + L1 * Math.cos(theta1);

  const x2 = x1 + L2 * Math.sin(theta2);
  const y2 = y1 + L2 * Math.cos(theta2);

  return { x1, y1, x2, y2 };
}

function drawPendulum(s, color) {
  const { x1, y1, x2, y2 } = position(s);

  ctx.strokeStyle = color;
  ctx.lineWidth = 3;

  ctx.beginPath();
  ctx.moveTo(originX, originY);
  ctx.lineTo(x1, y1);
  ctx.lineTo(x2, y2);
  ctx.stroke();

  ctx.fillStyle = color;

  ctx.beginPath();
  ctx.arc(x1, y1, 10, 0, Math.PI * 2);
  ctx.fill();

  ctx.beginPath();
  ctx.arc(x2, y2, 12, 0, Math.PI * 2);
  ctx.fill();
}

function drawTrail() {
  ctx.strokeStyle = "rgba(0, 180, 255, 0.7)";
  ctx.lineWidth = 1;

  ctx.beginPath();
  for (let i = 0; i < trail.length; i++) {
    const p = trail[i];
    if (i === 0) ctx.moveTo(p.x, p.y);
    else ctx.lineTo(p.x, p.y);
  }
  ctx.stroke();
}

function update() {
  // 1フレーム内で複数回進めると滑らか
  for (let i = 0; i < 5; i++) {
    state = rk4Step(state, dt);
    shadowState = rk4Step(shadowState, dt);

    normalizeShadow();

    time += dt;
  }

  const p = position(state);
  trail.push({ x: p.x2, y: p.y2 });
  if (trail.length > 1500) trail.shift();

  const lambda = lyapunovSum / time;
  lambdaText.textContent = lambda.toFixed(4);
}

function draw() {
  ctx.clearRect(0, 0, canvas.width, canvas.height);

  drawTrail();

  // 影の振り子
  drawPendulum(shadowState, "rgba(255, 80, 80, 0.45)");

  // 本体
  drawPendulum(state, "#ffffff");

  ctx.fillStyle = "white";
  ctx.beginPath();
  ctx.arc(originX, originY, 5, 0, Math.PI * 2);
  ctx.fill();
}

function animate() {
  update();
  draw();
  requestAnimationFrame(animate);
}

animate();
</script>

ポイントはここです。

```js
lyapunovSum += Math.log(d / delta0);
lambda = lyapunovSum / time;
```

これは、

[
\lambda \approx \frac{1}{t}\sum \log\frac{\delta(t)}{\delta_0}
]

を計算しています。

値が **正** なら、近い初期条件が指数関数的に離れていくので、カオス的です。

<script>
const simCanvas = document.getElementById("sim");
const simCtx = simCanvas.getContext("2d");

const mapCanvas = document.getElementById("map");
const mapCtx = mapCanvas.getContext("2d");

const originX = simCanvas.width / 2;
const originY = 170;

const g = 9.8;
const L1 = 120;
const L2 = 120;
const m1 = 1.0;
const m2 = 1.0;

const dt = 0.005;

// state = [theta1, omega1, theta2, omega2]
let state = [
  Math.PI / 2,
  0.0,
  Math.PI / 2 + 0.3,
  0.0
];

let prevTheta2 = normalizeAngle(state[2]);

let points = [];

function normalizeAngle(a) {
  return Math.atan2(Math.sin(a), Math.cos(a));
}

function derivatives(s) {
  const [theta1, omega1, theta2, omega2] = s;
  const delta = theta2 - theta1;

  const den1 =
    (m1 + m2) * L1 -
    m2 * L1 * Math.cos(delta) * Math.cos(delta);

  const den2 = (L2 / L1) * den1;

  const domega1 =
    (
      m2 * L1 * omega1 * omega1 * Math.sin(delta) * Math.cos(delta)
      + m2 * g * Math.sin(theta2) * Math.cos(delta)
      + m2 * L2 * omega2 * omega2 * Math.sin(delta)
      - (m1 + m2) * g * Math.sin(theta1)
    ) / den1;

  const domega2 =
    (
      -m2 * L2 * omega2 * omega2 * Math.sin(delta) * Math.cos(delta)
      + (m1 + m2) * g * Math.sin(theta1) * Math.cos(delta)
      - (m1 + m2) * L1 * omega1 * omega1 * Math.sin(delta)
      - (m1 + m2) * g * Math.sin(theta2)
    ) / den2;

  return [omega1, domega1, omega2, domega2];
}

function rk4Step(s, dt) {
  const k1 = derivatives(s);
  const k2 = derivatives(s.map((v, i) => v + 0.5 * dt * k1[i]));
  const k3 = derivatives(s.map((v, i) => v + 0.5 * dt * k2[i]));
  const k4 = derivatives(s.map((v, i) => v + dt * k3[i]));

  return s.map((v, i) =>
    v + dt / 6 * (k1[i] + 2 * k2[i] + 2 * k3[i] + k4[i])
  );
}

function position(s) {
  const [theta1, , theta2] = s;

  const x1 = originX + L1 * Math.sin(theta1);
  const y1 = originY + L1 * Math.cos(theta1);

  const x2 = x1 + L2 * Math.sin(theta2);
  const y2 = y1 + L2 * Math.cos(theta2);

  return { x1, y1, x2, y2 };
}

function checkPoincareCrossing(prev, current, s) {
  // θ₂ = 0 を負から正へ通過
  if (prev < 0 && current >= 0 && s[3] > 0) {
    const theta1 = normalizeAngle(s[0]);
    const omega1 = s[1];

    points.push({ theta1, omega1 });

    if (points.length > 10000) points.shift();
  }
}

function drawSimulation() {
  simCtx.clearRect(0, 0, simCanvas.width, simCanvas.height);

  const { x1, y1, x2, y2 } = position(state);

  simCtx.strokeStyle = "white";
  simCtx.lineWidth = 3;

  simCtx.beginPath();
  simCtx.moveTo(originX, originY);
  simCtx.lineTo(x1, y1);
  simCtx.lineTo(x2, y2);
  simCtx.stroke();

  simCtx.fillStyle = "white";
  simCtx.beginPath();
  simCtx.arc(x1, y1, 9, 0, Math.PI * 2);
  simCtx.fill();

  simCtx.beginPath();
  simCtx.arc(x2, y2, 11, 0, Math.PI * 2);
  simCtx.fill();

  simCtx.fillStyle = "red";
  simCtx.beginPath();
  simCtx.arc(originX, originY, 5, 0, Math.PI * 2);
  simCtx.fill();

  simCtx.fillStyle = "white";
  simCtx.fillText("二重振り子", 20, 30);
}

function drawMap() {
  mapCtx.fillStyle = "rgba(24, 24, 24, 0.25)";
  mapCtx.fillRect(0, 0, mapCanvas.width, mapCanvas.height);

  drawAxes();

  for (const p of points) {
    const x = mapCanvas.width / 2 + (p.theta1 / Math.PI) * 220;
    const y = mapCanvas.height / 2 - (p.omega1 / 8) * 220;

    if (x >= 0 && x < mapCanvas.width && y >= 0 && y < mapCanvas.height) {
      mapCtx.fillStyle = "cyan";
      mapCtx.fillRect(x, y, 2, 2);
    }
  }

  mapCtx.fillStyle = "white";
  mapCtx.fillText("Poincare map: (θ₁, ω₁)", 20, 30);
  mapCtx.fillText("θ₁", 465, 255);
  mapCtx.fillText("ω₁", 255, 20);
  mapCtx.fillText("points: " + points.length, 20, 50);
}

function drawAxes() {
  mapCtx.strokeStyle = "rgba(255,255,255,0.35)";
  mapCtx.lineWidth = 1;

  mapCtx.beginPath();
  mapCtx.moveTo(0, mapCanvas.height / 2);
  mapCtx.lineTo(mapCanvas.width, mapCanvas.height / 2);
  mapCtx.moveTo(mapCanvas.width / 2, 0);
  mapCtx.lineTo(mapCanvas.width / 2, mapCanvas.height);
  mapCtx.stroke();

  mapCtx.strokeStyle = "rgba(255,255,255,0.15)";
  for (let i = -3; i <= 3; i++) {
    const x = mapCanvas.width / 2 + (i / 3) * 220;
    mapCtx.beginPath();
    mapCtx.moveTo(x, 0);
    mapCtx.lineTo(x, mapCanvas.height);
    mapCtx.stroke();
  }
}

function update() {
  for (let i = 0; i < 20; i++) {
    const oldTheta2 = normalizeAngle(state[2]);

    state = rk4Step(state, dt);

    state[0] = normalizeAngle(state[0]);
    state[2] = normalizeAngle(state[2]);

    const newTheta2 = normalizeAngle(state[2]);

    checkPoincareCrossing(oldTheta2, newTheta2, state);

    prevTheta2 = newTheta2;
  }
}

function animate() {
  update();
  drawSimulation();
  drawMap();
  requestAnimationFrame(animate);
}

animate();
</script>
