  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const temperatureSlider = document.getElementById("temperature");
    const fieldSlider = document.getElementById("field");
    const magnetPowerSlider = document.getElementById("magnetPower");
    const noiseSlider = document.getElementById("noise");
    const couplingSlider = document.getElementById("coupling");

    // -------------------------
    // Ising model
    // -------------------------
    const N = 56;
    let spins = [];

    function initIsing() {
      spins = Array.from({ length: N }, () =>
        Array.from({ length: N }, () => Math.random() < 0.5 ? 1 : -1)
      );
    }

    function randomizeIsing() {
      initIsing();
    }

    function alignIsing(v) {
      spins = Array.from({ length: N }, () => Array.from({ length: N }, () => v));
    }

    function isingStep() {
      const T = Number(temperatureSlider.value);
      const h = Number(fieldSlider.value);
      const J = 1;

      for (let k = 0; k < N * N * 0.35; k++) {
        const i = Math.floor(Math.random() * N);
        const j = Math.floor(Math.random() * N);
        const s = spins[i][j];
        const up = spins[(i - 1 + N) % N][j];
        const down = spins[(i + 1) % N][j];
        const left = spins[i][(j - 1 + N) % N];
        const right = spins[i][(j + 1) % N];
        const neighborSum = up + down + left + right;

        // ΔE = 2 s (J Σneighbor + h)
        const dE = 2 * s * (J * neighborSum + h);
        if (dE <= 0 || Math.random() < Math.exp(-dE / T)) {
          spins[i][j] = -s;
        }
      }
    }

    function isingMagnetization() {
      let sum = 0;
      for (let i = 0; i < N; i++) {
        for (let j = 0; j < N; j++) sum += spins[i][j];
      }
      return sum / (N * N);
    }

    // -------------------------
    // Bar magnets
    // -------------------------
    const bars = [];
    const barArea = { x: 480, y: 95, w: 360, h: 440 };
    const externalMagnet = { x: 730, y: 320, angle: Math.PI, dragging: false };
    let showField = true;

    function initBars() {
      bars.length = 0;
      const cols = 11;
      const rows = 11;
      for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
          bars.push({
            x: barArea.x + 28 + x * ((barArea.w - 56) / (cols - 1)),
            y: barArea.y + 28 + y * ((barArea.h - 56) / (rows - 1)),
            angle: Math.random() * Math.PI * 2,
            omega: 0,
            len: 20
          });
        }
      }
    }

    function randomizeBars() {
      for (const b of bars) {
        b.angle = Math.random() * Math.PI * 2;
        b.omega = 0;
      }
    }

    function angleDiff(a, b) {
      let d = a - b;
      while (d > Math.PI) d -= Math.PI * 2;
      while (d < -Math.PI) d += Math.PI * 2;
      return d;
    }

    function vecFromAngle(a) {
      return { x: Math.cos(a), y: Math.sin(a) };
    }

    function externalFieldAt(x, y) {
      // 簡易双極子風：外部磁石の向きと位置から磁場っぽいベクトルを作る
      const dx = x - externalMagnet.x;
      const dy = y - externalMagnet.y;
      const r2 = dx * dx + dy * dy + 900;
      const r = Math.sqrt(r2);
      const nx = dx / r;
      const ny = dy / r;
      const m = vecFromAngle(externalMagnet.angle);
      const mdotr = m.x * nx + m.y * ny;
      const strength = Number(magnetPowerSlider.value) * 52000 / (r2 * r);

      // dipole-like field: 3(m·r)r - m
      return {
        x: strength * (3 * mdotr * nx - m.x),
        y: strength * (3 * mdotr * ny - m.y)
      };
    }

    function neighborFieldAt(index) {
      const target = bars[index];
      let fx = 0;
      let fy = 0;
      const coupling = Number(couplingSlider.value);
      for (let k = 0; k < bars.length; k++) {
        if (k === index) continue;
        const b = bars[k];
        const dx = b.x - target.x;
        const dy = b.y - target.y;
        const r2 = dx * dx + dy * dy + 1200;
        if (r2 > 5000) continue;
        const v = vecFromAngle(b.angle);
        const s = coupling * 18 / r2;
        fx += v.x * s;
        fy += v.y * s;
      }
      return { x: fx, y: fy };
    }

    function updateBars() {
      const noise = Number(noiseSlider.value);
      for (let i = 0; i < bars.length; i++) {
        const b = bars[i];
        const ef = externalFieldAt(b.x, b.y);
        const nf = neighborFieldAt(i);
        const fx = ef.x + nf.x;
        const fy = ef.y + nf.y;

        const targetAngle = Math.atan2(fy, fx);
        const diff = angleDiff(targetAngle, b.angle);
        const torque = Math.sin(diff);

        b.omega += torque * 0.055;
        b.omega += (Math.random() - 0.5) * noise * 0.16;
        b.omega *= 0.90;
        b.angle += b.omega;
      }
    }

    function barAlignment() {
      let sx = 0;
      let sy = 0;
      for (const b of bars) {
        sx += Math.cos(b.angle);
        sy += Math.sin(b.angle);
      }
      return Math.sqrt(sx * sx + sy * sy) / bars.length;
    }

    // -------------------------
    // Drawing
    // -------------------------
    function drawIsing() {
      const x0 = 38;
      const y0 = 95;
      const size = 392;
      const cell = size / N;

      ctx.fillStyle = "#111827";
      ctx.font = "18px system-ui";
      ctx.fillText("イジングモデル：温度で磁化が変わる", x0, 42);
      ctx.font = "14px system-ui";
      ctx.fillStyle = "#475569";
      ctx.fillText("赤=上向きスピン、青=下向きスピン。低温では同じ向きにそろいやすい。", x0, 66);

      for (let i = 0; i < N; i++) {
        for (let j = 0; j < N; j++) {
          ctx.fillStyle = spins[i][j] === 1 ? "#ef4444" : "#2563eb";
          ctx.fillRect(x0 + j * cell, y0 + i * cell, cell + 0.3, cell + 0.3);
        }
      }

      ctx.strokeStyle = "#334155";
      ctx.lineWidth = 3;
      ctx.strokeRect(x0, y0, size, size);

      const M = isingMagnetization();
      drawMagnetizationMeter(x0, y0 + size + 36, size, M);
    }

    function drawMagnetizationMeter(x, y, w, M) {
      ctx.fillStyle = "#e5e7eb";
      ctx.fillRect(x, y, w, 22);
      ctx.fillStyle = M >= 0 ? "#ef4444" : "#2563eb";
      const center = x + w / 2;
      ctx.fillRect(center, y, (w / 2) * M, 22);
      ctx.strokeStyle = "#334155";
      ctx.strokeRect(x, y, w, 22);
      ctx.fillStyle = "#111827";
      ctx.font = "13px system-ui";
      ctx.fillText("磁化 M：左が下向き優勢、右が上向き優勢", x, y - 8);
      ctx.fillText(M.toFixed(3), center - 16, y + 43);
    }

    function drawFieldLines() {
      if (!showField) return;
      const step = 42;
      for (let y = barArea.y + 22; y < barArea.y + barArea.h; y += step) {
        for (let x = barArea.x + 22; x < barArea.x + barArea.w; x += step) {
          const f = externalFieldAt(x, y);
          const len = Math.min(24, Math.sqrt(f.x * f.x + f.y * f.y) * 20);
          const a = Math.atan2(f.y, f.x);
          drawArrow(x, y, a, len, "rgba(99,102,241,0.34)");
        }
      }
    }

    function drawArrow(x, y, angle, len, color) {
      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(angle);
      ctx.strokeStyle = color;
      ctx.fillStyle = color;
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(-len * 0.4, 0);
      ctx.lineTo(len * 0.6, 0);
      ctx.stroke();
      ctx.beginPath();
      ctx.moveTo(len * 0.6, 0);
      ctx.lineTo(len * 0.25, -5);
      ctx.lineTo(len * 0.25, 5);
      ctx.closePath();
      ctx.fill();
      ctx.restore();
    }

    function drawBarMagnet(x, y, angle, len, width = 8, alpha = 1) {
      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(angle);

      ctx.fillStyle = `rgba(220,38,38,${alpha})`;
      ctx.fillRect(0, -width / 2, len / 2, width);
      ctx.fillStyle = `rgba(37,99,235,${alpha})`;
      ctx.fillRect(-len / 2, -width / 2, len / 2, width);
      ctx.strokeStyle = `rgba(17,24,39,${0.45 * alpha})`;
      ctx.lineWidth = 1.5;
      ctx.strokeRect(-len / 2, -width / 2, len, width);

      ctx.fillStyle = `rgba(255,255,255,${0.9 * alpha})`;
      ctx.font = "7px system-ui";
      ctx.fillText("N", len / 4 - 3, 2.5);
      ctx.fillText("S", -len / 4 - 3, 2.5);
      ctx.restore();
    }

    function drawBarsSimulation() {
      ctx.fillStyle = "#111827";
      ctx.font = "18px system-ui";
      ctx.fillText("小さい棒磁石の集団：外部磁石を近づける", 480, 42);
      ctx.font = "14px system-ui";
      ctx.fillStyle = "#475569";
      ctx.fillText("外部磁石はドラッグで移動。棒磁石は磁場にそろいながらランダムに揺れる。", 480, 66);

      ctx.fillStyle = "rgba(248,250,252,0.78)";
      ctx.strokeStyle = "#cbd5e1";
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.roundRect(barArea.x, barArea.y, barArea.w, barArea.h, 18);
      ctx.fill();
      ctx.stroke();

      drawFieldLines();

      for (const b of bars) {
        const f = externalFieldAt(b.x, b.y);
        const mag = Math.min(1, Math.sqrt(f.x * f.x + f.y * f.y) * 2.5);
        drawBarMagnet(b.x, b.y, b.angle, b.len, 7, 0.55 + mag * 0.45);
      }

      // external magnet
      drawBarMagnet(externalMagnet.x, externalMagnet.y, externalMagnet.angle, 82, 24, 1);
      ctx.fillStyle = "#111827";
      ctx.font = "13px system-ui";
      ctx.fillText("外部磁石", externalMagnet.x - 32, externalMagnet.y - 34);

      // average direction arrow
      let sx = 0;
      let sy = 0;
      for (const b of bars) {
        sx += Math.cos(b.angle);
        sy += Math.sin(b.angle);
      }
      const a = Math.atan2(sy, sx);
      const align = barAlignment();
      drawArrow(658, 575, a, 30 + align * 80, "rgba(124,58,237,0.8)");
      ctx.fillStyle = "#111827";
      ctx.font = "13px system-ui";
      ctx.fillText("集団全体の磁力方向", 590, 612);
    }

    function drawCriticalLine() {
      const T = Number(temperatureSlider.value);
      const x = 38;
      const y = 590;
      const w = 392;
      ctx.fillStyle = "#e5e7eb";
      ctx.fillRect(x, y, w, 12);
      ctx.fillStyle = "#f97316";
      const tx = x + ((T - 0.5) / (5.0 - 0.5)) * w;
      ctx.fillRect(tx - 2, y - 5, 4, 22);
      ctx.strokeStyle = "#334155";
      ctx.strokeRect(x, y, w, 12);
      const crit = x + ((2.27 - 0.5) / (5.0 - 0.5)) * w;
      ctx.strokeStyle = "#111827";
      ctx.beginPath();
      ctx.moveTo(crit, y - 8);
      ctx.lineTo(crit, y + 20);
      ctx.stroke();
      ctx.fillStyle = "#111827";
      ctx.font = "12px system-ui";
      ctx.fillText("臨界温度っぽい場所 T≈2.27", crit - 64, y - 14);
    }

    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      drawIsing();
      drawCriticalLine();
      drawBarsSimulation();
    }

    function updateUI() {
      const T = Number(temperatureSlider.value);
      const h = Number(fieldSlider.value);
      const M = isingMagnetization();
      const align = barAlignment();
      const dx = externalMagnet.x - (barArea.x + barArea.w / 2);
      const dy = externalMagnet.y - (barArea.y + barArea.h / 2);
      const dist = Math.sqrt(dx * dx + dy * dy);

      document.getElementById("tempText").textContent = T.toFixed(2);
      document.getElementById("fieldText").textContent = h.toFixed(2);
      document.getElementById("magnetPowerText").textContent = Number(magnetPowerSlider.value).toFixed(2);
      document.getElementById("noiseText").textContent = Number(noiseSlider.value).toFixed(2);
      document.getElementById("couplingText").textContent = Number(couplingSlider.value).toFixed(2);
      document.getElementById("isingMagStat").textContent = M.toFixed(3);
      document.getElementById("barAlignStat").textContent = `${(align * 100).toFixed(0)} %`;
      document.getElementById("distanceStat").textContent = `${dist.toFixed(0)} px`;

      let phase = "ランダム";
      if (Math.abs(M) > 0.65) phase = "強く磁化";
      else if (Math.abs(M) > 0.25) phase = "弱く磁化";
      document.getElementById("phaseStat").textContent = phase;
    }

    function loop() {
      for (let i = 0; i < 2; i++) isingStep();
      updateBars();
      updateUI();
      draw();
      requestAnimationFrame(loop);
    }

    function getMousePos(e) {
      const rect = canvas.getBoundingClientRect();
      const scaleX = canvas.width / rect.width;
      const scaleY = canvas.height / rect.height;
      return {
        x: (e.clientX - rect.left) * scaleX,
        y: (e.clientY - rect.top) * scaleY
      };
    }

    canvas.addEventListener("pointerdown", e => {
      const p = getMousePos(e);
      const dx = p.x - externalMagnet.x;
      const dy = p.y - externalMagnet.y;
      if (Math.sqrt(dx * dx + dy * dy) < 60) {
        externalMagnet.dragging = true;
        canvas.setPointerCapture(e.pointerId);
      }
    });

    canvas.addEventListener("pointermove", e => {
      const p = getMousePos(e);
      if (externalMagnet.dragging) {
        externalMagnet.x = p.x;
        externalMagnet.y = p.y;
      }
    });

    canvas.addEventListener("pointerup", e => {
      externalMagnet.dragging = false;
      try { canvas.releasePointerCapture(e.pointerId); } catch (_) {}
    });

    canvas.addEventListener("wheel", e => {
      e.preventDefault();
      externalMagnet.angle += e.deltaY > 0 ? 0.15 : -0.15;
    }, { passive: false });

    initIsing();
    initBars();
    loop();
  </script>
