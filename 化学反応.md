<script>
const canvas = document.getElementById('canvas');
const ctx = canvas.getContext('2d');
const modeEl = document.getElementById('mode');
const tempEl = document.getElementById('temp');
const eaEl = document.getElementById('ea');
const concEl = document.getElementById('conc');
const enzymeCountEl = document.getElementById('enzymeCount');
const enzymeControl = document.getElementById('enzymeControl');
const formulaEl = document.getElementById('formula');
const legendEl = document.getElementById('legend');

const tempText = document.getElementById('tempText');
const eaText = document.getElementById('eaText');
const concText = document.getElementById('concText');
const enzymeText = document.getElementById('enzymeText');
const reactionCountEl = document.getElementById('reactionCount');
const rateStat = document.getElementById('rateStat');
const substrateStat = document.getElementById('substrateStat');
const productStat = document.getElementById('productStat');
const startBtn = document.getElementById('startBtn');
const resetBtn = document.getElementById('resetBtn');

let particles = [];
let running = true;
let reactionCount = 0;
let lastReactionCount = 0;
let lastRateTime = performance.now();
let currentRate = 0;

function rand(a,b){ return a + Math.random()*(b-a); }
function clamp(x,a,b){ return Math.max(a, Math.min(b, x)); }
function dist(a,b){ return Math.hypot(a.x-b.x, a.y-b.y); }

function color(type){
  return { A:'#2563eb', B:'#dc2626', C:'#7c3aed', S:'#f97316', E:'#16a34a', ES:'#0f766e', P:'#9333ea' }[type] || '#111827';
}

function makeParticle(type){
  const speed = 0.25 + Number(tempEl.value)/70;
  return {
    type,
    x: rand(30, canvas.width-30),
    y: rand(30, canvas.height-30),
    vx: rand(-speed, speed),
    vy: rand(-speed, speed),
    r: type === 'E' || type === 'ES' ? 18 : 9,
    boundTime: 0,
    pulse: 0
  };
}

function reset(){
  particles = [];
  reactionCount = 0;
  lastReactionCount = 0;
  currentRate = 0;
  const n = Number(concEl.value);
  if(modeEl.value === 'normal'){
    for(let i=0;i<n;i++) particles.push(makeParticle('A'));
    for(let i=0;i<n;i++) particles.push(makeParticle('B'));
  } else {
    for(let i=0;i<n;i++) particles.push(makeParticle('S'));
    for(let i=0;i<Number(enzymeCountEl.value);i++) particles.push(makeParticle('E'));
  }
  updateLabels();
}

function reactionProbability(){
  const temp = Number(tempEl.value);
  const ea = Number(eaEl.value);
  return clamp((temp - ea + 35) / 90, 0.01, 0.9);
}

function enzymeProbability(){
  const temp = Number(tempEl.value);
  const ea = Number(eaEl.value) * 0.45;
  return clamp((temp - ea + 45) / 80, 0.08, 0.98);
}

function updateParticle(p){
  const speedFactor = 0.96 + Number(tempEl.value)/500;
  p.x += p.vx * speedFactor;
  p.y += p.vy * speedFactor;
  if(p.x < p.r || p.x > canvas.width-p.r){ p.vx *= -1; p.x = clamp(p.x, p.r, canvas.width-p.r); }
  if(p.y < p.r || p.y > canvas.height-p.r){ p.vy *= -1; p.y = clamp(p.y, p.r, canvas.height-p.r); }
  p.pulse *= 0.9;
}

function reactNormal(){
  const prob = reactionProbability();
  for(let i=0;i<particles.length;i++){
    const a = particles[i];
    if(a.type !== 'A') continue;
    for(let j=i+1;j<particles.length;j++){
      const b = particles[j];
      if(b.type !== 'B') continue;
      if(dist(a,b) < a.r + b.r + 4 && Math.random() < prob){
        const c = makeParticle('C');
        c.x = (a.x+b.x)/2; c.y = (a.y+b.y)/2;
        c.vx = (a.vx+b.vx)/2 + rand(-0.5,0.5);
        c.vy = (a.vy+b.vy)/2 + rand(-0.5,0.5);
        c.r = 12; c.pulse = 1;
        particles.splice(j,1);
        particles.splice(i,1);
        particles.push(c);
        reactionCount++;
        return;
      }
    }
  }
}

function reactEnzyme(){
  for(const e of particles){
    if(e.type === 'ES'){
      e.boundTime++;
      e.pulse = 0.6;
      if(e.boundTime > 35 && Math.random() < enzymeProbability()){
        e.type = 'E';
        e.boundTime = 0;
        const p = makeParticle('P');
        p.x = e.x + rand(-20,20); p.y = e.y + rand(-20,20);
        p.vx = rand(-2,2); p.vy = rand(-2,2); p.r = 10; p.pulse = 1;
        particles.push(p);
        reactionCount++;
      }
    }
  }

  for(const e of particles){
    if(e.type !== 'E') continue;
    for(let i=0;i<particles.length;i++){
      const s = particles[i];
      if(s.type !== 'S') continue;
      if(dist(e,s) < e.r + s.r + 6){
        e.type = 'ES';
        e.boundTime = 0;
        e.pulse = 1;
        particles.splice(i,1);
        return;
      }
    }
  }
}

function drawBackground(){
  ctx.clearRect(0,0,canvas.width,canvas.height);
  ctx.fillStyle = '#f8fafc';
  ctx.fillRect(0,0,canvas.width,canvas.height);
  ctx.strokeStyle = '#e2e8f0';
  for(let x=0;x<canvas.width;x+=40){ ctx.beginPath(); ctx.moveTo(x,0); ctx.lineTo(x,canvas.height); ctx.stroke(); }
  for(let y=0;y<canvas.height;y+=40){ ctx.beginPath(); ctx.moveTo(0,y); ctx.lineTo(canvas.width,y); ctx.stroke(); }
}

function drawParticle(p){
  ctx.save();
  const glow = p.pulse * 12;
  if(glow > 0){
    ctx.fillStyle = color(p.type) + '33';
    ctx.beginPath(); ctx.arc(p.x,p.y,p.r+glow,0,Math.PI*2); ctx.fill();
  }
  ctx.fillStyle = color(p.type);
  ctx.beginPath(); ctx.arc(p.x,p.y,p.r,0,Math.PI*2); ctx.fill();
  ctx.strokeStyle = 'rgba(15,23,42,.25)'; ctx.lineWidth = 2; ctx.stroke();
  ctx.fillStyle = 'white';
  ctx.font = `bold ${p.type.length === 2 ? 12 : 14}px system-ui`;
  ctx.textAlign = 'center'; ctx.textBaseline = 'middle';
  ctx.fillText(p.type, p.x, p.y);
  if(p.type === 'E' || p.type === 'ES'){
    ctx.strokeStyle = 'rgba(255,255,255,.75)';
    ctx.lineWidth = 3;
    ctx.beginPath();
    ctx.arc(p.x, p.y-p.r+7, 7, 0, Math.PI);
    ctx.stroke();
  }
  ctx.restore();
}

function drawEnergyDiagram(){
  const x0 = 32, y0 = canvas.height - 100, w = 210, h = 70;
  ctx.fillStyle = 'rgba(255,255,255,.82)';
  ctx.fillRect(x0-12, y0-72, w+24, h+86);
  ctx.strokeStyle = '#cbd5e1'; ctx.strokeRect(x0-12, y0-72, w+24, h+86);
  ctx.fillStyle = '#0f172a'; ctx.font = 'bold 13px system-ui';
  ctx.fillText('活性化エネルギー', x0, y0-50);

  const ea = Number(eaEl.value);
  const enzymatic = modeEl.value === 'enzyme';
  const peak = y0 - ea * 0.58;
  const peak2 = y0 - ea * 0.58 * 0.45;
  ctx.lineWidth = 4;
  ctx.strokeStyle = '#dc2626';
  ctx.beginPath(); ctx.moveTo(x0,y0); ctx.quadraticCurveTo(x0+w/2,peak,x0+w,y0-8); ctx.stroke();
  if(enzymatic){
    ctx.strokeStyle = '#16a34a';
    ctx.beginPath(); ctx.moveTo(x0,y0); ctx.quadraticCurveTo(x0+w/2,peak2,x0+w,y0-8); ctx.stroke();
  }
  ctx.fillStyle = '#475569'; ctx.font = '12px system-ui';
  ctx.fillText('赤: 酵素なし', x0, y0+18);
  if(enzymatic) ctx.fillText('緑: 酵素あり', x0+95, y0+18);
}

function updateLabels(){
  tempText.textContent = tempEl.value;
  eaText.textContent = eaEl.value;
  concText.textContent = concEl.value;
  enzymeText.textContent = enzymeCountEl.value;
  enzymeControl.style.display = modeEl.value === 'enzyme' ? 'block' : 'none';
  formulaEl.textContent = modeEl.value === 'enzyme' ? 'E + S → ES → E + P' : 'A + B → C';
  legendEl.innerHTML = modeEl.value === 'enzyme'
    ? '<span class="chip">S: 基質</span><span class="chip">E: 酵素</span><span class="chip">ES: 酵素-基質複合体</span><span class="chip">P: 生成物</span>'
    : '<span class="chip">A: 反応物</span><span class="chip">B: 反応物</span><span class="chip">C: 生成物</span>';
}

function updateStats(){
  const now = performance.now();
  if(now - lastRateTime > 1000){
    currentRate = reactionCount - lastReactionCount;
    lastReactionCount = reactionCount;
    lastRateTime = now;
  }
  const substrate = modeEl.value === 'enzyme'
    ? particles.filter(p=>p.type==='S' || p.type==='ES').length
    : particles.filter(p=>p.type==='A' || p.type==='B').length;
  const product = modeEl.value === 'enzyme'
    ? particles.filter(p=>p.type==='P').length
    : particles.filter(p=>p.type==='C').length;
  reactionCountEl.textContent = reactionCount;
  rateStat.textContent = `${currentRate} /s`;
  substrateStat.textContent = substrate;
  productStat.textContent = product;
}

function loop(){
  updateLabels();
  if(running){
    for(const p of particles) updateParticle(p);
    if(modeEl.value === 'normal') reactNormal();
    else reactEnzyme();
  }
  drawBackground();
  for(const p of particles) drawParticle(p);
  drawEnergyDiagram();
  updateStats();
  requestAnimationFrame(loop);
}

startBtn.addEventListener('click', ()=> running = !running);
resetBtn.addEventListener('click', reset);
modeEl.addEventListener('change', reset);
concEl.addEventListener('change', reset);
enzymeCountEl.addEventListener('change', reset);
[tempEl, eaEl].forEach(el => el.addEventListener('input', updateLabels));

reset();
loop();
</script>
