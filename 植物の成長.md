  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const sunlightSlider = document.getElementById("sunlight");
    const co2Slider = document.getElementById("co2");
    const o2Slider = document.getElementById("o2");

    let nutrients = 45;
    let plantSize = 1.0;
    let health = 100;
    let alive = true;
    let particles = [];
    let history = [];
    let tick = 0;

    function clamp(v, min, max) {
      return Math.max(min, Math.min(max, v));
    }

    function setDay() {
      sunlightSlider.value = 80;
    }

    function setNight() {
      sunlightSlider.value = 0;
    }

    function addNutrients() {
      if (!alive) health = Math.max(20, health);
      alive = true;
      nutrients = clamp(nutrients + 25, 0, 100);
    }

    function resetSim() {
      nutrients = 45;
      plantSize = 1.0;
      health = 100;
      alive = true;
      sunlightSlider.value = 70;
      co2Slider.value = 55;
      o2Slider.value = 45;
      particles = [];
      history = [];
    }

    function photosynthesisRate() {
      if (!alive) return 0;
      const light = Number(sunlightSlider.value) / 100;
      const co2 = Number(co2Slider.value) / 100;
      const leafFactor = Math.min(2.2, plantSize);
      const healthFactor = health / 100;
      return light * co2 * leafFactor * healthFactor;
    }

    function respirationRate() {
      if (!alive) return 0;
      const light = Number(sunlightSlider.value) / 100;
      const darkness = 1 - light;
      const o2 = Number(o2Slider.value) / 100;
      const sizeFactor = 0.4 + plantSize * 0.55;
      return (0.18 + darkness * 0.75) * o2 * sizeFactor;
    }

    function updateModel() {
      tick++;
      if (!alive) {
        health = Math.max(0, health - 0.02);
        return;
      }

      let co2 = Number(co2Slider.value);
      let o2 = Number(o2Slider.value);
      const pRate = photosynthesisRate();
      const rRate = respirationRate();

      // 光合成：CO2を消費、O2と栄養を生産
      const photo = Math.min(co2, pRate * 0.16);
      co2 -= photo;
      o2 += photo * 0.95;
      nutrients += photo * 0.75;

      // 呼吸：O2と栄養を消費、CO2を排出
      const resp = Math.min(o2, nutrients, rRate * 0.12);
      o2 -= resp;
      co2 += resp * 0.95;
      nutrients -= resp * 0.9;

      // 成長：栄養に余裕があると消費して大きくなる
      if (nutrients > 35 && health > 40) {
        const growth = Math.min(nutrients - 35, 0.018 * plantSize * (0.5 + health / 100));
        nutrients -= growth * 1.9;
        plantSize += growth * 0.018;
      }

      // 栄養不足・酸素不足・CO2不足による弱り
      if (nutrients < 8) health -= 0.16;
      else if (nutrients < 18) health -= 0.045;
      else health += 0.025;

      if (o2 < 2 && Number(sunlightSlider.value) < 10) health -= 0.08;
      if (co2 < 2 && Number(sunlightSlider.value) > 30) health -= 0.04;

      nutrients = clamp(nutrients, 0, 100);
      health = clamp(health, 0, 100);
      co2 = clamp(co2, 0, 100);
      o2 = clamp(o2, 0, 100);

      co2Slider.value = co2;
      o2Slider.value = o2;

      if (health <= 0 || nutrients <= 0.2) {
        alive = false;
        health = 0;
      }

      if (tick % 8 === 0) {
        history.push({ co2, o2, nutrients, size: plantSize, health });
        if (history.length > 110) history.shift();
      }

      makeParticles(photo, resp);
    }

    function makeParticles(photo, resp) {
      const sun = Number(sunlightSlider.value);
      if (photo > 0.01) {
        for (let i = 0; i < Math.min(8, Math.floor(photo * 25)); i++) {
          particles.push({
            type: Math.random() < 0.5 ? "co2_in" : "o2_out",
            x: Math.random() * canvas.width,
            y: 80 + Math.random() * 180,
            vx: 0,
            vy: 0,
            life: 1,
            r: 4 + Math.random() * 4
          });
        }
        for (let i = 0; i < Math.min(5, Math.floor(sun / 18)); i++) {
          particles.push({
            type: "sun",
            x: 80 + Math.random() * 260,
            y: 35,
            vx: 1.2 + Math.random() * 1.2,
            vy: 1.4 + Math.random() * 1.1,
            life: 1,
            r: 5
          });
        }
      }

      if (resp > 0.01) {
        for (let i = 0; i < Math.min(6, Math.floor(resp * 18)); i++) {
          particles.push({
            type: Math.random() < 0.5 ? "o2_in" : "co2_out",
            x: 360 + Math.random() * 180,
            y: 270 + Math.random() * 110,
            vx: 0,
            vy: 0,
            life: 1,
            r: 4 + Math.random() * 4
          });
        }
      }
      if (particles.length > 260) particles.splice(0, particles.length - 260);
    }

    function updateParticles() {
      const targetLeaf = { x: 430, y: 260 };
      for (const p of particles) {
        if (p.type === "co2_in" || p.type === "o2_in") {
          const dx = targetLeaf.x - p.x;
          const dy = targetLeaf.y - p.y;
          p.vx += dx * 0.0009;
          p.vy += dy * 0.0009;
        } else if (p.type === "o2_out" || p.type === "co2_out") {
          p.vx += (p.x < targetLeaf.x ? -0.035 : 0.035);
          p.vy -= 0.025;
        } else if (p.type === "sun") {
          // fixed velocity
        }

        p.x += p.vx;
        p.y += p.vy;
        p.vx *= 0.98;
        p.vy *= 0.98;
        p.life -= 0.008;
      }
      particles = particles.filter(p => p.life > 0 && p.x > -40 && p.x < canvas.width + 40 && p.y > -40 && p.y < canvas.height + 40);
    }

    function drawBackground() {
      const sun = Number(sunlightSlider.value) / 100;
      const skyGrad = ctx.createLinearGradient(0, 0, 0, 430);
      if (sun > 0.05) {
        skyGrad.addColorStop(0, `rgba(191, 219, 254, ${0.45 + sun * 0.55})`);
        skyGrad.addColorStop(1, "#eff6ff");
      } else {
        skyGrad.addColorStop(0, "#111827");
        skyGrad.addColorStop(1, "#334155");
      }
      ctx.fillStyle = skyGrad;
      ctx.fillRect(0, 0, canvas.width, 430);

      ctx.fillStyle = "#9a6a3a";
      ctx.fillRect(0, 430, canvas.width, 210);
      ctx.fillStyle = "rgba(80, 47, 25, 0.25)";
      for (let i = 0; i < 70; i++) {
        ctx.beginPath();
        ctx.arc((i * 97) % canvas.width, 440 + ((i * 43) % 170), 1.5, 0, Math.PI * 2);
        ctx.fill();
      }

      // sun / moon
      if (sun > 0.05) {
        ctx.fillStyle = `rgba(250, 204, 21, ${0.35 + sun * 0.65})`;
        ctx.beginPath();
        ctx.arc(92, 86, 36 + sun * 8, 0, Math.PI * 2);
        ctx.fill();
      } else {
        ctx.fillStyle = "rgba(226,232,240,0.9)";
        ctx.beginPath();
        ctx.arc(92, 86, 31, 0, Math.PI * 2);
        ctx.fill();
        ctx.fillStyle = "#111827";
        ctx.beginPath();
        ctx.arc(105, 78, 30, 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function drawPlant() {
      const baseX = 430;
      const baseY = 440;
      const s = plantSize;
      const wilt = alive ? 0 : 28;
      const healthFactor = health / 100;
      const stemColor = alive ? `rgb(${70 + (1-healthFactor)*80}, ${130 + healthFactor*70}, 60)` : "#7c5c3e";
      const leafColor = alive ? `rgb(${35 + (1-healthFactor)*90}, ${120 + healthFactor*90}, ${50})` : "#8b7355";

      ctx.lineCap = "round";
      ctx.strokeStyle = stemColor;
      ctx.lineWidth = 10 + s * 3;
      ctx.beginPath();
      ctx.moveTo(baseX, baseY);
      ctx.quadraticCurveTo(baseX + 5, baseY - 90 * s, baseX - 4, baseY - 165 * s + wilt);
      ctx.stroke();

      const leafPositions = [
        { x: -42, y: -78, a: -0.45, scale: 1.0 },
        { x: 46, y: -105, a: 0.55, scale: 1.1 },
        { x: -48, y: -138, a: -0.65, scale: 0.95 },
        { x: 42, y: -165, a: 0.7, scale: 0.85 },
        { x: 0, y: -195, a: -0.05, scale: 0.75 }
      ];

      for (const lp of leafPositions) {
        drawLeaf(baseX + lp.x * s, baseY + lp.y * s + wilt, lp.a + (alive ? Math.sin(Date.now()*0.002 + lp.x)*0.05 : 0.6), 44 * s * lp.scale, 22 * s * lp.scale, leafColor);
      }

      // roots
      ctx.strokeStyle = "#6b4423";
      ctx.lineWidth = 3;
      for (let i = -3; i <= 3; i++) {
        ctx.beginPath();
        ctx.moveTo(baseX, baseY);
        ctx.quadraticCurveTo(baseX + i * 18 * s, baseY + 22, baseX + i * 30 * s, baseY + 50 + Math.abs(i) * 5);
        ctx.stroke();
      }

      if (!alive) {
        ctx.fillStyle = "rgba(120,53,15,0.16)";
        ctx.beginPath();
        ctx.arc(baseX, baseY - 80, 120 * Math.min(1.4, s), 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function drawLeaf(x, y, angle, w, h, color) {
      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(angle);
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.ellipse(0, 0, w, h, 0, 0, Math.PI * 2);
      ctx.fill();
      ctx.strokeStyle = "rgba(22,101,52,0.45)";
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(-w * 0.75, 0);
      ctx.lineTo(w * 0.75, 0);
      ctx.stroke();
      ctx.restore();
    }

    function drawParticles() {
      for (const p of particles) {
        let color = "#fff";
        let text = "";
        if (p.type === "co2_in" || p.type === "co2_out") {
          color = "rgba(75,85,99," + (0.25 + p.life * 0.6) + ")";
          text = "CO₂";
        } else if (p.type === "o2_in" || p.type === "o2_out") {
          color = "rgba(56,189,248," + (0.25 + p.life * 0.6) + ")";
          text = "O₂";
        } else if (p.type === "sun") {
          color = "rgba(250,204,21," + (0.25 + p.life * 0.7) + ")";
          text = "光";
        }
        ctx.fillStyle = color;
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.r + 7, 0, Math.PI * 2);
        ctx.fill();
        ctx.fillStyle = "#111827";
        ctx.font = "11px system-ui";
        ctx.fillText(text, p.x - 10, p.y + 4);
      }
    }

    function drawBars() {
      const co2 = Number(co2Slider.value);
      const o2 = Number(o2Slider.value);
      const items = [
        { name: "CO₂", value: co2, color: "#4b5563", x: 30, y: 500 },
        { name: "O₂", value: o2, color: "#38bdf8", x: 155, y: 500 },
        { name: "栄養", value: nutrients, color: "#22c55e", x: 280, y: 500 },
        { name: "元気", value: health, color: "#f97316", x: 405, y: 500 }
      ];
      for (const it of items) {
        ctx.fillStyle = "rgba(255,255,255,0.65)";
        ctx.fillRect(it.x, it.y, 80, 100);
        ctx.strokeStyle = "#cbd5e1";
        ctx.strokeRect(it.x, it.y, 80, 100);
        ctx.fillStyle = it.color;
        ctx.fillRect(it.x, it.y + 100 - it.value, 80, it.value);
        ctx.fillStyle = "#111827";
        ctx.font = "13px system-ui";
        ctx.fillText(it.name, it.x + 20, it.y - 8);
        ctx.fillText(`${it.value.toFixed(0)}%`, it.x + 19, it.y + 120);
      }
    }

    function drawGraph() {
      const x = 550;
      const y = 455;
      const w = 275;
      const h = 145;
      ctx.fillStyle = "rgba(255,255,255,0.75)";
      ctx.beginPath();
      ctx.roundRect(x, y, w, h, 12);
      ctx.fill();
      ctx.strokeStyle = "#cbd5e1";
      ctx.strokeRect(x, y, w, h);
      ctx.fillStyle = "#111827";
      ctx.font = "13px system-ui";
      ctx.fillText("変化の履歴", x + 10, y + 20);

      drawLine(history.map(d => d.co2), x, y, w, h, "#4b5563");
      drawLine(history.map(d => d.o2), x, y, w, h, "#38bdf8");
      drawLine(history.map(d => d.nutrients), x, y, w, h, "#22c55e");
    }

    function drawLine(values, x, y, w, h, color) {
      if (values.length < 2) return;
      ctx.strokeStyle = color;
      ctx.lineWidth = 2;
      ctx.beginPath();
      values.forEach((v, i) => {
        const px = x + 8 + (i / (110 - 1)) * (w - 16);
        const py = y + h - 10 - (v / 100) * (h - 34);
        if (i === 0) ctx.moveTo(px, py);
        else ctx.lineTo(px, py);
      });
      ctx.stroke();
    }

    function drawEquationAndStatus() {
      const sun = Number(sunlightSlider.value);
      const pRate = photosynthesisRate();
      const rRate = respirationRate();
      let status = "";
      if (!alive) status = "枯れてしまった：栄養が尽きた";
      else if (sun > 5) status = "光合成中：CO₂を使って栄養とO₂を作る";
      else status = "夜の呼吸中：栄養とO₂を使ってCO₂を出す";

      ctx.fillStyle = "#111827";
      ctx.font = "18px system-ui";
      ctx.fillText(status, 30, 38);

      ctx.font = "15px system-ui";
      if (sun > 5) {
        ctx.fillText("光合成： CO₂ + 光 → 栄養 + O₂", 30, 66);
      } else {
        ctx.fillText("呼吸： 栄養 + O₂ → CO₂", 30, 66);
      }

      ctx.fillStyle = "#475569";
      ctx.font = "13px system-ui";
      ctx.fillText(`光合成速度 ${pRate.toFixed(2)} / 呼吸速度 ${rRate.toFixed(2)}`, 30, 91);
    }

    function updateUI() {
      const sun = Number(sunlightSlider.value);
      const co2 = Number(co2Slider.value);
      const o2 = Number(o2Slider.value);
      const pRate = photosynthesisRate();
      const rRate = respirationRate();

      document.getElementById("sunText").textContent = `${sun.toFixed(0)} %`;
      document.getElementById("co2Text").textContent = `${co2.toFixed(0)} %`;
      document.getElementById("o2Text").textContent = `${o2.toFixed(0)} %`;
      document.getElementById("modeStat").textContent = alive ? (sun > 5 ? "光合成" : "呼吸") : "枯死";
      document.getElementById("sizeStat").textContent = plantSize.toFixed(2);
      document.getElementById("nutrientStat").textContent = `${nutrients.toFixed(0)} %`;
      document.getElementById("healthStat").textContent = `${health.toFixed(0)} %`;
      document.getElementById("photoRateStat").textContent = pRate.toFixed(2);
      document.getElementById("respRateStat").textContent = rRate.toFixed(2);
    }

    function draw() {
      drawBackground();
      drawEquationAndStatus();
      drawParticles();
      drawPlant();
      drawBars();
      drawGraph();
    }

    function loop() {
      updateModel();
      updateParticles();
      updateUI();
      draw();
      requestAnimationFrame(loop);
    }

    loop();
  </script>
