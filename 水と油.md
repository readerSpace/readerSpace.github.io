  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const addAmount = document.getElementById("addAmount");
    const shakePower = document.getElementById("shakePower");

    let waterVolume = 0;
    let oilVolume = 0;
    let mix = 0;
    let shaking = false;
    let shakeTimer = 0;
    let particles = [];
    let pourDrops = [];

    const maxVolume = 1000;
    const beaker = { x: 210, y: 80, w: 360, h: 390 };

    function totalVolume() {
      return waterVolume + oilVolume;
    }

    function liquidBounds() {
      const fillRatio = Math.min(1, totalVolume() / maxVolume);
      const liquidH = beaker.h * fillRatio;
      const liquidTop = beaker.y + beaker.h - liquidH;
      return { liquidH, liquidTop, liquidBottom: beaker.y + beaker.h };
    }

    function randomInsideLiquid() {
      const b = liquidBounds();
      const margin = 42;
      return {
        x: beaker.x + margin + Math.random() * (beaker.w - margin * 2),
        y: b.liquidTop + 15 + Math.random() * Math.max(1, b.liquidH - 30)
      };
    }

    function addWater() {
      const amount = Number(addAmount.value);
      waterVolume = Math.min(maxVolume - oilVolume, waterVolume + amount);
      addPourDrops("water", amount);
      createParticles("water", Math.floor(amount / 12));
    }

    function addOil() {
      const amount = Number(addAmount.value);
      oilVolume = Math.min(maxVolume - waterVolume, oilVolume + amount);
      addPourDrops("oil", amount);
      createParticles("oil", Math.floor(amount / 12));
    }

    function addPourDrops(type, amount) {
      for (let i = 0; i < Math.min(24, amount / 5); i++) {
        pourDrops.push({
          type,
          x: 360 + Math.random() * 70 - 35,
          y: 25 + Math.random() * 25,
          vy: 2 + Math.random() * 3,
          r: 4 + Math.random() * 5
        });
      }
    }

    function createParticles(type, n) {
      if (totalVolume() <= 0) return;
      for (let i = 0; i < n; i++) {
        const p = randomInsideLiquid();
        particles.push({
          type,
          x: p.x,
          y: p.y,
          vx: Math.random() * 2 - 1,
          vy: Math.random() * 2 - 1,
          r: 3 + Math.random() * 7,
          life: 1
        });
      }
      if (particles.length > 500) particles.splice(0, particles.length - 500);
    }

    function startShake() {
      if (totalVolume() <= 0) return;
      shaking = true;
      shakeTimer = 90;
    }

    function resetSim() {
      waterVolume = 0;
      oilVolume = 0;
      mix = 0;
      shaking = false;
      shakeTimer = 0;
      particles = [];
      pourDrops = [];
    }

    function updateUI() {
      document.getElementById("addAmountText").textContent = `${addAmount.value} mL`;
      document.getElementById("shakePowerText").textContent = shakePower.value;
      document.getElementById("waterStat").textContent = `${waterVolume.toFixed(0)} mL`;
      document.getElementById("oilStat").textContent = `${oilVolume.toFixed(0)} mL`;
      document.getElementById("totalStat").textContent = `${totalVolume().toFixed(0)} mL`;
      document.getElementById("mixStat").textContent = `${(mix * 100).toFixed(0)} %`;
      document.getElementById("oilDropStat").textContent = particles.filter(p => p.type === "oil").length;
      document.getElementById("waterDropStat").textContent = particles.filter(p => p.type === "water").length;
    }

    function physicsStep() {
      const total = totalVolume();
      const power = Number(shakePower.value) / 100;
      const b = liquidBounds();

      if (shaking && total > 0) {
        mix = Math.min(1, mix + 0.025 + power * 0.04);

        const n = Math.floor(2 + power * 8);
        createParticles("oil", n);
        createParticles("water", Math.max(1, Math.floor(n * 0.6)));

        shakeTimer--;
        if (shakeTimer <= 0) shaking = false;
      } else {
        mix *= 0.992;
      }

      const waterRatio = total > 0 ? waterVolume / total : 0;
      const oilRatio = total > 0 ? oilVolume / total : 0;
      const waterTop = b.liquidBottom - b.liquidH * waterRatio;
      const oilTop = b.liquidTop;
      const oilBottom = waterTop;

      for (const p of particles) {
        if (shaking) {
          const force = 2 + power * 6;
          p.vx += (Math.random() - 0.5) * force;
          p.vy += (Math.random() - 0.5) * force;
        } else {
          if (p.type === "oil") {
            const targetY = oilTop + Math.random() * Math.max(1, oilBottom - oilTop);
            p.vy += (targetY - p.y) * 0.0018 - 0.015;
          } else {
            const targetY = waterTop + Math.random() * Math.max(1, b.liquidBottom - waterTop);
            p.vy += (targetY - p.y) * 0.0018 + 0.015;
          }
          p.vx *= 0.96;
          p.vy *= 0.96;
        }

        p.x += p.vx;
        p.y += p.vy;
        p.vx *= 0.91;
        p.vy *= 0.91;

        const left = beaker.x + 38;
        const right = beaker.x + beaker.w - 38;
        if (p.x < left) { p.x = left; p.vx *= -0.55; }
        if (p.x > right) { p.x = right; p.vx *= -0.55; }
        if (p.y < b.liquidTop + 8) { p.y = b.liquidTop + 8; p.vy *= -0.4; }
        if (p.y > b.liquidBottom - 8) { p.y = b.liquidBottom - 8; p.vy *= -0.4; }

        p.life -= shaking ? 0.0005 : 0.0028;
        p.r *= shaking ? 0.998 : 0.999;
      }

      particles = particles.filter(p => p.life > 0.12 && p.r > 1.2);

      pourDrops = pourDrops
        .map(d => ({ ...d, y: d.y + d.vy, vy: d.vy + 0.06 }))
        .filter(d => d.y < b.liquidBottom);
    }

    function drawBeaker() {
      ctx.lineWidth = 5;
      ctx.strokeStyle = "rgba(30, 41, 59, 0.72)";
      ctx.beginPath();
      ctx.moveTo(beaker.x, beaker.y);
      ctx.lineTo(beaker.x + 35, beaker.y + beaker.h);
      ctx.lineTo(beaker.x + beaker.w - 35, beaker.y + beaker.h);
      ctx.lineTo(beaker.x + beaker.w, beaker.y);
      ctx.stroke();

      ctx.lineWidth = 3;
      ctx.strokeStyle = "rgba(148, 163, 184, 0.7)";
      for (let i = 1; i <= 5; i++) {
        const yy = beaker.y + beaker.h - i * beaker.h / 6;
        ctx.beginPath();
        ctx.moveTo(beaker.x + 28, yy);
        ctx.lineTo(beaker.x + 54, yy);
        ctx.stroke();
      }
    }

    function drawLiquidLayers() {
      const total = totalVolume();
      if (total <= 0) return;

      const b = liquidBounds();
      const waterRatio = waterVolume / total;
      const oilRatio = oilVolume / total;
      const waterH = b.liquidH * waterRatio;
      const oilH = b.liquidH * oilRatio;
      const waterY = b.liquidBottom - waterH;
      const oilY = waterY - oilH;

      const shakeOffset = shaking ? Math.sin(Date.now() * 0.04) * 8 : Math.sin(Date.now() * 0.002) * 2;

      if (waterVolume > 0) {
        ctx.fillStyle = mix > 0.35 ? "rgba(125, 211, 252, 0.78)" : "rgba(14, 165, 233, 0.68)";
        ctx.beginPath();
        ctx.moveTo(beaker.x + 35, b.liquidBottom);
        ctx.lineTo(beaker.x + beaker.w - 35, b.liquidBottom);
        ctx.lineTo(beaker.x + beaker.w - 35 + 25 * waterRatio, waterY + shakeOffset * 0.4);
        ctx.quadraticCurveTo(beaker.x + beaker.w / 2, waterY - shakeOffset, beaker.x + 35 - 25 * waterRatio, waterY + shakeOffset * 0.4);
        ctx.closePath();
        ctx.fill();
      }

      if (oilVolume > 0) {
        const alpha = mix > 0.45 ? 0.58 : 0.75;
        ctx.fillStyle = `rgba(245, 158, 11, ${alpha})`;
        ctx.beginPath();
        ctx.moveTo(beaker.x + 35 - 25 * waterRatio, waterY + shakeOffset * 0.4);
        ctx.quadraticCurveTo(beaker.x + beaker.w / 2, waterY - shakeOffset, beaker.x + beaker.w - 35 + 25 * waterRatio, waterY + shakeOffset * 0.4);
        ctx.lineTo(beaker.x + beaker.w - 35 + 25, oilY + shakeOffset);
        ctx.quadraticCurveTo(beaker.x + beaker.w / 2, oilY + Math.sin(Date.now() * 0.003) * 5, beaker.x + 35 - 25, oilY + shakeOffset);
        ctx.closePath();
        ctx.fill();
      }

      if (mix > 0.2) {
        ctx.fillStyle = `rgba(255,255,255,${0.12 + mix * 0.22})`;
        ctx.beginPath();
        ctx.roundRect(beaker.x + 45, b.liquidTop + 8, beaker.w - 90, b.liquidH - 16, 18);
        ctx.fill();
      }
    }

    function drawParticles() {
      for (const p of particles) {
        if (p.type === "oil") {
          ctx.fillStyle = `rgba(251, 191, 36, ${0.35 + mix * 0.55})`;
          ctx.strokeStyle = "rgba(146, 64, 14, 0.25)";
        } else {
          ctx.fillStyle = `rgba(56, 189, 248, ${0.25 + mix * 0.45})`;
          ctx.strokeStyle = "rgba(12, 74, 110, 0.18)";
        }
        ctx.beginPath();
        ctx.arc(p.x, p.y, p.r, 0, Math.PI * 2);
        ctx.fill();
        ctx.stroke();
      }
    }

    function drawPourDrops() {
      for (const d of pourDrops) {
        ctx.fillStyle = d.type === "oil" ? "rgba(245, 158, 11, 0.85)" : "rgba(14, 165, 233, 0.75)";
        ctx.beginPath();
        ctx.arc(d.x, d.y, d.r, 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function draw() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      const wobble = shaking ? Math.sin(Date.now() * 0.06) * 10 : 0;
      ctx.save();
      ctx.translate(wobble, 0);

      drawLiquidLayers();
      drawParticles();
      drawBeaker();
      drawPourDrops();

      ctx.restore();

      ctx.fillStyle = "#111827";
      ctx.font = "16px system-ui";
      let status = "静置中：油が上、水が下に分離";
      if (shaking) status = "振っている：小さな液滴に分かれて一時的に混ざる";
      else if (mix > 0.35) status = "白っぽい混合状態：少しずつ分離中";
      ctx.fillText(status, 32, 40);

      ctx.fillStyle = "#475569";
      ctx.font = "14px system-ui";
      ctx.fillText("ポイント：水は極性分子、油は非極性分子なので、互いに混ざりにくい", 32, 66);

      // Layer labels
      if (totalVolume() > 0) {
        const b = liquidBounds();
        const total = totalVolume();
        const waterRatio = waterVolume / total;
        const waterY = b.liquidBottom - b.liquidH * waterRatio;

        if (oilVolume > 0) {
          ctx.fillStyle = "#92400e";
          ctx.font = "15px system-ui";
          ctx.fillText("油：軽いので上", 585, Math.max(b.liquidTop + 35, waterY - 45));
        }
        if (waterVolume > 0) {
          ctx.fillStyle = "#075985";
          ctx.fillText("水：重いので下", 585, Math.min(b.liquidBottom - 35, waterY + 70));
        }
      }

      // Shake meter
      ctx.fillStyle = "#e5e7eb";
      ctx.fillRect(32, 500, 220, 16);
      ctx.fillStyle = "#7c3aed";
      ctx.fillRect(32, 500, 220 * mix, 16);
      ctx.strokeStyle = "#334155";
      ctx.strokeRect(32, 500, 220, 16);
      ctx.fillStyle = "#111827";
      ctx.font = "13px system-ui";
      ctx.fillText("混ざり具合", 32, 492);
    }

    function loop() {
      physicsStep();
      updateUI();
      draw();
      requestAnimationFrame(loop);
    }

    updateUI();
    loop();
  </script>

水・油を別々に追加
油が上、水が下に分離
「振る」で一時的に混ざる
時間経過でまた分離
液滴・白っぽい乳化状態を表示

※界面活性剤なしの簡易モデルです。
