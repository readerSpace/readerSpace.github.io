  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const freq = document.getElementById("freq");
    const power = document.getElementById("power");
    const lengthSlider = document.getElementById("length");
    const angle = document.getElementById("angle");
    const mode = document.getElementById("mode");
    const pauseBtn = document.getElementById("pauseBtn");
    const resetBtn = document.getElementById("resetBtn");

    const freqVal = document.getElementById("freqVal");
    const powerVal = document.getElementById("powerVal");
    const lenVal = document.getElementById("lenVal");
    const angleVal = document.getElementById("angleVal");

    let W = 0, H = 0;
    let t = 0;
    let paused = false;
    let lastTime = performance.now();

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      W = canvas.clientWidth;
      H = canvas.clientHeight;
      canvas.width = Math.floor(W * dpr);
      canvas.height = Math.floor(H * dpr);
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    }
    window.addEventListener("resize", resize);
    resize();

    function updateLabels() {
      freqVal.textContent = Number(freq.value).toFixed(2);
      powerVal.textContent = Number(power.value).toFixed(2);
      lenVal.textContent = Number(lengthSlider.value).toFixed(2);
      angleVal.textContent = `${angle.value}°`;
    }
    [freq, power, lengthSlider, angle].forEach(el => el.addEventListener("input", updateLabels));
    updateLabels();

    pauseBtn.addEventListener("click", () => {
      paused = !paused;
      pauseBtn.textContent = paused ? "再開" : "一時停止";
    });
    resetBtn.addEventListener("click", () => {
      t = 0;
      paused = false;
      pauseBtn.textContent = "一時停止";
    });

    function drawGrid() {
      ctx.save();
      ctx.strokeStyle = "rgba(148,163,184,0.08)";
      ctx.lineWidth = 1;
      const step = 40;
      for (let x = 0; x < W; x += step) {
        ctx.beginPath();
        ctx.moveTo(x, 0);
        ctx.lineTo(x, H);
        ctx.stroke();
      }
      for (let y = 0; y < H; y += step) {
        ctx.beginPath();
        ctx.moveTo(0, y);
        ctx.lineTo(W, y);
        ctx.stroke();
      }
      ctx.restore();
    }

    function drawArrow(x1, y1, x2, y2, color, width = 2) {
      const ang = Math.atan2(y2 - y1, x2 - x1);
      const head = 8;
      ctx.save();
      ctx.strokeStyle = color;
      ctx.fillStyle = color;
      ctx.lineWidth = width;
      ctx.beginPath();
      ctx.moveTo(x1, y1);
      ctx.lineTo(x2, y2);
      ctx.stroke();
      ctx.beginPath();
      ctx.moveTo(x2, y2);
      ctx.lineTo(x2 - head * Math.cos(ang - Math.PI / 6), y2 - head * Math.sin(ang - Math.PI / 6));
      ctx.lineTo(x2 - head * Math.cos(ang + Math.PI / 6), y2 - head * Math.sin(ang + Math.PI / 6));
      ctx.closePath();
      ctx.fill();
      ctx.restore();
    }

    function drawDipole(x, y, len, current, title, receiveAngle = 0, receiveStrength = 1) {
      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(receiveAngle);

      const half = len / 2;
      ctx.lineCap = "round";
      ctx.lineWidth = 8;
      ctx.strokeStyle = "rgba(226,232,240,0.95)";
      ctx.beginPath();
      ctx.moveTo(0, -half);
      ctx.lineTo(0, -12);
      ctx.moveTo(0, 12);
      ctx.lineTo(0, half);
      ctx.stroke();

      ctx.lineWidth = 2;
      ctx.strokeStyle = "rgba(56,189,248,0.75)";
      ctx.beginPath();
      ctx.arc(0, 0, 12, 0, Math.PI * 2);
      ctx.stroke();

      const qOffset = current * half * 0.32;
      const topColor = current >= 0 ? "#f87171" : "#93c5fd";
      const bottomColor = current >= 0 ? "#93c5fd" : "#f87171";

      ctx.fillStyle = topColor;
      ctx.beginPath();
      ctx.arc(0, -half + 18 + qOffset, 8, 0, Math.PI * 2);
      ctx.fill();
      ctx.fillStyle = bottomColor;
      ctx.beginPath();
      ctx.arc(0, half - 18 + qOffset, 8, 0, Math.PI * 2);
      ctx.fill();

      const arrowLen = 42 * Math.abs(current) * receiveStrength;
      if (arrowLen > 2) {
        const dir = current >= 0 ? -1 : 1;
        drawArrow(0, 20 * dir, 0, (20 + arrowLen) * dir, "rgba(134,239,172,0.9)", 3);
      }

      ctx.restore();

      ctx.save();
      ctx.fillStyle = "rgba(226,232,240,0.9)";
      ctx.font = "15px system-ui";
      ctx.textAlign = "center";
      ctx.fillText(title, x, y + len / 2 + 34);
      ctx.restore();
    }

    function drawWave(tx, rx, y, amp, f, time, drawMode) {
      const waveLen = 220 / f;
      const k = 2 * Math.PI / waveLen;
      const omega = 2 * Math.PI * f;
      const start = tx + 55;
      const end = rx - 55;

      ctx.save();
      ctx.lineWidth = 3;

      if (drawMode === "both" || drawMode === "electric") {
        ctx.strokeStyle = "rgba(96,165,250,0.95)";
        ctx.beginPath();
        for (let x = start; x <= end; x += 3) {
          const phase = k * (x - start) - omega * time;
          const decay = 0.85 - 0.20 * ((x - start) / Math.max(1, end - start));
          const yy = y - Math.sin(phase) * amp * decay;
          if (x === start) ctx.moveTo(x, yy);
          else ctx.lineTo(x, yy);
        }
        ctx.stroke();

        for (let x = start + 20; x < end; x += 48) {
          const phase = k * (x - start) - omega * time;
          const e = Math.sin(phase) * amp * 0.75;
          drawArrow(x, y, x, y - e, "rgba(96,165,250,0.78)", 2);
        }
      }

      if (drawMode === "both" || drawMode === "magnetic") {
        ctx.strokeStyle = "rgba(244,114,182,0.85)";
        ctx.beginPath();
        for (let x = start; x <= end; x += 3) {
          const phase = k * (x - start) - omega * time - Math.PI / 2;
          const decay = 0.85 - 0.20 * ((x - start) / Math.max(1, end - start));
          const yy = y + Math.sin(phase) * amp * 0.55 * decay;
          if (x === start) ctx.moveTo(x, yy);
          else ctx.lineTo(x, yy);
        }
        ctx.stroke();

        for (let x = start + 45; x < end; x += 62) {
          const phase = k * (x - start) - omega * time - Math.PI / 2;
          const b = Math.sin(phase);
          ctx.fillStyle = b >= 0 ? "rgba(244,114,182,0.9)" : "rgba(244,114,182,0.35)";
          ctx.beginPath();
          ctx.arc(x, y + 72, 9 + Math.abs(b) * 6, 0, Math.PI * 2);
          ctx.fill();
          ctx.strokeStyle = "rgba(244,114,182,0.45)";
          ctx.stroke();
          ctx.fillStyle = "rgba(255,255,255,0.8)";
          ctx.font = "12px system-ui";
          ctx.textAlign = "center";
          ctx.fillText(b >= 0 ? "⊙" : "⊗", x, y + 76);
        }
      }

      if (drawMode === "energy") {
        for (let x = start + 20; x < end; x += 42) {
          const phase = k * (x - start) - omega * time;
          const s = Math.max(0.25, Math.abs(Math.sin(phase)));
          drawArrow(x - 14, y, x + 18 + s * 18, y, `rgba(134,239,172,${0.35 + s * 0.6})`, 2 + s * 2);
        }
        ctx.fillStyle = "rgba(134,239,172,0.9)";
        ctx.font = "14px system-ui";
        ctx.textAlign = "center";
        ctx.fillText("エネルギーの流れ：ポインティングベクトル S = E × B", (start + end) / 2, y - amp - 25);
      }

      ctx.restore();
    }

    function drawResonanceMeter(x, y, w, h, value, label) {
      ctx.save();
      ctx.fillStyle = "rgba(15,23,42,0.9)";
      ctx.strokeStyle = "rgba(255,255,255,0.12)";
      ctx.lineWidth = 1;
      ctx.beginPath();
      ctx.roundRect(x, y, w, h, 12);
      ctx.fill();
      ctx.stroke();

      ctx.fillStyle = "rgba(148,163,184,0.9)";
      ctx.font = "13px system-ui";
      ctx.fillText(label, x + 12, y + 22);

      ctx.fillStyle = "rgba(30,41,59,0.9)";
      ctx.fillRect(x + 12, y + 34, w - 24, 12);
      ctx.fillStyle = "rgba(134,239,172,0.9)";
      ctx.fillRect(x + 12, y + 34, (w - 24) * Math.max(0, Math.min(1, value)), 12);
      ctx.fillStyle = "rgba(226,232,240,0.9)";
      ctx.font = "12px system-ui";
      ctx.fillText(`${Math.round(value * 100)}%`, x + w - 46, y + 22);
      ctx.restore();
    }

    function render(now) {
      const dt = Math.min(0.04, (now - lastTime) / 1000);
      lastTime = now;
      if (!paused) t += dt;

      ctx.clearRect(0, 0, W, H);
      drawGrid();

      const tx = W * 0.18;
      const rx = W * 0.82;
      const cy = H * 0.48;
      const f = Number(freq.value);
      const p = Number(power.value);
      const antennaLength = Math.min(H * 0.48, 250) * Number(lengthSlider.value);
      const theta = Number(angle.value) * Math.PI / 180;
      const omega = 2 * Math.PI * f;
      const current = Math.sin(omega * t);
      const amp = 46 * p;
      const waveLen = 220 / f;

      // Simple resonance model: good when antenna length ≈ half wavelength in this scaled world.
      const halfWave = waveLen / 2;
      const resonance = Math.exp(-Math.pow((antennaLength - halfWave) / (halfWave * 0.55), 2));
      const polarization = Math.abs(Math.cos(theta));

      const rxPhase = 2 * Math.PI * f * t - (rx - tx) / waveLen * 2 * Math.PI;
      const receivedCurrent = Math.sin(rxPhase) * p * resonance * polarization;

      drawWave(tx, rx, cy, amp, f, t, mode.value);

      drawDipole(tx, cy, antennaLength, current, "送信アンテナ", 0, 1);
      drawDipole(rx, cy, antennaLength, receivedCurrent, "受信アンテナ", theta, Math.max(0.15, resonance * polarization));

      ctx.save();
      ctx.fillStyle = "rgba(226,232,240,0.94)";
      ctx.font = "17px system-ui";
      ctx.textAlign = "center";
      ctx.fillText("電波は、電場と磁場が互いに作り合いながら空間を進む波", W / 2, 34);

      ctx.fillStyle = "rgba(148,163,184,0.9)";
      ctx.font = "13px system-ui";
      ctx.fillText("送信：交流電流で電荷が加速 → 電磁波を放射", tx, cy - antennaLength / 2 - 30);
      ctx.fillText("受信：到来した電場が電子を揺らす → 交流信号", rx, cy - antennaLength / 2 - 30);
      ctx.restore();

      drawResonanceMeter(18, H - 150, 230, 62, resonance, "アンテナ長と波長の相性");
      drawResonanceMeter(18, H - 80, 230, 62, polarization, "偏波の向きの一致");

      ctx.save();
      ctx.fillStyle = "rgba(15,23,42,0.88)";
      ctx.strokeStyle = "rgba(255,255,255,0.12)";
      ctx.beginPath();
      ctx.roundRect(W - 270, H - 108, 246, 86, 12);
      ctx.fill();
      ctx.stroke();
      ctx.fillStyle = "rgba(226,232,240,0.92)";
      ctx.font = "13px system-ui";
      ctx.fillText(`受信信号 ≈ ${receivedCurrent.toFixed(2)}`, W - 250, H - 78);
      ctx.fillText(`波長スケール ≈ ${waveLen.toFixed(0)} px`, W - 250, H - 54);
      ctx.fillText(`半波長 ≈ ${(halfWave).toFixed(0)} px`, W - 250, H - 30);
      ctx.restore();

      requestAnimationFrame(render);
    }

    if (!CanvasRenderingContext2D.prototype.roundRect) {
      CanvasRenderingContext2D.prototype.roundRect = function(x, y, w, h, r) {
        this.beginPath();
        this.moveTo(x + r, y);
        this.lineTo(x + w - r, y);
        this.quadraticCurveTo(x + w, y, x + w, y + r);
        this.lineTo(x + w, y + h - r);
        this.quadraticCurveTo(x + w, y + h, x + w - r, y + h);
        this.lineTo(x + r, y + h);
        this.quadraticCurveTo(x, y + h, x, y + h - r);
        this.lineTo(x, y + r);
        this.quadraticCurveTo(x, y, x + r, y);
        this.closePath();
      };
    }

    requestAnimationFrame(render);
  </script>
</body>
