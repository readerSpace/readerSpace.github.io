    function resize() {
      const rect = canvas.getBoundingClientRect();
      const dpr = window.devicePixelRatio || 1;
      canvas.width = Math.floor(rect.width * dpr);
      canvas.height = Math.floor(rect.height * dpr);

    function vec(x, y) { return { x, y }; }
    function add(a, b) { return vec(a.x + b.x, a.y + b.y); }
    function sub(a, b) { return vec(a.x - b.x, a.y - b.y); }
    function mul(a, s) { return vec(a.x * s, a.y * s); }
    function dot(a, b) { return a.x * b.x + a.y * b.y; }
    function len(a) { return Math.hypot(a.x, a.y); }
    function norm(a) { const l = len(a); return vec(a.x / l, a.y / l); }
    function reflect(I, N) { return sub(I, mul(N, 2 * dot(I, N))); }

    function refract(I, N, n1, n2) {
      // I: 入射方向、N: 境界面の法線。戻り値は屈折方向。
      let cosi = Math.max(-1, Math.min(1, dot(I, N)));
      let etai = n1;
      let etat = n2;
      let n = N;
      if (cosi < 0) {
        cosi = -cosi;
      } else {
        n = mul(N, -1);
        [etai, etat] = [etat, etai];
      }
      const eta = etai / etat;
      const k = 1 - eta * eta * (1 - cosi * cosi);
      if (k < 0) return null;
      return norm(add(mul(I, eta), mul(n, eta * cosi - Math.sqrt(k))));
    }

    function intersectCircle(P, D, C, R, minT = 0.001) {
      const f = sub(P, C);
      const a = dot(D, D);
      const b = 2 * dot(f, D);
      const c = dot(f, f) - R * R;
      const disc = b * b - 4 * a * c;
      if (disc < 0) return null;
      const s = Math.sqrt(disc);
      const t1 = (-b - s) / (2 * a);
      const t2 = (-b + s) / (2 * a);
      const t = [t1, t2].filter(v => v > minT).sort((a, b) => a - b)[0];
      if (t === undefined) return null;
      return add(P, mul(D, t));
    }

    function drawLine(points, color, width = 2, alpha = 1) {
      ctx.save();
      ctx.globalAlpha = alpha;
      ctx.strokeStyle = color;
      ctx.lineWidth = width;
      ctx.lineCap = "round";
      ctx.lineJoin = "round";
      ctx.beginPath();
      ctx.moveTo(points[0].x, points[0].y);
      for (let i = 1; i < points.length; i++) ctx.lineTo(points[i].x, points[i].y);
      ctx.stroke();
      ctx.restore();
    }

    function drawArrow(P, D, color, label) {
      const end = add(P, mul(D, 55));
      drawLine([P, end], color, 2, 0.9);
      const angle = Math.atan2(D.y, D.x);
      ctx.save();
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.moveTo(end.x, end.y);
      ctx.lineTo(end.x - 10 * Math.cos(angle - 0.35), end.y - 10 * Math.sin(angle - 0.35));
      ctx.lineTo(end.x - 10 * Math.cos(angle + 0.35), end.y - 10 * Math.sin(angle + 0.35));
      ctx.closePath();
      ctx.fill();
      if (label) {
        ctx.font = "13px system-ui";
        ctx.fillText(label, end.x + 8, end.y + 4);
      }
      ctx.restore();
    }

    function drawAngleArc(center, startAngle, endAngle, radius, text) {
      ctx.save();
      ctx.strokeStyle = "rgba(15, 23, 42, 0.45)";
      ctx.lineWidth = 1.5;
      ctx.beginPath();
      ctx.arc(center.x, center.y, radius, startAngle, endAngle);
      ctx.stroke();
      const mid = (startAngle + endAngle) / 2;
      ctx.fillStyle = "#334155";
      ctx.font = "13px system-ui";
      ctx.fillText(text, center.x + Math.cos(mid) * (radius + 8), center.y + Math.sin(mid) * (radius + 8));
      ctx.restore();
    }

    function getSelectedColors() {
      if (mode.value === "white") return colors;
      return colors.filter(c => c.name === mode.value);
    }

    function simulateRay(C, R, impactRatio, colorInfo, bounceCount) {
      const start = vec(C.x - R - 210, C.y + impactRatio * R);
      let D = norm(vec(1, 0));
      const entry = intersectCircle(start, D, C, R);
      if (!entry) return null;

      let N = norm(sub(entry, C));
      let inside = refract(D, N, 1.0, colorInfo.n);
      if (!inside) return null;

      const points = [start, entry];
      let P = entry;
      let currentD = inside;

      for (let i = 0; i < bounceCount; i++) {
        const hit = intersectCircle(P, currentD, C, R, 0.01);
        if (!hit) return null;
        points.push(hit);
        const normal = norm(sub(hit, C));
        currentD = reflect(currentD, normal);
        P = hit;
      }

      const exit = intersectCircle(P, currentD, C, R, 0.01);
      if (!exit) return null;
      points.push(exit);
      N = norm(sub(exit, C));
      const out = refract(currentD, N, colorInfo.n, 1.0);
      if (!out) return null;
      points.push(add(exit, mul(out, 380)));

      return { points, out, exit };
    }

    function draw() {
      const w = canvas.clientWidth;
      const h = canvas.clientHeight;
      ctx.clearRect(0, 0, w, h);

      const R = Number(radiusSlider.value);
      const C = vec(w * 0.48, h * 0.5);
      const impactRatio = Number(impact.value);
      const bounceCount = Number(bounces.value);
      impactValue.textContent = impactRatio.toFixed(2);
      radiusValue.textContent = R + " px";

      // 太陽
      ctx.save();
      const sun = vec(70, 70);
      const gradSun = ctx.createRadialGradient(sun.x, sun.y, 5, sun.x, sun.y, 55);
      gradSun.addColorStop(0, "rgba(253, 224, 71, 1)");
      gradSun.addColorStop(1, "rgba(253, 224, 71, 0)");
      ctx.fillStyle = gradSun;
      ctx.beginPath();
      ctx.arc(sun.x, sun.y, 55, 0, Math.PI * 2);
      ctx.fill();
      ctx.fillStyle = "#92400e";
      ctx.font = "14px system-ui";
      ctx.fillText("太陽光", 28, 132);
      ctx.restore();

      // 水滴
      ctx.save();
      const grad = ctx.createRadialGradient(C.x - R * 0.35, C.y - R * 0.4, R * 0.05, C.x, C.y, R);
      grad.addColorStop(0, "rgba(255,255,255,0.92)");
      grad.addColorStop(0.45, "rgba(147,197,253,0.25)");
      grad.addColorStop(1, "rgba(37,99,235,0.22)");
      ctx.fillStyle = grad;
      ctx.strokeStyle = "rgba(37, 99, 235, 0.55)";
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.arc(C.x, C.y, R, 0, Math.PI * 2);
      ctx.fill();
      ctx.stroke();
      ctx.fillStyle = "rgba(255,255,255,0.85)";
      ctx.beginPath();
      ctx.ellipse(C.x - R * 0.35, C.y - R * 0.35, R * 0.22, R * 0.12, -0.7, 0, Math.PI * 2);
      ctx.fill();
      ctx.restore();

      // 入射する白色光の束
      const baseY = C.y + impactRatio * R;
      for (let i = -2; i <= 2; i++) {
        drawLine([vec(C.x - R - 210, baseY + i * 7), vec(C.x - R + 4, baseY + i * 7)], "rgba(255, 255, 255, 0.88)", 3, 0.8);
      }
      ctx.save();
      ctx.fillStyle = "#334155";
      ctx.font = "13px system-ui";
      ctx.fillText("白色光が入る", C.x - R - 200, baseY - 20);
      ctx.restore();

      const selected = getSelectedColors();
      const offsetStep = mode.value === "white" ? 3.5 : 0;
      selected.forEach((col, i) => {
        const shiftedImpact = impactRatio + (i - (selected.length - 1) / 2) * 0.012;
        const ray = simulateRay(C, R, shiftedImpact, col, bounceCount);
        if (!ray) return;
        drawLine(ray.points, col.color, 2.7, 0.92);
        drawArrow(ray.exit, ray.out, col.color, mode.value === "white" ? col.label : col.label + "の光");
      });

      // 説明テキスト
      ctx.save();
      ctx.fillStyle = "rgba(255,255,255,0.86)";
      ctx.strokeStyle = "rgba(148,163,184,0.6)";
      ctx.lineWidth = 1;
      const boxX = 22, boxY = h - 152, boxW = Math.min(560, w - 44), boxH = 124;
      ctx.beginPath();
      ctx.roundRect(boxX, boxY, boxW, boxH, 14);
      ctx.fill();
      ctx.stroke();
      ctx.fillStyle = "#0f172a";
      ctx.font = "bold 15px system-ui";
      ctx.fillText(bounceCount === 1 ? "主虹：1回反射" : "副虹：2回反射", boxX + 16, boxY + 28);
      ctx.font = "13px system-ui";
      ctx.fillText("1. 空気から水へ入るとき、光は屈折する。", boxX + 16, boxY + 54);
      ctx.fillText("2. 水滴の内側で反射する。", boxX + 16, boxY + 78);
      ctx.fillText("3. 水から空気へ出るとき、色ごとに少し違う向きへ分かれる。", boxX + 16, boxY + 102);
      ctx.restore();

      // 観察者
      const observer = vec(w - 95, h * 0.66);
      ctx.save();
      ctx.strokeStyle = "#334155";
      ctx.lineWidth = 3;
      ctx.beginPath();
      ctx.arc(observer.x, observer.y - 20, 14, 0, Math.PI * 2);
      ctx.stroke();
      ctx.beginPath();
      ctx.moveTo(observer.x, observer.y - 6);
      ctx.lineTo(observer.x, observer.y + 44);
      ctx.moveTo(observer.x, observer.y + 8);
      ctx.lineTo(observer.x - 22, observer.y + 24);
      ctx.moveTo(observer.x, observer.y + 8);
      ctx.lineTo(observer.x + 22, observer.y + 24);
      ctx.moveTo(observer.x, observer.y + 44);
      ctx.lineTo(observer.x - 18, observer.y + 70);
      ctx.moveTo(observer.x, observer.y + 44);
      ctx.lineTo(observer.x + 18, observer.y + 70);
      ctx.stroke();
      ctx.fillStyle = "#334155";
      ctx.font = "13px system-ui";
      ctx.fillText("観察者", observer.x - 22, observer.y + 94);
      ctx.restore();
    }

要点：

虹は 屈折 → 水滴内で反射 → 再び屈折 で起きる
色ごとに屈折率が少し違うので、赤・紫などに分かれる
主虹は反射1回、副虹は反射2回
副虹では色の順番が逆になる
