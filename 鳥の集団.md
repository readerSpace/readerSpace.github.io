  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const cohesionSlider = document.getElementById("cohesion");
    const separationSlider = document.getElementById("separation");
    const alignmentSlider = document.getElementById("alignment");
    const predatorForceSlider = document.getElementById("predatorForce");

    const cohesionValue = document.getElementById("cohesionValue");
    const separationValue = document.getElementById("separationValue");
    const alignmentValue = document.getElementById("alignmentValue");
    const predatorValue = document.getElementById("predatorValue");

    const togglePredatorButton = document.getElementById("togglePredator");
    const toggleOddButton = document.getElementById("toggleOdd");
    const clearObstaclesButton = document.getElementById("clearObstacles");
    const resetButton = document.getElementById("reset");

    let width = 0;
    let height = 0;
    let boids = [];
    let obstacles = [];
    let predatorEnabled = true;
    let oddBoidEnabled = true;
    let pointerDown = false;
    let pointerMoved = false;

    const settings = {
      boidCount: 95,
      perceptionRadius: 72,
      separationRadius: 28,
      maxSpeed: 3.0,
      maxForce: 0.065,
      obstacleAvoidanceForce: 2.2,
      predatorRadius: 150,
      obstacleRadius: 28,
    };

    const mouse = {
      x: 0,
      y: 0,
    };

    const predator = {
      x: 0,
      y: 0,
      vx: 2.2,
      vy: 1.4,
    };

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      predator.x = width * 0.5;
      predator.y = height * 0.5;
    }

    class Vector {
      constructor(x = 0, y = 0) {
        this.x = x;
        this.y = y;
      }

      add(v) {
        this.x += v.x;
        this.y += v.y;
        return this;
      }

      sub(v) {
        this.x -= v.x;
        this.y -= v.y;
        return this;
      }

      mult(n) {
        this.x *= n;
        this.y *= n;
        return this;
      }

      div(n) {
        if (n !== 0) {
          this.x /= n;
          this.y /= n;
        }
        return this;
      }

      mag() {
        return Math.hypot(this.x, this.y);
      }

      normalize() {
        const m = this.mag();
        if (m > 0) this.div(m);
        return this;
      }

      limit(max) {
        const m = this.mag();
        if (m > max) {
          this.normalize().mult(max);
        }
        return this;
      }

      copy() {
        return new Vector(this.x, this.y);
      }

      static sub(a, b) {
        return new Vector(a.x - b.x, a.y - b.y);
      }

      static random2D() {
        const angle = Math.random() * Math.PI * 2;
        return new Vector(Math.cos(angle), Math.sin(angle));
      }
    }

    class Boid {
      constructor(x, y, isOdd = false) {
        this.position = new Vector(x, y);
        this.velocity = Vector.random2D().mult(1.5 + Math.random() * 1.5);
        this.acceleration = new Vector();
        this.isOdd = isOdd;
        this.size = isOdd ? 8 : 6;
      }

      edges() {
        // 画面端はワープ。群れが消えずに遊び続けられる。
        if (this.position.x > width + 20) this.position.x = -20;
        if (this.position.x < -20) this.position.x = width + 20;
        if (this.position.y > height + 20) this.position.y = -20;
        if (this.position.y < -20) this.position.y = height + 20;
      }

      applyForce(force) {
        this.acceleration.add(force);
      }

      flock(boids) {
        let alignment = new Vector();
        let cohesion = new Vector();
        let separation = new Vector();
        let total = 0;
        let closeTotal = 0;

        const perception = this.isOdd ? settings.perceptionRadius * 1.9 : settings.perceptionRadius;
        const sepRadius = this.isOdd ? settings.separationRadius * 0.6 : settings.separationRadius;

        for (const other of boids) {
          if (other === this) continue;
          const dx = this.position.x - other.position.x;
          const dy = this.position.y - other.position.y;
          const d = Math.hypot(dx, dy);

          if (d < perception) {
            alignment.add(other.velocity);
            cohesion.add(other.position);
            total++;
          }

          if (d < sepRadius && d > 0) {
            const diff = new Vector(dx, dy);
            diff.div(d * d);
            separation.add(diff);
            closeTotal++;
          }
        }

        if (total > 0) {
          alignment.div(total);
          alignment.normalize().mult(settings.maxSpeed);
          alignment.sub(this.velocity);
          alignment.limit(settings.maxForce);

          cohesion.div(total);
          cohesion.sub(this.position);
          cohesion.normalize().mult(settings.maxSpeed);
          cohesion.sub(this.velocity);
          cohesion.limit(settings.maxForce);
        }

        if (closeTotal > 0) {
          separation.div(closeTotal);
          separation.normalize().mult(settings.maxSpeed);
          separation.sub(this.velocity);
          separation.limit(settings.maxForce * 1.8);
        }

        // 変な1匹：みんなと逆に動きやすく、群れをかき乱す。
        if (this.isOdd && oddBoidEnabled) {
          alignment.mult(-0.9);
          cohesion.mult(-0.4);
          separation.mult(1.4);
        }

        alignment.mult(parseFloat(alignmentSlider.value));
        cohesion.mult(parseFloat(cohesionSlider.value));
        separation.mult(parseFloat(separationSlider.value));

        this.applyForce(alignment);
        this.applyForce(cohesion);
        this.applyForce(separation);
      }

      avoidPredator() {
        if (!predatorEnabled) return;

        const predatorVector = new Vector(predator.x, predator.y);
        const away = Vector.sub(this.position, predatorVector);
        const d = away.mag();

        if (d < settings.predatorRadius && d > 0) {
          away.normalize().mult(settings.maxSpeed * 1.8);
          away.sub(this.velocity);
          away.limit(settings.maxForce * parseFloat(predatorForceSlider.value));
          this.applyForce(away);
        }
      }

      avoidObstacles() {
        for (const obstacle of obstacles) {
          const obstacleVector = new Vector(obstacle.x, obstacle.y);
          const away = Vector.sub(this.position, obstacleVector);
          const d = away.mag();
          const avoidRadius = obstacle.r + 48;

          if (d < avoidRadius && d > 0) {
            away.normalize().mult(settings.maxSpeed * 1.4);
            away.sub(this.velocity);
            away.limit(settings.maxForce * settings.obstacleAvoidanceForce);
            this.applyForce(away);
          }
        }
      }

      update() {
        const maxSpeed = this.isOdd && oddBoidEnabled ? settings.maxSpeed * 1.35 : settings.maxSpeed;
        this.velocity.add(this.acceleration);
        this.velocity.limit(maxSpeed);
        this.position.add(this.velocity);
        this.acceleration.mult(0);
      }

      draw() {
        const angle = Math.atan2(this.velocity.y, this.velocity.x);
        ctx.save();
        ctx.translate(this.position.x, this.position.y);
        ctx.rotate(angle);

        ctx.beginPath();
        ctx.moveTo(this.size * 1.8, 0);
        ctx.lineTo(-this.size, this.size * 0.85);
        ctx.lineTo(-this.size * 0.45, 0);
        ctx.lineTo(-this.size, -this.size * 0.85);
        ctx.closePath();

        if (this.isOdd && oddBoidEnabled) {
          ctx.fillStyle = "#facc15";
          ctx.shadowColor = "#facc15";
          ctx.shadowBlur = 14;
        } else {
          ctx.fillStyle = "rgba(147, 197, 253, 0.95)";
          ctx.shadowColor = "rgba(96, 165, 250, 0.8)";
          ctx.shadowBlur = 8;
        }

        ctx.fill();
        ctx.restore();
      }
    }

    function resetBoids() {
      boids = [];
      for (let i = 0; i < settings.boidCount; i++) {
        const isOdd = i === 0;
        boids.push(new Boid(Math.random() * width, Math.random() * height, isOdd));
      }
    }

    function drawPredator() {
      if (!predatorEnabled) return;

      ctx.save();
      ctx.translate(predator.x, predator.y);
      ctx.beginPath();
      ctx.arc(0, 0, 18, 0, Math.PI * 2);
      ctx.fillStyle = "rgba(248, 113, 113, 0.95)";
      ctx.shadowColor = "rgba(248, 113, 113, 0.9)";
      ctx.shadowBlur = 20;
      ctx.fill();

      ctx.beginPath();
      ctx.arc(0, 0, settings.predatorRadius, 0, Math.PI * 2);
      ctx.strokeStyle = "rgba(248, 113, 113, 0.13)";
      ctx.lineWidth = 2;
      ctx.stroke();
      ctx.restore();
    }

    function updatePredator() {
      if (!predatorEnabled || pointerDown) return;

      predator.x += predator.vx;
      predator.y += predator.vy;

      if (predator.x < 30 || predator.x > width - 30) predator.vx *= -1;
      if (predator.y < 30 || predator.y > height - 30) predator.vy *= -1;
    }

    function drawObstacles() {
      for (const obstacle of obstacles) {
        ctx.save();
        ctx.beginPath();
        ctx.arc(obstacle.x, obstacle.y, obstacle.r, 0, Math.PI * 2);
        ctx.fillStyle = "rgba(148, 163, 184, 0.9)";
        ctx.shadowColor = "rgba(226, 232, 240, 0.6)";
        ctx.shadowBlur = 10;
        ctx.fill();

        ctx.beginPath();
        ctx.arc(obstacle.x, obstacle.y, obstacle.r + 48, 0, Math.PI * 2);
        ctx.strokeStyle = "rgba(203, 213, 225, 0.08)";
        ctx.lineWidth = 2;
        ctx.stroke();
        ctx.restore();
      }
    }

    function drawBackgroundTrails() {
      ctx.fillStyle = "rgba(5, 9, 18, 0.24)";
      ctx.fillRect(0, 0, width, height);
    }

    function animate() {
      drawBackgroundTrails();
      updatePredator();
      drawObstacles();
      drawPredator();

      for (const boid of boids) {
        boid.edges();
        boid.flock(boids);
        boid.avoidPredator();
        boid.avoidObstacles();
        boid.update();
        boid.draw();
      }

      updateLabels();
      requestAnimationFrame(animate);
    }

    function updateLabels() {
      cohesionValue.textContent = parseFloat(cohesionSlider.value).toFixed(2);
      separationValue.textContent = parseFloat(separationSlider.value).toFixed(2);
      alignmentValue.textContent = parseFloat(alignmentSlider.value).toFixed(2);
      predatorValue.textContent = parseFloat(predatorForceSlider.value).toFixed(2);
    }

    function getPointerPosition(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return {
        x: clientX - rect.left,
        y: clientY - rect.top,
      };
    }

    canvas.addEventListener("pointerdown", (event) => {
      pointerDown = true;
      pointerMoved = false;
      const p = getPointerPosition(event);
      mouse.x = p.x;
      mouse.y = p.y;
      predator.x = p.x;
      predator.y = p.y;
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!pointerDown) return;
      pointerMoved = true;
      const p = getPointerPosition(event);
      mouse.x = p.x;
      mouse.y = p.y;
      predator.x = p.x;
      predator.y = p.y;
    });

    canvas.addEventListener("pointerup", (event) => {
      const p = getPointerPosition(event);
      if (!pointerMoved) {
        obstacles.push({ x: p.x, y: p.y, r: settings.obstacleRadius });
      }
      pointerDown = false;
    });

    togglePredatorButton.addEventListener("click", () => {
      predatorEnabled = !predatorEnabled;
    });

    toggleOddButton.addEventListener("click", () => {
      oddBoidEnabled = !oddBoidEnabled;
    });

    clearObstaclesButton.addEventListener("click", () => {
      obstacles = [];
    });

    resetButton.addEventListener("click", () => {
      obstacles = [];
      resetBoids();
    });

    window.addEventListener("resize", () => {
      resize();
      resetBoids();
    });

    resize();
    resetBoids();
    updateLabels();
    animate();
  </script>
