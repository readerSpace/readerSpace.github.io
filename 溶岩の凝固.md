  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const coolRateSlider = document.getElementById("coolRate");
    const tempStartSlider = document.getElementById("tempStart");
    const nucleationSlider = document.getElementById("nucleation");
    const growthSlider = document.getElementById("growth");

    const chamber = { x: 90, y: 105, w: 560, h: 410 };
    let running = true;
    let temperature = Number(tempStartSlider.value);
    let crystals = [];
    let bubbles = [];
    let sparks = [];
    let history = [];
    let frame = 0;

    const liquidus = 1050;
    const solidus = 720;

    function clamp(v, min, max) {
      return Math.max(min, Math.min(max, v));
    }

    function quickCoolPreset() {
      coolRateSlider.value = 88;
      nucleationSlider.value = 85;
      growthSlider.value = 38;
      resetSim();
    }

    function slowCoolPreset() {
      coolRateSlider.value = 12;
      nucleationSlider.value = 35;
      growthSlider.value = 80;
      resetSim();
    }

    function reheat() {
      temperature = Number(tempStartSlider.value);
      for (const c of crystals) c.melting = 1;
      sparks.push(...Array.from({ length: 30 }, () => ({
        x: chamber.x + Math.random() * chamber.w,
        y: chamber.y + chamber.h - Math.random() * 40,
        vx: Math.random() * 2 - 1,
        vy: -1 - Math.random() * 3,
        life: 1
      })));
    }

    function resetSim() {
      running = true;
      temperature = Number(tempStartSlider.value);
      crystals = [];
      bubbles = [];
      sparks = [];
      history = [];
    }

    function solidFraction() {
      return clamp((liquidus - temperature) / (liquidus - solidus), 0, 1);
    }

    function stateName() {
      if (temperature > liquidus) return "完全に溶けている";
      if (temperature > solidus) return "結晶化中";
      return "ほぼ固まった";
    }

    function rockType() {
      const avg = averageCrystalSize();
      if (solidFraction() < 0.95) return "形成中";
      if (avg < 7) return "ガラス質〜細粒";
      if (avg < 16) return "細粒火成岩";
      if (avg < 30) return "中粒";
      return "粗粒火成岩";
    }

    function randomPointInChamber() {
      return {
        x: chamber.x + 20 + Math.random() * (chamber.w - 40),
        y: chamber.y + 20 + Math.random() * (chamber.h - 40)
      };
    }

    function canPlaceCrystal(x, y, r) {
      for (const c of crystals) {
        const dx = x - c.x;
        const dy = y - c.y;
        const minDist = r + c.r * 0.75;
        if (dx * dx + dy * dy < minDist * minDist) return false;
      }
      return true;
    }

    function nucleateCrystals() {
      if (temperature > liquidus || temperature < solidus - 30) return;

      const undercool = clamp((liquidus - temperature) / 260, 0, 1);
      const cooling = Number(coolRateSlider.value) / 100;
      const nucleation = Number(nucleationSlider.value) / 100;

      // 急冷ほど核生成が多い。徐冷では少なめ。
      const chance = (0.012 + cooling * 0.07) * nucleation * undercool;
      const attempts = 2 + Math.floor(cooling * 7);

      for (let i = 0; i < attempts; i++) {
        if (Math.random() < chance && crystals.length < 420) {
          const p = randomPointInChamber();
          const r = 1.8 + Math.random() * 2.5;
          if (canPlaceCrystal(p.x, p.y, r)) {
            crystals.push({
              x: p.x,
              y: p.y,
              r,
              targetR: r,
              angle: Math.random() * Math.PI,
              sides: 5 + Math.floor(Math.random() * 4),
              hue: 28 + Math.random() * 38,
              growthBias: 0.75 + Math.random() * 0.8,
              melting: 0
            });
          }
        }
      }
    }

    function growCrystals() {
      if (temperature > liquidus) {
        // 再加熱時は結晶が溶ける
        for (const c of crystals) {
          c.r *= 0.985;
          c.melting = 1;
        }
        crystals = crystals.filter(c => c.r > 1.2);
        return;
      }

      if (temperature < solidus - 60) return;

      const cooling = Number(coolRateSlider.value) / 100;
      const growth = Number(growthSlider.value) / 100;
      const tempWindow = clamp(1 - Math.abs(temperature - 900) / 260, 0.1, 1);

      // 徐冷ほど1ステップあたりの成長時間が長い、という見せ方
      const slowFactor = 1.25 - cooling * 0.95;
      const baseGrow = 0.018 * growth * slowFactor * tempWindow;

      for (const c of crystals) {
        let blocked = false;
        for (const other of crystals) {
          if (other === c) continue;
          const dx = c.x - other.x;
          const dy = c.y - other.y;
          const limit = c.r + other.r * 0.95;
          if (dx * dx + dy * dy < limit * limit) {
            blocked = true;
            break;
          }
        }
        if (!blocked) {
          const roomX = Math.min(c.x - chamber.x, chamber.x + chamber.w - c.x);
          const roomY = Math.min(c.y - chamber.y, chamber.y + chamber.h - c.y);
          const room = Math.min(roomX, roomY);
          if (c.r < room - 5) c.r += baseGrow * c.growthBias;
        }
        c.melting *= 0.94;
      }
    }

    function updateBubblesAndSparks() {
      const hot = clamp((temperature - 800) / 500, 0, 1);
      if (temperature > 760 && Math.random() < hot * 0.45) {
        bubbles.push({
          x: chamber.x + 30 + Math.random() * (chamber.w - 60),
          y: chamber.y + chamber.h - 10,
          r: 3 + Math.random() * 8,
          vy: 0.7 + Math.random() * 1.8,
          life: 1
        });
      }
      bubbles = bubbles
        .map(b => ({ ...b, y: b.y - b.vy, r: b.r * 1.002, life: b.life - 0.006 }))
        .filter(b => b.life > 0 && b.y > chamber.y + 5);

      sparks = sparks
        .map(s => ({ ...s, x: s.x + s.vx, y: s.y + s.vy, vy: s.vy + 0.035, life: s.life - 0.018 }))
        .filter(s => s.life > 0);
    }

    function updateModel() {
      if (running && temperature > 300) {
        const cool = Number(coolRateSlider.value);
        const environmentPull = clamp((temperature - 25) / 1200, 0.1, 1);
        temperature -= (0.035 + cool * 0.0065) * environmentPull;
      }

      nucleateCrystals();
      growCrystals();
      updateBubblesAndSparks();

      if (frame % 6 === 0) {
        history.push({ temp: temperature, solid: solidFraction() * 100, avg: averageCrystalSize() });
        if (history.length > 130) history.shift();
      }
    }

    function averageCrystalSize() {
      if (crystals.length === 0) return 0;
      return crystals.reduce((sum, c) => sum + c.r * 2, 0) / crystals.length;
    }

    function drawBackground() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      const g = ctx.createLinearGradient(0, 0, 0, canvas.height);
      g.addColorStop(0, "#1c120f");
      g.addColorStop(1, "#090504");
      ctx.fillStyle = g;
      ctx.fillRect(0, 0, canvas.width, canvas.height);

      ctx.fillStyle = "#2a1810";
      ctx.fillRect(0, 530, canvas.width, 110);
      ctx.fillStyle = "rgba(0,0,0,0.25)";
      for (let i = 0; i < 80; i++) {
        ctx.beginPath();
        ctx.arc((i * 83) % canvas.width, 540 + ((i * 47) % 90), 1.5, 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function lavaColor() {
      const hot = clamp((temperature - 500) / 800, 0, 1);
      const solid = solidFraction();
      const r = Math.round(70 + hot * 185 - solid * 45);
      const g = Math.round(35 + hot * 90 - solid * 20);
      const b = Math.round(25 + hot * 15 - solid * 10);
      return `rgb(${clamp(r,30,255)}, ${clamp(g,20,150)}, ${clamp(b,15,80)})`;
    }

    function drawChamber() {
      ctx.fillStyle = "rgba(0,0,0,0.38)";
      ctx.beginPath();
      ctx.roundRect(chamber.x - 14, chamber.y - 14, chamber.w + 28, chamber.h + 28, 24);
      ctx.fill();

      const grad = ctx.createRadialGradient(chamber.x + chamber.w * 0.35, chamber.y + chamber.h * 0.3, 10, chamber.x + chamber.w/2, chamber.y + chamber.h/2, chamber.w * 0.65);
      grad.addColorStop(0, lavaColor());
      grad.addColorStop(1, temperature > 700 ? "#3b160d" : "#2b2521");
      ctx.fillStyle = grad;
      ctx.beginPath();
      ctx.roundRect(chamber.x, chamber.y, chamber.w, chamber.h, 22);
      ctx.fill();

      // glowing convection streaks
      if (temperature > 740) {
        ctx.save();
        ctx.beginPath();
        ctx.roundRect(chamber.x, chamber.y, chamber.w, chamber.h, 22);
        ctx.clip();
        for (let i = 0; i < 16; i++) {
          const x = chamber.x + ((i * 73 + frame * 0.7) % chamber.w);
          const y = chamber.y + 40 + ((i * 39) % (chamber.h - 80));
          ctx.strokeStyle = `rgba(255, ${120 + i * 4}, 40, ${0.08 + clamp((temperature-750)/500,0,1)*0.2})`;
          ctx.lineWidth = 12;
          ctx.beginPath();
          ctx.moveTo(x - 80, y + Math.sin(frame * 0.02 + i) * 12);
          ctx.quadraticCurveTo(x, y - 30, x + 90, y + 20);
          ctx.stroke();
        }
        ctx.restore();
      }

      ctx.strokeStyle = "rgba(254,215,170,0.45)";
      ctx.lineWidth = 3;
      ctx.beginPath();
      ctx.roundRect(chamber.x, chamber.y, chamber.w, chamber.h, 22);
      ctx.stroke();
    }

    function drawCrystals() {
      ctx.save();
      ctx.beginPath();
      ctx.roundRect(chamber.x, chamber.y, chamber.w, chamber.h, 22);
      ctx.clip();

      for (const c of crystals) {
        const alpha = clamp(0.35 + solidFraction() * 0.65 - c.melting * 0.5, 0.1, 1);
        drawCrystal(c.x, c.y, c.r, c.sides, c.angle, c.hue, alpha);
      }

      ctx.restore();
    }

    function drawCrystal(x, y, r, sides, angle, hue, alpha) {
      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(angle);
      ctx.beginPath();
      for (let i = 0; i < sides; i++) {
        const a = (i / sides) * Math.PI * 2;
        const rr = r * (0.82 + 0.25 * Math.sin(i * 2.1));
        const px = Math.cos(a) * rr;
        const py = Math.sin(a) * rr;
        if (i === 0) ctx.moveTo(px, py);
        else ctx.lineTo(px, py);
      }
      ctx.closePath();
      ctx.fillStyle = `hsla(${hue}, 70%, ${55 + r * 0.5}%, ${alpha})`;
      ctx.fill();
      ctx.strokeStyle = `rgba(255,255,255,${0.25 * alpha})`;
      ctx.lineWidth = 1.2;
      ctx.stroke();

      ctx.strokeStyle = `rgba(255,255,255,${0.15 * alpha})`;
      ctx.beginPath();
      ctx.moveTo(-r * 0.45, 0);
      ctx.lineTo(r * 0.45, 0);
      ctx.moveTo(0, -r * 0.45);
      ctx.lineTo(0, r * 0.45);
      ctx.stroke();
      ctx.restore();
    }

    function drawBubblesAndSparks() {
      for (const b of bubbles) {
        ctx.strokeStyle = `rgba(255,244,220,${0.45 * b.life})`;
        ctx.lineWidth = 2;
        ctx.beginPath();
        ctx.arc(b.x, b.y, b.r, 0, Math.PI * 2);
        ctx.stroke();
      }

      for (const s of sparks) {
        ctx.fillStyle = `rgba(251,191,36,${s.life})`;
        ctx.beginPath();
        ctx.arc(s.x, s.y, 3, 0, Math.PI * 2);
        ctx.fill();
      }
    }

    function drawThermometer() {
      const x = 705;
      const y = 105;
      const h = 410;
      const ratio = clamp((temperature - 300) / 1100, 0, 1);

      ctx.fillStyle = "rgba(255,255,255,0.08)";
      ctx.beginPath();
      ctx.roundRect(x - 28, y - 18, 90, h + 52, 16);
      ctx.fill();

      ctx.strokeStyle = "#fed7aa";
      ctx.lineWidth = 4;
      ctx.strokeRect(x, y, 24, h);
      ctx.fillStyle = temperature > liquidus ? "#f97316" : temperature > solidus ? "#facc15" : "#94a3b8";
      ctx.fillRect(x + 4, y + h - ratio * h, 16, ratio * h);

      ctx.fillStyle = "#fff7ed";
      ctx.font = "14px system-ui";
      ctx.fillText("温度", x - 4, y - 28);
      ctx.fillText(`${temperature.toFixed(0)}℃`, x - 16, y + h + 30);

      const ly = y + h - clamp((liquidus - 300) / 1100, 0, 1) * h;
      const sy = y + h - clamp((solidus - 300) / 1100, 0, 1) * h;
      ctx.strokeStyle = "rgba(255,255,255,0.7)";
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(x - 8, ly);
      ctx.lineTo(x + 38, ly);
      ctx.moveTo(x - 8, sy);
      ctx.lineTo(x + 38, sy);
      ctx.stroke();
      ctx.font = "11px system-ui";
      ctx.fillText("結晶化開始", x + 45, ly + 4);
      ctx.fillText("ほぼ固化", x + 45, sy + 4);
    }

    function drawGraph() {
      const x = 90;
      const y = 540;
      const w = 560;
      const h = 75;
      ctx.fillStyle = "rgba(255,255,255,0.08)";
      ctx.beginPath();
      ctx.roundRect(x, y, w, h, 12);
      ctx.fill();
      ctx.strokeStyle = "rgba(254,215,170,0.25)";
      ctx.stroke();

      ctx.fillStyle = "#ffedd5";
      ctx.font = "12px system-ui";
      ctx.fillText("履歴：温度と固化率", x + 10, y + 18);
      drawLine(history.map(d => d.temp), x, y, w, h, 300, 1400, "#f97316");
      drawLine(history.map(d => d.solid), x, y, w, h, 0, 100, "#94a3b8");
    }

    function drawLine(values, x, y, w, h, min, max, color) {
      if (values.length < 2) return;
      ctx.strokeStyle = color;
      ctx.lineWidth = 2;
      ctx.beginPath();
      values.forEach((v, i) => {
        const px = x + 8 + (i / 129) * (w - 16);
        const py = y + h - 8 - clamp((v - min) / (max - min), 0, 1) * (h - 30);
        if (i === 0) ctx.moveTo(px, py);
        else ctx.lineTo(px, py);
      });
      ctx.stroke();
    }

    function drawLabels() {
      ctx.fillStyle = "#fff7ed";
      ctx.font = "19px system-ui";
      ctx.fillText("冷却速度で結晶サイズが変わる", 34, 40);

      ctx.fillStyle = "#fed7aa";
      ctx.font = "14px system-ui";
      const cool = Number(coolRateSlider.value);
      const message = cool > 65
        ? "急冷：結晶が育つ前に固まり、小さい結晶・ガラス質になりやすい"
        : cool < 25
          ? "徐冷：時間をかけて結晶が成長し、大きな結晶になりやすい"
          : "中くらいの冷却：細粒〜中粒の結晶ができる";
      ctx.fillText(message, 34, 66);

      ctx.fillStyle = "rgba(255,255,255,0.1)";
      ctx.beginPath();
      ctx.roundRect(670, 540, 160, 74, 12);
      ctx.fill();
      ctx.fillStyle = "#ffedd5";
      ctx.font = "13px system-ui";
      ctx.fillText("色の意味", 686, 562);
      ctx.fillText("橙：熱い溶岩", 686, 584);
      ctx.fillText("明るい多角形：結晶", 686, 604);
    }

    function updateUI() {
      const cool = Number(coolRateSlider.value);
      document.getElementById("coolText").textContent = `${cool.toFixed(0)} %`;
      document.getElementById("tempStartText").textContent = `${tempStartSlider.value} ℃`;
      document.getElementById("nucleationText").textContent = `${nucleationSlider.value} %`;
      document.getElementById("growthText").textContent = `${growthSlider.value} %`;
      document.getElementById("tempStat").textContent = `${temperature.toFixed(0)} ℃`;
      document.getElementById("stateStat").textContent = stateName();
      document.getElementById("solidStat").textContent = `${(solidFraction() * 100).toFixed(0)} %`;
      document.getElementById("crystalCountStat").textContent = `${crystals.length}`;
      document.getElementById("avgSizeStat").textContent = `${averageCrystalSize().toFixed(1)} px`;
      document.getElementById("rockTypeStat").textContent = rockType();
    }

    function draw() {
      drawBackground();
      drawChamber();
      drawCrystals();
      drawBubblesAndSparks();
      drawThermometer();
      drawGraph();
      drawLabels();
    }

    function loop() {
      frame++;
      updateModel();
      updateUI();
      draw();
      requestAnimationFrame(loop);
    }

    resetSim();
    loop();
  </script>
