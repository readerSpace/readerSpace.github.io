レーザーの物理原理を確認しつつ、教育用に「反転分布→誘導放出→光増幅→しきい値発振」が見えるJSシミュレーションとしてまとめます。

思考時間: 6s

以下は「レーザー発振のしきい値」が見える簡易シミュレーションです。レーザーは、励起で反転分布を作り、光子が来たときに同じ位相・方向の光子を出す誘導放出で光を増幅し、鏡のある光共振器で何度も往復させることで発振します。反転分布が損失を上回るとレーザー発振が始まります

<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

const pumpSlider = document.getElementById("pump");
const lossSlider = document.getElementById("loss");
const pumpVal = document.getElementById("pumpVal");
const lossVal = document.getElementById("lossVal");

let atoms = [];
let photons = [];

const N = 90;

for (let i = 0; i < N; i++) {
  atoms.push({
    x: 130 + Math.random() * 540,
    y: 90 + Math.random() * 220,
    excited: Math.random() < 0.2
  });
}

function addPhoton(x, y, dir = 1) {
  photons.push({
    x, y,
    vx: dir * (2.5 + Math.random()),
    phase: Math.random() * Math.PI * 2
  });
}

for (let i = 0; i < 6; i++) {
  addPhoton(200 + Math.random() * 400, 120 + Math.random() * 180, Math.random() < 0.5 ? 1 : -1);
}

function loop() {
  const pump = parseFloat(pumpSlider.value);
  const loss = parseFloat(lossSlider.value);

  pumpVal.textContent = pump.toFixed(2);
  lossVal.textContent = loss.toFixed(2);

  ctx.clearRect(0, 0, canvas.width, canvas.height);

  // 共振器の鏡
  ctx.fillStyle = "#ccc";
  ctx.fillRect(90, 70, 10, 260);
  ctx.fillRect(700, 70, 10, 260);

  ctx.fillStyle = "white";
  ctx.fillText("全反射鏡", 60, 355);
  ctx.fillText("半透鏡", 685, 355);

  // ポンプによる励起
  for (const atom of atoms) {
    if (!atom.excited && Math.random() < pump * 0.015) {
      atom.excited = true;
    }

    // 自然放出
    if (atom.excited && Math.random() < 0.0015) {
      atom.excited = false;
      addPhoton(atom.x, atom.y, Math.random() < 0.5 ? 1 : -1);
    }
  }

  // 光子の移動
  for (const p of photons) {
    p.x += p.vx;
    p.phase += 0.15;

    // 鏡で反射
    if (p.x < 100) {
      p.x = 100;
      p.vx *= -1;
    }

    if (p.x > 700) {
      // 半透鏡：一部は外へ出る、一部は反射
      if (Math.random() < 0.22) {
        p.x = 720; // 出力光
      } else {
        p.x = 700;
        p.vx *= -1;
      }
    }

    // 損失
    if (Math.random() < loss * 0.004) {
      p.dead = true;
    }

    // 誘導放出
    for (const atom of atoms) {
      const dx = atom.x - p.x;
      const dy = atom.y - p.y;
      const d2 = dx * dx + dy * dy;

      if (atom.excited && d2 < 80) {
        atom.excited = false;

        // 入射光子と同じ方向の光子を追加
        photons.push({
          x: atom.x,
          y: atom.y,
          vx: p.vx,
          phase: p.phase
        });
      }
    }
  }

  photons = photons.filter(p => !p.dead && p.x < 850 && p.x > -50);

  // 原子を描画
  for (const atom of atoms) {
    ctx.beginPath();
    ctx.arc(atom.x, atom.y, 5, 0, Math.PI * 2);
    ctx.fillStyle = atom.excited ? "#ff5555" : "#4477ff";
    ctx.fill();
  }

  // 光子を描画
  for (const p of photons) {
    ctx.beginPath();
    ctx.arc(p.x, p.y + Math.sin(p.phase) * 3, 3, 0, Math.PI * 2);
    ctx.fillStyle = "#ffff66";
    ctx.fill();

    ctx.beginPath();
    ctx.moveTo(p.x - 8, p.y);
    ctx.lineTo(p.x + 8, p.y);
    ctx.strokeStyle = "#ffff66";
    ctx.stroke();
  }

  // 統計表示
  const excitedCount = atoms.filter(a => a.excited).length;
  const threshold = pump > loss * 2.4;

  ctx.fillStyle = "white";
  ctx.fillText("励起原子数: " + excitedCount, 30, 30);
  ctx.fillText("光子数: " + photons.length, 30, 50);

  ctx.fillStyle = threshold ? "#00ff99" : "#ffaa00";
  ctx.fillText(
    threshold ? "発振しやすい状態：利得 > 損失" : "まだ弱い：利得 < 損失",
    300, 30
  );

  // 光子数が増えすぎると重くなるので制限
  if (photons.length > 900) {
    photons = photons.slice(photons.length - 900);
  }

  requestAnimationFrame(loop);
}

loop();
</script>

ポイントは、赤い点＝励起原子、黄色＝光子です。ポンプ強度を上げると励起原子が増え、光子が通過したときに同じ向きの光子が増えます。これが誘導放出による光増幅です。
