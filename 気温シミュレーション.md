<script>
const canvas = document.getElementById("canvas");
const ctx = canvas.getContext("2d");

const ghg = document.getElementById("ghg");
const distance = document.getElementById("distance");
const cloud = document.getElementById("cloud");
const ocean = document.getElementById("ocean");

const ghgText = document.getElementById("ghgText");
const distanceText = document.getElementById("distanceText");
const cloudText = document.getElementById("cloudText");
const oceanText = document.getElementById("oceanText");

const tempStat = document.getElementById("tempStat");
const climateStat = document.getElementById("climateStat");
const solarStat = document.getElementById("solarStat");
const iceStat = document.getElementById("iceStat");

const earthBtn = document.getElementById("earthBtn");
const snowBtn = document.getElementById("snowBtn");
const hotBtn = document.getElementById("hotBtn");

let angle = 0;

function lerp(a, b, t) {
  return a + (b - a) * t;
}

function clamp(x, a, b) {
  return Math.max(a, Math.min(b, x));
}

function computeClimate() {
  const ghgLevel = Number(ghg.value);
  const distAU = Number(distance.value);
  const cloudiness = Number(cloud.value) / 100;
  const oceanRatio = Number(ocean.value) / 100;

  const solarConstant = 1361 / (distAU * distAU);

  let albedo = 0.3 + cloudiness * 0.18;

  const greenhouse = 33 * Math.log2(1 + ghgLevel);

  let equilibrium = Math.pow((solarConstant * (1 - albedo)) / (4 * 5.67e-8), 0.25) - 273.15;

  equilibrium += greenhouse;

  equilibrium = lerp(equilibrium, 15, oceanRatio * 0.18);

  let ice = clamp((5 - equilibrium) / 35, 0, 1);

  equilibrium -= ice * 9;

  ice = clamp((5 - equilibrium) / 35, 0, 1);

  return {
    temp: equilibrium,
    solar: solarConstant,
    ice,
    cloudiness,
    oceanRatio,
    ghgLevel,
  };
}

function climateName(temp) {
  if (temp < -30) return "全球凍結";
  if (temp < -5) return "氷河世界";
  if (temp < 10) return "寒冷";
  if (temp < 25) return "温暖";
  if (temp < 40) return "高温";
  return "灼熱";
}

function drawStars() {
  for (let i = 0; i < 140; i++) {
    const x = (i * 137.7) % canvas.width;
    const y = (i * 91.3) % canvas.height;
    const r = (i % 3) + 1;
    ctx.fillStyle = `rgba(255,255,255,${0.15 + (i % 10) / 12})`;
    ctx.beginPath();
    ctx.arc(x, y, r * 0.5, 0, Math.PI * 2);
    ctx.fill();
  }
}

function drawSun(distanceAU) {
  const x = 120;
  const y = 120;
  const size = clamp(140 / distanceAU, 55, 180);

  const glow = ctx.createRadialGradient(x, y, 10, x, y, size * 1.8);
  glow.addColorStop(0, "rgba(253,224,71,0.95)");
  glow.addColorStop(0.5, "rgba(251,191,36,0.35)");
  glow.addColorStop(1, "rgba(251,191,36,0)");
  ctx.fillStyle = glow;
  ctx.beginPath();
  ctx.arc(x, y, size * 1.8, 0, Math.PI * 2);
  ctx.fill();

  ctx.fillStyle = "#fde047";
  ctx.beginPath();
  ctx.arc(x, y, size * 0.45, 0, Math.PI * 2);
  ctx.fill();
}

function drawEarth(climate) {
  const cx = canvas.width * 0.64;
  const cy = canvas.height * 0.52;
  const R = 170;

  const oceanRatio = climate.oceanRatio;
  const ice = climate.ice;
  const temp = climate.temp;

  const oceanBlue = `rgb(${Math.round(20 + ice * 60)}, ${Math.round(90 + ice * 40)}, ${Math.round(170 + ice * 45)})`;

  const gradient = ctx.createRadialGradient(cx - 60, cy - 60, 20, cx, cy, R);
  gradient.addColorStop(0, "#bfdbfe");
  gradient.addColorStop(0.35, oceanBlue);
  gradient.addColorStop(1, "#082f49");

  ctx.fillStyle = gradient;
  ctx.beginPath();
  ctx.arc(cx, cy, R, 0, Math.PI * 2);
  ctx.fill();

  ctx.save();
  ctx.beginPath();
  ctx.arc(cx, cy, R, 0, Math.PI * 2);
  ctx.clip();

  drawContinents(cx, cy, R, climate);
  drawIce(cx, cy, R, climate);
  drawClouds(cx, cy, R, climate);
  drawAtmosphere(cx, cy, R, climate);

  ctx.restore();

  ctx.strokeStyle = "rgba(255,255,255,0.15)";
  ctx.lineWidth = 3;
  ctx.beginPath();
  ctx.arc(cx, cy, R, 0, Math.PI * 2);
  ctx.stroke();
}

function drawContinents(cx, cy, R, climate) {
  const temp = climate.temp;
  const desert = clamp((temp - 25) / 25, 0, 1);

  const green = {
    r: lerp(34, 194, desert),
    g: lerp(139, 120, desert),
    b: lerp(34, 70, desert)
  };

  ctx.fillStyle = `rgb(${green.r|0},${green.g|0},${green.b|0})`;

  const shift = angle * 0.18;

  for (let i = 0; i < 5; i++) {
    const a = shift + i * 1.4;
    const x = cx + Math.cos(a) * (R * 0.45);
    const y = cy + Math.sin(a * 1.3) * (R * 0.28);

    ctx.beginPath();
    ctx.moveTo(x - 30, y - 18);
    ctx.bezierCurveTo(x + 60, y - 90, x + 80, y + 40, x - 10, y + 55);
    ctx.bezierCurveTo(x - 80, y + 25, x - 70, y - 60, x - 30, y - 18);
    ctx.fill();
  }
}

function drawIce(cx, cy, R, climate) {
  const ice = climate.ice;
  const polar = clamp(ice * 1.5, 0, 1);

  ctx.fillStyle = `rgba(255,255,255,${0.4 + polar * 0.55})`;

  const cap = R * polar * 0.75;

  ctx.beginPath();
  ctx.ellipse(cx, cy - R + cap * 0.6, R * 0.9, cap, 0, 0, Math.PI * 2);
  ctx.fill();

  ctx.beginPath();
  ctx.ellipse(cx, cy + R - cap * 0.6, R * 0.9, cap, 0, 0, Math.PI * 2);
  ctx.fill();

  if (ice > 0.45) {
    ctx.fillStyle = `rgba(255,255,255,${(ice - 0.4) * 0.9})`;
    for (let i = 0; i < 14; i++) {
      const a = i * 0.45 + angle * 0.02;
      const x = cx + Math.cos(a) * R * 0.5;
      const y = cy + Math.sin(a * 1.7) * R * 0.42;
      ctx.beginPath();
      ctx.arc(x, y, 12 + ice * 20, 0, Math.PI * 2);
      ctx.fill();
    }
  }
}

function drawClouds(cx, cy, R, climate) {
  const c = climate.cloudiness;
  ctx.fillStyle = `rgba(255,255,255,${0.12 + c * 0.45})`;

  for (let i = 0; i < 28; i++) {
    const a = i * 0.36 + angle * 0.01;
    const rr = R * (0.2 + (i % 7) / 10);
    const x = cx + Math.cos(a * 1.2) * rr;
    const y = cy + Math.sin(a * 1.8) * rr * 0.65;

    ctx.beginPath();
    ctx.ellipse(x, y, 22 + c * 26, 10 + c * 14, a, 0, Math.PI * 2);
    ctx.fill();
  }
}

function drawAtmosphere(cx, cy, R, climate) {
  const temp = climate.temp;

  let color;
  if (temp < -5) color = "rgba(147,197,253,0.26)";
  else if (temp < 30) color = "rgba(56,189,248,0.18)";
  else color = "rgba(251,146,60,0.22)";

  ctx.strokeStyle = color;
  ctx.lineWidth = 18;
  ctx.beginPath();
  ctx.arc(cx, cy, R + 10, 0, Math.PI * 2);
  ctx.stroke();
}

function update() {
  angle += 0.01;

  ghgText.textContent = `${Number(ghg.value).toFixed(2)} × 現代地球`;
  distanceText.textContent = `${Number(distance.value).toFixed(2)} AU`;
  cloudText.textContent = `${cloud.value}%`;
  oceanText.textContent = `${ocean.value}%`;

  const climate = computeClimate();

  tempStat.textContent = `${climate.temp.toFixed(1)}°C`;
  climateStat.textContent = climateName(climate.temp);
  solarStat.textContent = `${climate.solar.toFixed(0)} W/m²`;
  iceStat.textContent = `${(climate.ice * 100).toFixed(0)}%`;

  ctx.clearRect(0, 0, canvas.width, canvas.height);
  drawStars();
  drawSun(Number(distance.value));
  drawEarth(climate);

  requestAnimationFrame(update);
}

function setEarth() {
  ghg.value = 1;
  distance.value = 1;
  cloud.value = 40;
  ocean.value = 70;
}

function setSnowball() {
  ghg.value = 0.2;
  distance.value = 1.4;
  cloud.value = 65;
  ocean.value = 60;
}

function setHot() {
  ghg.value = 4.2;
  distance.value = 0.72;
  cloud.value = 18;
  ocean.value = 35;
}

earthBtn.addEventListener("click", setEarth);
snowBtn.addEventListener("click", setSnowball);
hotBtn.addEventListener("click", setHot);

update();
</script>
