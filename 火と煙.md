  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");
    const clearButton = document.getElementById("clear");
    const campfireButton = document.getElementById("campfire");
    const sparkButton = document.getElementById("spark");
    const heatSlider = document.getElementById("heat");
    const smokeSlider = document.getElementById("smoke");
    const buoyancySlider = document.getElementById("buoyancy");
    const windSlider = document.getElementById("wind");
    const brushSlider = document.getElementById("brush");
    const heatValue = document.getElementById("heatValue");
    const smokeValue = document.getElementById("smokeValue");
    const buoyancyValue = document.getElementById("buoyancyValue");
    const windValue = document.getElementById("windValue");
    const brushValue = document.getElementById("brushValue");
    const particleCountSpan = document.getElementById("particleCount");
    const resolutionSpan = document.getElementById("resolution");

    let width = 0;
    let height = 0;
    let scale = 4;
    let gridW = 0;
    let gridH = 0;
    let heat, smoke, vx, vy, walls;
    let nextHeat, nextSmoke, nextVx, nextVy;
    let imageData;
    let renderCanvas = document.createElement("canvas");
    let renderCtx = renderCanvas.getContext("2d");
    let particles = [];
    let running = true;
    let drawing = false;
    let tool = "fire";
    let lastPointer = null;
    let time = 0;

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

      scale = width < 700 ? 5 : 4;
      gridW = Math.floor(width / scale);
      gridH = Math.floor(height / scale);
      renderCanvas.width = gridW;
      renderCanvas.height = gridH;
      imageData = renderCtx.createImageData(gridW, gridH);
      resetWorld();
    }

    function idx(x, y) { return y * gridW + x; }
    function inside(x, y) { return x >= 1 && x < gridW - 1 && y >= 1 && y < gridH - 1; }
    function clamp(v, min, max) { return Math.max(min, Math.min(max, v)); }

    function resetWorld() {
      const n = gridW * gridH;
      heat = new Float32Array(n);
      smoke = new Float32Array(n);
      vx = new Float32Array(n);
      vy = new Float32Array(n);
      walls = new Uint8Array(n);
      nextHeat = new Float32Array(n);
      nextSmoke = new Float32Array(n);
      nextVx = new Float32Array(n);
      nextVy = new Float32Array(n);
      particles = [];
      time = 0;
    }

    function addFire(cx, cy, radius, strength) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y) || walls[idx(x, y)]) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const falloff = 1 - d / (r + 1);
            const i = idx(x, y);
            heat[i] = Math.min(255, heat[i] + strength * falloff);
            smoke[i] = Math.min(255, smoke[i] + parseFloat(smokeSlider.value) * 0.35 * falloff);
            vy[i] -= 1.2 * falloff;
            vx[i] += (Math.random() - 0.5) * 0.8 * falloff;
          }
        }
      }

      for (let i = 0; i < 4; i++) {
        particles.push(new Spark(cx * scale, cy * scale, true));
      }
      trimParticles();
    }

    function addSmoke(cx, cy, radius, amount) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y) || walls[idx(x, y)]) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const falloff = 1 - d / (r + 1);
            const i = idx(x, y);
            smoke[i] = Math.min(255, smoke[i] + amount * falloff);
            vy[i] -= 0.45 * falloff;
          }
        }
      }
    }

    function coolArea(cx, cy, radius) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y)) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const i = idx(x, y);
            heat[i] *= 0.3;
            smoke[i] *= 0.75;
            vx[i] *= 0.4;
            vy[i] *= 0.4;
          }
        }
      }
    }

    function paintWall(cx, cy, radius, value) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y)) continue;
          if (Math.hypot(dx, dy) <= r) {
            const i = idx(x, y);
            walls[i] = value;
            if (value) {
              heat[i] = smoke[i] = vx[i] = vy[i] = 0;
            }
          }
        }
      }
    }

    function stepSimulation() {
      const wind = parseFloat(windSlider.value);
      const buoyancy = parseFloat(buoyancySlider.value);

      nextHeat.fill(0);
      nextSmoke.fill(0);
      nextVx.fill(0);
      nextVy.fill(0);

      for (let y = 1; y < gridH - 1; y++) {
        for (let x = 1; x < gridW - 1; x++) {
          const i = idx(x, y);
          if (walls[i]) continue;

          const h = heat[i];
          const s = smoke[i];

          // 熱いほど上にのぼる。横風も少し加える。
          vx[i] += wind * 0.012 + (Math.random() - 0.5) * 0.015;
          vy[i] += -buoyancy * h / 255 * 0.055 - s / 255 * 0.012;

          vx[i] *= 0.992;
          vy[i] *= 0.992;

          const backX = clamp(x - vx[i] * 0.7, 1, gridW - 2);
          const backY = clamp(y - vy[i] * 0.7, 1, gridH - 2);
          const sample = bilinear(backX, backY);

          const diffusionHeat = average(heat, x, y) * 0.06;
          const diffusionSmoke = average(smoke, x, y) * 0.035;

          nextHeat[i] = Math.max(0, sample.h * 0.965 + diffusionHeat - 0.28);
          nextSmoke[i] = Math.max(0, sample.s * 0.992 + diffusionSmoke - 0.035);
          nextVx[i] = sample.vx;
          nextVy[i] = sample.vy;
        }
      }

      let tmp = heat; heat = nextHeat; nextHeat = tmp;
      tmp = smoke; smoke = nextSmoke; nextSmoke = tmp;
      tmp = vx; vx = nextVx; nextVx = tmp;
      tmp = vy; vy = nextVy; nextVy = tmp;

      updateParticles();
      time++;
    }

    function average(arr, x, y) {
      return (arr[idx(x - 1, y)] + arr[idx(x + 1, y)] + arr[idx(x, y - 1)] + arr[idx(x, y + 1)]) * 0.25;
    }

    function bilinear(x, y) {
      const x0 = Math.floor(x), y0 = Math.floor(y);
      const x1 = Math.min(gridW - 1, x0 + 1), y1 = Math.min(gridH - 1, y0 + 1);
      const tx = x - x0, ty = y - y0;
      const i00 = idx(x0, y0), i10 = idx(x1, y0), i01 = idx(x0, y1), i11 = idx(x1, y1);

      function interp(arr) {
        const a = arr[i00] * (1 - tx) + arr[i10] * tx;
        const b = arr[i01] * (1 - tx) + arr[i11] * tx;
        return a * (1 - ty) + b * ty;
      }

      return { h: interp(heat), s: interp(smoke), vx: interp(vx), vy: interp(vy) };
    }

    class Spark {
      constructor(x, y, hot = false) {
        this.x = x;
        this.y = y;
        this.vx = (Math.random() - 0.5) * 3.2;
        this.vy = hot ? -2.8 - Math.random() * 4.2 : -0.8 - Math.random() * 1.6;
        this.life = 1;
        this.size = hot ? 1.8 + Math.random() * 2.2 : 1.2 + Math.random() * 1.5;
      }

      update() {
        const gx = Math.floor(this.x / scale);
        const gy = Math.floor(this.y / scale);
        if (inside(gx, gy)) {
          const i = idx(gx, gy);
          this.vx += vx[i] * 0.07;
          this.vy += vy[i] * 0.07;
        }
        this.vy += 0.015;
        this.x += this.vx;
        this.y += this.vy;
        this.life *= 0.975;
      }

      draw() {
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.size * this.life, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(255, ${Math.floor(120 + 100 * this.life)}, 30, ${this.life})`;
        ctx.shadowColor = "#fb923c";
        ctx.shadowBlur = 8;
        ctx.fill();
      }
    }

    function updateParticles() {
      for (const p of particles) p.update();
      particles = particles.filter(p => p.life > 0.06 && p.x > -20 && p.x < width + 20 && p.y > -40 && p.y < height + 20);
    }

    function trimParticles() {
      if (particles.length > 1200) particles.splice(0, particles.length - 1200);
    }

    function render() {
      const data = imageData.data;
      for (let y = 0; y < gridH; y++) {
        for (let x = 0; x < gridW; x++) {
          const i = idx(x, y);
          const p = i * 4;

          if (walls[i]) {
            data[p] = 130;
            data[p + 1] = 140;
            data[p + 2] = 155;
            data[p + 3] = 255;
            continue;
          }

          const h = clamp(heat[i], 0, 255);
          const s = clamp(smoke[i], 0, 255);
          const glow = Math.pow(h / 255, 0.65);
          const sm = Math.pow(s / 255, 0.85);

          let r = 5 + sm * 95 + glow * 255;
          let g = 12 + sm * 95 + glow * 150;
          let bcol = 28 + sm * 105 + glow * 18;

          // 白熱部分
          if (h > 190) {
            r = 255;
            g = 225;
            bcol = 120 + (h - 190) * 1.4;
          }

          data[p] = clamp(r, 0, 255);
          data[p + 1] = clamp(g, 0, 255);
          data[p + 2] = clamp(bcol, 0, 255);
          data[p + 3] = 255;
        }
      }

      renderCtx.putImageData(imageData, 0, 0);
      ctx.imageSmoothingEnabled = false;
      ctx.drawImage(renderCanvas, 0, 0, width, height);

      for (const p of particles) p.draw();
      updateLabels();
    }

    function animate() {
      if (running) {
        for (let i = 0; i < 2; i++) stepSimulation();
      }
      render();
      requestAnimationFrame(animate);
    }

    function updateLabels() {
      heatValue.textContent = heatSlider.value;
      smokeValue.textContent = smokeSlider.value;
      buoyancyValue.textContent = parseFloat(buoyancySlider.value).toFixed(2);
      windValue.textContent = parseFloat(windSlider.value).toFixed(2);
      brushValue.textContent = `${brushSlider.value}px`;
      particleCountSpan.textContent = particles.length;
      resolutionSpan.textContent = `${gridW}×${gridH}`;
    }

    function pointerPos(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return { x: clientX - rect.left, y: clientY - rect.top };
    }

    function applyTool(event) {
      const p = pointerPos(event);
      const gx = Math.floor(p.x / scale);
      const gy = Math.floor(p.y / scale);
      const brush = parseInt(brushSlider.value, 10) / scale;

      if (tool === "fire") {
        addFire(gx, gy, brush, parseFloat(heatSlider.value));
      } else if (tool === "smoke") {
        addSmoke(gx, gy, brush * 1.2, parseFloat(smokeSlider.value));
      } else if (tool === "cool") {
        coolArea(gx, gy, brush * 1.3);
      } else if (tool === "wall") {
        paintWall(gx, gy, brush * 0.7, 1);
      } else if (tool === "erase") {
        paintWall(gx, gy, brush, 0);
        coolArea(gx, gy, brush * 1.4);
      } else if (tool === "wind") {
        const dx = lastPointer ? p.x - lastPointer.x : 0;
        const dy = lastPointer ? p.y - lastPointer.y : 0;
        addWind(gx, gy, brush * 1.8, dx * 0.05, dy * 0.05);
      }

      lastPointer = p;
    }

    function addWind(cx, cy, radius, ax, ay) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y) || walls[idx(x, y)]) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const f = 1 - d / (r + 1);
            const i = idx(x, y);
            vx[i] += ax * f;
            vy[i] += ay * f;
          }
        }
      }
    }

    function makeCampfire() {
      resetWorld();
      const y = Math.floor(gridH * 0.78);
      for (let x = Math.floor(gridW * 0.35); x < Math.floor(gridW * 0.65); x++) {
        for (let yy = y; yy < y + 4; yy++) {
          if (inside(x, yy)) walls[idx(x, yy)] = 1;
        }
      }
      for (let i = 0; i < 24; i++) {
        addFire(Math.floor(gridW * (0.42 + Math.random() * 0.16)), y - 3, 5 + Math.random() * 5, 130);
      }
    }

    function addSparkBurst() {
      for (let i = 0; i < 120; i++) {
        particles.push(new Spark(width * (0.45 + Math.random() * 0.1), height * 0.70, true));
      }
      trimParticles();
    }

    canvas.addEventListener("pointerdown", (event) => {
      drawing = true;
      lastPointer = pointerPos(event);
      applyTool(event);
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!drawing) return;
      applyTool(event);
    });

    window.addEventListener("pointerup", () => {
      drawing = false;
      lastPointer = null;
    });

    document.querySelectorAll("[data-tool]").forEach(button => {
      button.addEventListener("click", () => {
        tool = button.dataset.tool;
        document.querySelectorAll("[data-tool]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
      });
    });

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    clearButton.addEventListener("click", resetWorld);
    campfireButton.addEventListener("click", makeCampfire);
    sparkButton.addEventListener("click", addSparkBurst);

    window.addEventListener("resize", resize);

    resize();
    makeCampfire();
    animate();
  </script>
