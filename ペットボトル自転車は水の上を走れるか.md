  <script>
    const g = 9.81;
    const rhoWater = 1000;
    const dt = 1 / 60;

    const ids = [
      "riderMass", "bikeMass", "bottleVolume", "bottleMass", "pedalForce", "drag",
      "frontLeft", "frontRight", "rearLeft", "rearRight", "midLeft", "midRight"
    ];

    const el = Object.fromEntries(ids.map(id => [id, document.getElementById(id)]));
    const canvas = document.getElementById("sim");
    const ctx = canvas.getContext("2d");

    let x = 160;
    let speed = 0;
    let time = 0;

    const slots = [
      { id: "frontLeft", name: "前左", x: 1, y: -1 },
      { id: "frontRight", name: "前右", x: 1, y: 1 },
      { id: "rearLeft", name: "後左", x: -1, y: -1 },
      { id: "rearRight", name: "後右", x: -1, y: 1 },
      { id: "midLeft", name: "中央左", x: 0, y: -1 },
      { id: "midRight", name: "中央右", x: 0, y: 1 },
    ];

    function number(id) {
      return Number(el[id].value);
    }

    function clamp(v, lo, hi) {
      return Math.max(lo, Math.min(hi, v));
    }

    function calc() {
      const riderMass = number("riderMass");
      const bikeMass = number("bikeMass");
      const bottleVolumeL = number("bottleVolume");
      const bottleMass = number("bottleMass");
      const pedalForce = number("pedalForce");
      const drag = number("drag");

      const counts = Object.fromEntries(slots.map(s => [s.id, Math.max(0, Number(el[s.id].value) || 0)]));
      const bottleCount = Object.values(counts).reduce((a, b) => a + b, 0);

      const bottleVolumeM3 = bottleVolumeL / 1000;
      const bottleBuoyancyMass = rhoWater * bottleVolumeM3;
      const maxSupportedMass = bottleCount * bottleBuoyancyMass;

      const totalMass = riderMass + bikeMass + bottleCount * bottleMass;
      const totalWeightN = totalMass * g;
      const maxBuoyancyN = maxSupportedMass * g;
      const marginN = maxBuoyancyN - totalWeightN;
      const floatRatio = maxBuoyancyN / totalWeightN;

      let frontSupport = counts.frontLeft + counts.frontRight + 0.5 * (counts.midLeft + counts.midRight);
      let rearSupport = counts.rearLeft + counts.rearRight + 0.5 * (counts.midLeft + counts.midRight);
      let leftSupport = counts.frontLeft + counts.rearLeft + counts.midLeft;
      let rightSupport = counts.frontRight + counts.rearRight + counts.midRight;

      const pitchBalance = (frontSupport - rearSupport) / Math.max(1, bottleCount);
      const rollBalance = (leftSupport - rightSupport) / Math.max(1, bottleCount);

      return {
        riderMass, bikeMass, bottleVolumeL, bottleMass, pedalForce, drag,
        counts, bottleCount, totalMass, totalWeightN, maxBuoyancyN, marginN,
        floatRatio, pitchBalance, rollBalance
      };
    }

    function updatePhysics(c) {
      const floating = c.marginN > 0;

      if (!floating) {
        speed *= 0.96;
        return;
      }

      const balancePenalty = clamp(1 - Math.abs(c.pitchBalance) * 1.5 - Math.abs(c.rollBalance) * 1.8, 0.05, 1);
      const effectiveForce = c.pedalForce * balancePenalty;
      const waterDrag = c.drag * speed * Math.abs(speed);
      const accel = (effectiveForce - waterDrag) / c.totalMass;

      speed += accel * dt;
      speed = clamp(speed, 0, 8);
      x += speed * 28 * dt;

      if (x > canvas.width + 140) x = -140;
      time += dt;
    }

    function drawWater() {
      const waterY = 300;
      ctx.fillStyle = "rgba(255,255,255,0.55)";
      ctx.beginPath();
      for (let i = 0; i <= canvas.width; i += 10) {
        const y = waterY + Math.sin(i * 0.025 + time * 2.2) * 6;
        if (i === 0) ctx.moveTo(i, y);
        else ctx.lineTo(i, y);
      }
      ctx.lineTo(canvas.width, canvas.height);
      ctx.lineTo(0, canvas.height);
      ctx.closePath();
      ctx.fill();
    }

    function drawBottle(cx, cy, count, side) {
      const rows = Math.ceil(count / 5);
      const visible = Math.min(count, 35);
      for (let i = 0; i < visible; i++) {
        const col = i % 5;
        const row = Math.floor(i / 5);
        const bx = cx + (col - 2) * 13;
        const by = cy + row * 10 * side;
        ctx.fillStyle = "rgba(170,230,255,0.9)";
        ctx.strokeStyle = "#126e9c";
        ctx.lineWidth = 1;
        ctx.beginPath();
        ctx.roundRect(bx - 5, by - 16, 10, 30, 5);
        ctx.fill();
        ctx.stroke();
      }

      if (count > visible) {
        ctx.fillStyle = "#0b4e75";
        ctx.font = "12px sans-serif";
        ctx.fillText("+" + (count - visible), cx + 35, cy);
      }
    }

    function drawBike(c) {
      const waterY = 300;
      const sink = c.marginN >= 0 ? 0 : clamp(-c.marginN / 450, 0, 100);
      const bob = Math.sin(time * 3) * 3;
      const y = waterY - 65 + sink + bob;
      const tilt = clamp(c.pitchBalance * 0.35, -0.25, 0.25);
      const rollShift = clamp(c.rollBalance * 35, -45, 45);

      ctx.save();
      ctx.translate(x, y);
      ctx.rotate(tilt);

      // bottle groups
      drawBottle(-80, 55 + rollShift * 0.25, c.counts.rearLeft + c.counts.rearRight, 1);
      drawBottle(0, 70 + rollShift * 0.15, c.counts.midLeft + c.counts.midRight, 1);
      drawBottle(85, 55 + rollShift * 0.25, c.counts.frontLeft + c.counts.frontRight, 1);

      // wheels
      ctx.strokeStyle = "#222";
      ctx.lineWidth = 5;
      ctx.beginPath();
      ctx.arc(-80, 50, 36, 0, Math.PI * 2);
      ctx.arc(85, 50, 36, 0, Math.PI * 2);
      ctx.stroke();

      // frame
      ctx.strokeStyle = "#26384a";
      ctx.lineWidth = 6;
      ctx.beginPath();
      ctx.moveTo(-80, 50);
      ctx.lineTo(-20, -10);
      ctx.lineTo(35, 50);
      ctx.lineTo(-80, 50);
      ctx.moveTo(35, 50);
      ctx.lineTo(85, 50);
      ctx.lineTo(20, -22);
      ctx.lineTo(-20, -10);
      ctx.stroke();

      // handle and seat
      ctx.lineWidth = 4;
      ctx.beginPath();
      ctx.moveTo(20, -22);
      ctx.lineTo(45, -48);
      ctx.lineTo(73, -43);
      ctx.moveTo(-20, -10);
      ctx.lineTo(-28, -36);
      ctx.lineTo(-55, -36);
      ctx.stroke();

      // rider
      ctx.fillStyle = "#f0b37e";
      ctx.beginPath();
      ctx.arc(-15, -75, 16, 0, Math.PI * 2);
      ctx.fill();

      ctx.strokeStyle = "#3b5f91";
      ctx.lineWidth = 8;
      ctx.beginPath();
      ctx.moveTo(-15, -58);
      ctx.lineTo(-8, -18);
      ctx.lineTo(25, 20);
      ctx.moveTo(-8, -18);
      ctx.lineTo(-45, 18);
      ctx.moveTo(-12, -45);
      ctx.lineTo(50, -43);
      ctx.stroke();

      ctx.restore();

      // water line marker
      ctx.strokeStyle = "rgba(255,255,255,0.8)";
      ctx.setLineDash([8, 8]);
      ctx.beginPath();
      ctx.moveTo(0, waterY);
      ctx.lineTo(canvas.width, waterY);
      ctx.stroke();
      ctx.setLineDash([]);
    }

    function drawBar(c) {
      const bx = 30;
      const by = 30;
      const bw = 280;
      const bh = 24;
      const ratio = clamp(c.floatRatio, 0, 2);

      ctx.fillStyle = "rgba(255,255,255,0.85)";
      ctx.roundRect(bx - 12, by - 12, bw + 24, 92, 14);
      ctx.fill();

      ctx.fillStyle = "#dbe8ef";
      ctx.roundRect(bx, by, bw, bh, 12);
      ctx.fill();

      ctx.fillStyle = c.marginN >= 0 ? "#29a35a" : "#d64b4b";
      ctx.roundRect(bx, by, Math.min(bw, bw * ratio), bh, 12);
      ctx.fill();

      ctx.fillStyle = "#1d3348";
      ctx.font = "14px sans-serif";
      ctx.fillText("浮力 / 重量 = " + c.floatRatio.toFixed(2), bx, by + 50);
      ctx.fillText(c.marginN >= 0 ? "浮く可能性あり" : "沈む", bx, by + 72);
    }

    function updateLabels(c) {
      document.getElementById("riderMassLabel").textContent = c.riderMass + " kg";
      document.getElementById("bikeMassLabel").textContent = c.bikeMass + " kg";
      document.getElementById("bottleVolumeLabel").textContent = c.bottleVolumeL.toFixed(1) + " L";
      document.getElementById("bottleMassLabel").textContent = c.bottleMass.toFixed(3) + " kg";
      document.getElementById("pedalForceLabel").textContent = c.pedalForce + " N";
      document.getElementById("dragLabel").textContent = c.drag + "";

      document.getElementById("weightStat").textContent = (c.totalWeightN / g).toFixed(1) + " kg相当";
      document.getElementById("buoyStat").textContent = (c.maxBuoyancyN / g).toFixed(1) + " kg相当";
      document.getElementById("marginStat").textContent = (c.marginN / g).toFixed(1) + " kg相当";
      document.getElementById("speedStat").textContent = speed.toFixed(2) + " m/s";
      document.getElementById("pitchStat").textContent = c.pitchBalance.toFixed(2);
      document.getElementById("rollStat").textContent = c.rollBalance.toFixed(2);

      const need = Math.ceil((c.riderMass + c.bikeMass) / (c.bottleVolumeL - c.bottleMass));
      let msg = "必要本数の目安：" + need + "本以上。";
      if (c.marginN < 0) msg += " 今の設定では浮力不足です。ペットボトルを増やしてください。";
      else msg += " 浮きます。ただし実物では固定強度、横転、波、ペダル推進効率が大問題です。";

      if (Math.abs(c.pitchBalance) > 0.25) msg += " 前後バランスが悪いので傾きやすいです。";
      if (Math.abs(c.rollBalance) > 0.18) msg += " 左右バランスが悪いので横転しやすいです。";

      document.getElementById("message").textContent = msg;
    }

    function loop() {
      const c = calc();
      updatePhysics(c);
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      drawWater();
      drawBike(c);
      drawBar(c);
      updateLabels(c);
      requestAnimationFrame(loop);
    }

    document.getElementById("reset").addEventListener("click", () => {
      speed = 0;
      x = 160;
    });

    document.getElementById("balance").addEventListener("click", () => {
      const rider = number("riderMass");
      const bike = number("bikeMass");
      const vol = number("bottleVolume");
      const bm = number("bottleMass");
      const needed = Math.ceil((rider + bike) / Math.max(0.1, vol - bm)) + 6;
      const per = Math.ceil(needed / 6);
      slots.forEach(s => el[s.id].value = per);
      speed = 0;
    });

    loop();
  </script>

2Lペットボトル1本 ≒ 約2kgぶんの浮力
人55kg＋自転車16kgなら、最低でも約36本以上必要です。
実際は安定性・固定強度・横転・推進効率が厳しいので、シミュレーションでは「浮くけど走るのは難しい」方向に調整しています。
