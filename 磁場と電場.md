  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const electricModeButton = document.getElementById("electricMode");
    const magnetModeButton = document.getElementById("magnetMode");
    const addPositiveButton = document.getElementById("addPositive");
    const addNegativeButton = document.getElementById("addNegative");
    const sprinkleButton = document.getElementById("sprinkle");
    const clearParticlesButton = document.getElementById("clearParticles");
    const resetButton = document.getElementById("reset");
    const toggleLinesButton = document.getElementById("toggleLines");
    const strengthSlider = document.getElementById("strength");
    const particleSpeedSlider = document.getElementById("particleSpeed");
    const strengthValue = document.getElementById("strengthValue");
    const speedValue = document.getElementById("speedValue");
    const electricButtons = document.getElementById("electricButtons");

    let width = 0;
    let height = 0;
    let mode = "electric";
    let charges = [];
    let particles = [];
    let dragging = null;
    let showLines = true;

    const SOFTENING = 950;
    const MAX_PARTICLES = 900;

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
    }

    class Charge {
      constructor(x, y, q, pairId = null) {
        this.x = x;
        this.y = y;
        this.q = q;
        this.r = 18;
        this.pairId = pairId;
      }
    }

    class Particle {
      constructor(x, y) {
        this.x = x;
        this.y = y;
        this.vx = (Math.random() - 0.5) * 0.8;
        this.vy = (Math.random() - 0.5) * 0.8;
        this.life = 1;
      }

      update() {
        const f = fieldAt(this.x, this.y);
        const speed = parseFloat(particleSpeedSlider.value);
        this.vx += f.x * 0.09 * speed;
        this.vy += f.y * 0.09 * speed;

        const v = Math.hypot(this.vx, this.vy);
        const maxV = 4.5 * speed;
        if (v > maxV) {
          this.vx = this.vx / v * maxV;
          this.vy = this.vy / v * maxV;
        }

        this.x += this.vx;
        this.y += this.vy;
        this.vx *= 0.985;
        this.vy *= 0.985;
        this.life *= 0.999;

        if (this.x < -20 || this.x > width + 20 || this.y < -20 || this.y > height + 20 || this.life < 0.2) {
          this.x = Math.random() * width;
          this.y = Math.random() * height;
          this.vx = (Math.random() - 0.5) * 0.8;
          this.vy = (Math.random() - 0.5) * 0.8;
          this.life = 1;
        }
      }

      draw() {
        ctx.beginPath();
        ctx.arc(this.x, this.y, 2.2, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(134, 239, 172, ${0.25 + this.life * 0.55})`;
        ctx.fill();
      }
    }

    function fieldAt(x, y) {
      const strength = parseFloat(strengthSlider.value);
      let fx = 0;
      let fy = 0;

      for (const c of charges) {
        const dx = x - c.x;
        const dy = y - c.y;
        const r2 = dx * dx + dy * dy + SOFTENING;
        const inv = 1 / Math.pow(r2, 1.5);

        // 正電荷/N極から出て、負電荷/S極へ入る向き。
        fx += c.q * dx * inv * 65000 * strength;
        fy += c.q * dy * inv * 65000 * strength;
      }

      return { x: fx, y: fy };
    }

    function resetElectric() {
      mode = "electric";
      charges = [
        new Charge(width * 0.38, height * 0.5, 1),
        new Charge(width * 0.62, height * 0.5, -1),
      ];
      particles = [];
      sprinkleParticles(380);
      updateModeUI();
    }

    function resetMagnet() {
      mode = "magnet";
      charges = [
        new Charge(width * 0.42, height * 0.5, 1, "magnet"),
        new Charge(width * 0.58, height * 0.5, -1, "magnet"),
      ];
      particles = [];
      sprinkleParticles(420);
      updateModeUI();
    }

    function sprinkleParticles(count = 220) {
      for (let i = 0; i < count; i++) {
        if (particles.length >= MAX_PARTICLES) particles.shift();
        particles.push(new Particle(Math.random() * width, Math.random() * height));
      }
    }

    function drawBackground() {
      ctx.fillStyle = "rgba(2, 6, 23, 0.22)";
      ctx.fillRect(0, 0, width, height);
    }

    function drawFieldLines() {
      if (!showLines) return;

      const starts = [];
      for (const c of charges) {
        if (c.q > 0) {
          for (let i = 0; i < 22; i++) {
            const a = (i / 22) * Math.PI * 2;
            starts.push({
              x: c.x + Math.cos(a) * 24,
              y: c.y + Math.sin(a) * 24,
            });
          }
        }
      }

      // 正電荷がない場合も、画面上から力線を少し出す。
      if (starts.length === 0) {
        for (let i = 0; i < 36; i++) {
          starts.push({ x: Math.random() * width, y: Math.random() * height });
        }
      }

      ctx.lineWidth = 1;
      for (const s of starts) {
        let x = s.x;
        let y = s.y;
        ctx.beginPath();
        ctx.moveTo(x, y);

        for (let step = 0; step < 220; step++) {
          const f = fieldAt(x, y);
          const m = Math.hypot(f.x, f.y);
          if (m < 0.0001) break;

          const dirX = f.x / m;
          const dirY = f.y / m;
          x += dirX * 7;
          y += dirY * 7;
          ctx.lineTo(x, y);

          if (x < 0 || x > width || y < 0 || y > height) break;
          if (nearNegativeCharge(x, y)) break;
        }

        ctx.strokeStyle = mode === "electric" ? "rgba(196, 181, 253, 0.32)" : "rgba(250, 204, 21, 0.34)";
        ctx.stroke();
      }
    }

    function nearNegativeCharge(x, y) {
      return charges.some(c => c.q < 0 && Math.hypot(x - c.x, y - c.y) < 24);
    }

    function drawCharges() {
      if (mode === "magnet" && charges.length >= 2) {
        const a = charges[0];
        const b = charges[1];
        ctx.save();
        ctx.lineCap = "round";
        ctx.lineWidth = 26;
        ctx.strokeStyle = "rgba(148, 163, 184, 0.72)";
        ctx.beginPath();
        ctx.moveTo(a.x, a.y);
        ctx.lineTo(b.x, b.y);
        ctx.stroke();
        ctx.restore();
      }

      for (const c of charges) {
        ctx.save();
        ctx.beginPath();
        ctx.arc(c.x, c.y, c.r, 0, Math.PI * 2);
        ctx.fillStyle = c.q > 0 ? "rgba(251, 113, 133, 0.96)" : "rgba(96, 165, 250, 0.96)";
        ctx.shadowColor = c.q > 0 ? "#fb7185" : "#60a5fa";
        ctx.shadowBlur = 18;
        ctx.fill();

        ctx.fillStyle = "white";
        ctx.font = "bold 18px system-ui";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";
        ctx.fillText(mode === "magnet" ? (c.q > 0 ? "N" : "S") : (c.q > 0 ? "+" : "−"), c.x, c.y + 1);
        ctx.restore();
      }
    }

    function drawVectorField() {
      const step = 54;
      ctx.lineWidth = 1;
      for (let y = 34; y < height; y += step) {
        for (let x = 34; x < width; x += step) {
          const f = fieldAt(x, y);
          const m = Math.hypot(f.x, f.y);
          if (m < 0.02) continue;
          const len = Math.min(18, 5 + Math.log(1 + m) * 5);
          const dx = f.x / m * len;
          const dy = f.y / m * len;
          drawArrow(x, y, x + dx, y + dy, "rgba(125, 211, 252, 0.28)");
        }
      }
    }

    function drawArrow(x1, y1, x2, y2, color) {
      const angle = Math.atan2(y2 - y1, x2 - x1);
      ctx.strokeStyle = color;
      ctx.fillStyle = color;
      ctx.beginPath();
      ctx.moveTo(x1, y1);
      ctx.lineTo(x2, y2);
      ctx.stroke();
      ctx.beginPath();
      ctx.moveTo(x2, y2);
      ctx.lineTo(x2 - Math.cos(angle - 0.55) * 5, y2 - Math.sin(angle - 0.55) * 5);
      ctx.lineTo(x2 - Math.cos(angle + 0.55) * 5, y2 - Math.sin(angle + 0.55) * 5);
      ctx.closePath();
      ctx.fill();
    }

    function animate() {
      drawBackground();
      drawVectorField();
      drawFieldLines();

      for (const p of particles) {
        p.update();
        p.draw();
      }

      drawCharges();
      updateLabels();
      requestAnimationFrame(animate);
    }

    function updateLabels() {
      strengthValue.textContent = parseFloat(strengthSlider.value).toFixed(2);
      speedValue.textContent = parseFloat(particleSpeedSlider.value).toFixed(2);
    }

    function updateModeUI() {
      electricModeButton.classList.toggle("active", mode === "electric");
      magnetModeButton.classList.toggle("active", mode === "magnet");
      electricButtons.style.display = mode === "electric" ? "grid" : "none";
    }

    function pointerPos(event) {
      const rect = canvas.getBoundingClientRect();
      return { x: event.clientX - rect.left, y: event.clientY - rect.top };
    }

    canvas.addEventListener("pointerdown", (event) => {
      const p = pointerPos(event);
      dragging = null;
      for (let i = charges.length - 1; i >= 0; i--) {
        const c = charges[i];
        if (Math.hypot(p.x - c.x, p.y - c.y) < c.r + 10) {
          dragging = { charge: c, startX: p.x, startY: p.y };
          break;
        }
      }
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!dragging) return;
      const p = pointerPos(event);
      const dx = p.x - dragging.charge.x;
      const dy = p.y - dragging.charge.y;

      if (mode === "magnet" && dragging.charge.pairId) {
        for (const c of charges) {
          if (c.pairId === dragging.charge.pairId) {
            c.x += dx;
            c.y += dy;
          }
        }
      } else {
        dragging.charge.x = p.x;
        dragging.charge.y = p.y;
      }
    });

    window.addEventListener("pointerup", () => {
      dragging = null;
    });

    electricModeButton.addEventListener("click", resetElectric);
    magnetModeButton.addEventListener("click", resetMagnet);

    addPositiveButton.addEventListener("click", () => {
      charges.push(new Charge(width * (0.25 + Math.random() * 0.5), height * (0.25 + Math.random() * 0.5), 1));
    });

    addNegativeButton.addEventListener("click", () => {
      charges.push(new Charge(width * (0.25 + Math.random() * 0.5), height * (0.25 + Math.random() * 0.5), -1));
    });

    sprinkleButton.addEventListener("click", () => sprinkleParticles(260));
    clearParticlesButton.addEventListener("click", () => particles = []);
    toggleLinesButton.addEventListener("click", () => showLines = !showLines);
    resetButton.addEventListener("click", () => mode === "electric" ? resetElectric() : resetMagnet());

    window.addEventListener("resize", () => {
      resize();
      mode === "electric" ? resetElectric() : resetMagnet();
    });

    resize();
    resetElectric();
    animate();
  </script>
</body>
