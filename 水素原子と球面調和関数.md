  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const lRange = document.getElementById("lRange");
    const mRange = document.getElementById("mRange");
    const ampRange = document.getElementById("ampRange");
    const resRange = document.getElementById("resRange");
    const modeSelect = document.getElementById("modeSelect");
    const resetBtn = document.getElementById("resetBtn");

    const lVal = document.getElementById("lVal");
    const mVal = document.getElementById("mVal");
    const ampVal = document.getElementById("ampVal");
    const resVal = document.getElementById("resVal");

    let W = 0, H = 0;
    let rotX = -0.55;
    let rotY = 0.75;
    let zoom = 1.0;
    let dragging = false;
    let lastX = 0, lastY = 0;
    let points = [];
    let dirty = true;

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      W = canvas.clientWidth;
      H = canvas.clientHeight;
      canvas.width = Math.floor(W * dpr);
      canvas.height = Math.floor(H * dpr);
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      dirty = true;
    }
    window.addEventListener("resize", resize);
    resize();

    function factorial(n) {
      let v = 1;
      for (let i = 2; i <= n; i++) v *= i;
      return v;
    }

    // Associated Legendre polynomial P_l^m(x), m >= 0
    function assocLegendre(l, m, x) {
      m = Math.abs(m);
      if (m > l) return 0;

      let pmm = 1.0;
      if (m > 0) {
        const somx2 = Math.sqrt(Math.max(0, 1 - x * x));
        let fact = 1.0;
        for (let i = 1; i <= m; i++) {
          pmm *= -fact * somx2;
          fact += 2.0;
        }
      }
      if (l === m) return pmm;

      let pmmp1 = x * (2 * m + 1) * pmm;
      if (l === m + 1) return pmmp1;

      let pll = 0;
      for (let ll = m + 2; ll <= l; ll++) {
        pll = ((2 * ll - 1) * x * pmmp1 - (ll + m - 1) * pmm) / (ll - m);
        pmm = pmmp1;
        pmmp1 = pll;
      }
      return pll;
    }

    function normalization(l, mAbs) {
      return Math.sqrt(((2 * l + 1) / (4 * Math.PI)) * factorial(l - mAbs) / factorial(l + mAbs));
    }

    function sphericalHarmonicValue(l, m, theta, phi, mode) {
      const ma = Math.abs(m);
      const P = assocLegendre(l, ma, Math.cos(theta));
      const N = normalization(l, ma);
      const base = N * P;

      // Real form approximation. This is good for educational visualization.
      const real = m === 0 ? base : Math.SQRT2 * base * Math.cos(ma * phi);
      const imag = m === 0 ? 0 : Math.SQRT2 * base * Math.sin(ma * phi) * (m < 0 ? -1 : 1);
      const abs = Math.sqrt(real * real + imag * imag);

      if (mode === "abs") return { value: abs, sign: 1 };
      if (mode === "prob") return { value: abs * abs, sign: 1 };
      if (mode === "phase") return { value: abs, sign: Math.cos(ma * phi) >= 0 ? 1 : -1 };
      return { value: Math.abs(real), sign: real >= 0 ? 1 : -1 };
    }

    function updateControls() {
      let l = Number(lRange.value);
      let m = Number(mRange.value);
      mRange.min = -l;
      mRange.max = l;
      if (m < -l) m = -l;
      if (m > l) m = l;
      mRange.value = m;
      lVal.textContent = l;
      mVal.textContent = m;
      ampVal.textContent = Number(ampRange.value).toFixed(2);
      resVal.textContent = resRange.value;
      dirty = true;
    }

    [lRange, mRange, ampRange, resRange, modeSelect].forEach(el => {
      el.addEventListener("input", updateControls);
      el.addEventListener("change", updateControls);
    });
    updateControls();

    function buildPoints() {
      const l = Number(lRange.value);
      const m = Number(mRange.value);
      const amp = Number(ampRange.value);
      const res = Number(resRange.value);
      const mode = modeSelect.value;
      const arr = [];

      let maxVal = 0;
      const raw = [];
      for (let i = 0; i <= res; i++) {
        const theta = Math.PI * i / res;
        for (let j = 0; j < res * 2; j++) {
          const phi = 2 * Math.PI * j / (res * 2);
          const y = sphericalHarmonicValue(l, m, theta, phi, mode);
          maxVal = Math.max(maxVal, Math.abs(y.value));
          raw.push({ theta, phi, value: y.value, sign: y.sign });
        }
      }
      maxVal = maxVal || 1;

      for (const p of raw) {
        const normalized = p.value / maxVal;
        const radius = 0.42 + amp * 0.34 * normalized;
        const sinT = Math.sin(p.theta);
        const x = radius * sinT * Math.cos(p.phi);
        const y = radius * Math.cos(p.theta);
        const z = radius * sinT * Math.sin(p.phi);
        arr.push({ x, y, z, sign: p.sign, strength: normalized });
      }
      points = arr;
      dirty = false;
    }

    function rotatePoint(p) {
      const cy = Math.cos(rotY), sy = Math.sin(rotY);
      const cx = Math.cos(rotX), sx = Math.sin(rotX);

      let x = p.x * cy + p.z * sy;
      let z = -p.x * sy + p.z * cy;
      let y = p.y;

      let y2 = y * cx - z * sx;
      let z2 = y * sx + z * cx;
      return { x, y: y2, z: z2 };
    }

    function drawAxes(scale) {
      const axes = [
        { p: { x: 1, y: 0, z: 0 }, label: "x" },
        { p: { x: 0, y: 1, z: 0 }, label: "z" },
        { p: { x: 0, y: 0, z: 1 }, label: "y" }
      ];
      ctx.save();
      ctx.translate(W / 2, H / 2);
      ctx.strokeStyle = "rgba(255,255,255,0.25)";
      ctx.fillStyle = "rgba(255,255,255,0.65)";
      ctx.lineWidth = 1;
      ctx.font = "14px system-ui";
      for (const a of axes) {
        const r = rotatePoint(a.p);
        ctx.beginPath();
        ctx.moveTo(0, 0);
        ctx.lineTo(r.x * scale * 0.85, -r.y * scale * 0.85);
        ctx.stroke();
        ctx.fillText(a.label, r.x * scale * 0.9, -r.y * scale * 0.9);
      }
      ctx.restore();
    }

    function render() {
      requestAnimationFrame(render);
      if (dirty) buildPoints();

      ctx.clearRect(0, 0, W, H);
      const grad = ctx.createRadialGradient(W * 0.45, H * 0.25, 20, W * 0.5, H * 0.5, Math.max(W, H));
      grad.addColorStop(0, "#1e293b");
      grad.addColorStop(1, "#020617");
      ctx.fillStyle = grad;
      ctx.fillRect(0, 0, W, H);

      const scale = Math.min(W, H) * 0.46 * zoom;
      drawAxes(scale);

      const projected = points.map(p => {
        const r = rotatePoint(p);
        const depth = 2.5 + r.z;
        const perspective = 1.4 / depth;
        return {
          sx: W / 2 + r.x * scale * perspective,
          sy: H / 2 - r.y * scale * perspective,
          z: r.z,
          sign: p.sign,
          strength: p.strength,
          size: Math.max(1.1, 3.5 * perspective * zoom)
        };
      }).sort((a, b) => a.z - b.z);

      for (const q of projected) {
        const alpha = 0.22 + 0.78 * q.strength;
        const light = Math.floor(120 + 110 * (q.z + 1.2) / 2.4);
        if (q.sign >= 0) {
          ctx.fillStyle = `rgba(${Math.min(130, light)}, ${Math.min(190, light + 30)}, 255, ${alpha})`;
        } else {
          ctx.fillStyle = `rgba(255, ${Math.min(150, light)}, ${Math.min(180, light)}, ${alpha})`;
        }
        ctx.beginPath();
        ctx.arc(q.sx, q.sy, q.size, 0, Math.PI * 2);
        ctx.fill();
      }

      ctx.fillStyle = "rgba(255,255,255,0.8)";
      ctx.font = "16px system-ui";
      const l = Number(lRange.value);
      const m = Number(mRange.value);
      ctx.fillText(`Y_${l}^{${m}}(θ, φ)   mode: ${modeSelect.options[modeSelect.selectedIndex].text}`, 18, 30);
    }
    render();

    canvas.addEventListener("mousedown", e => {
      dragging = true;
      lastX = e.clientX;
      lastY = e.clientY;
    });
    window.addEventListener("mouseup", () => dragging = false);
    window.addEventListener("mousemove", e => {
      if (!dragging) return;
      const dx = e.clientX - lastX;
      const dy = e.clientY - lastY;
      rotY += dx * 0.008;
      rotX += dy * 0.008;
      lastX = e.clientX;
      lastY = e.clientY;
    });
    canvas.addEventListener("wheel", e => {
      e.preventDefault();
      zoom *= e.deltaY > 0 ? 0.92 : 1.08;
      zoom = Math.max(0.45, Math.min(2.4, zoom));
    }, { passive: false });

    resetBtn.addEventListener("click", () => {
      rotX = -0.55;
      rotY = 0.75;
      zoom = 1.0;
    });
  </script>
</body>
