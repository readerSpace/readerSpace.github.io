  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");
    const clearButton = document.getElementById("clear");
    const twoSourcesButton = document.getElementById("twoSources");
    const rainButton = document.getElementById("rain");
    const waveSpeedSlider = document.getElementById("waveSpeed");
    const dampingSlider = document.getElementById("damping");
    const amplitudeSlider = document.getElementById("amplitude");
    const frequencySlider = document.getElementById("frequency");
    const speedValue = document.getElementById("speedValue");
    const dampingValue = document.getElementById("dampingValue");
    const ampValue = document.getElementById("ampValue");
    const freqValue = document.getElementById("freqValue");
    const sourceCountSpan = document.getElementById("sourceCount");
    const resolutionSpan = document.getElementById("resolution");

    let width = 0;
    let height = 0;
    let scale = 4;
    let gridW = 0;
    let gridH = 0;
    let current;
    let previous;
    let next;
    let walls;
    let imageData;
    let sources = [];
    let running = true;
    let rainMode = false;
    let tool = "drop";
    let drawing = false;
    let time = 0;
    let renderCanvas = document.createElement("canvas");
    let renderCtx = renderCanvas.getContext("2d");

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

      scale = width < 700 ? 5 : 4;
      gridW = Math.floor(width / scale);
      gridH = Math.floor(height / scale);
      renderCanvas.width = gridW;
      renderCanvas.height = gridH;
      imageData = renderCtx.createImageData(gridW, gridH);
      resetWorld();
    }

    function idx(x, y) { return y * gridW + x; }
    function inside(x, y) { return x >= 1 && x < gridW - 1 && y >= 1 && y < gridH - 1; }

    function resetWorld() {
      const n = gridW * gridH;
      current = new Float32Array(n);
      previous = new Float32Array(n);
      next = new Float32Array(n);
      walls = new Uint8Array(n);
      sources = [];
      time = 0;
    }

    function disturb(cx, cy, radius, amount) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y)) continue;
          if (walls[idx(x, y)]) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const falloff = Math.cos((d / r) * Math.PI * 0.5);
            current[idx(x, y)] += amount * falloff;
          }
        }
      }
    }

    function stepWave() {
      const c = parseFloat(waveSpeedSlider.value);
      const damping = parseFloat(dampingSlider.value);

      for (let y = 1; y < gridH - 1; y++) {
        for (let x = 1; x < gridW - 1; x++) {
          const i = idx(x, y);
          if (walls[i]) {
            next[i] = 0;
            continue;
          }

          const lap = current[idx(x - 1, y)] + current[idx(x + 1, y)] + current[idx(x, y - 1)] + current[idx(x, y + 1)] - 4 * current[i];
          next[i] = (2 * current[i] - previous[i] + c * c * lap) * damping;
        }
      }

      // 波源から連続的に波を出す。
      const amp = parseFloat(amplitudeSlider.value);
      const freq = parseFloat(frequencySlider.value);
      for (const s of sources) {
        disturb(Math.floor(s.x / scale), Math.floor(s.y / scale), 3, Math.sin(time * freq + s.phase) * amp * 0.08);
      }

      if (rainMode && Math.random() < 0.18) {
        disturb(2 + Math.floor(Math.random() * (gridW - 4)), 2 + Math.floor(Math.random() * (gridH - 4)), 4, 40 + Math.random() * 80);
      }

      const temp = previous;
      previous = current;
      current = next;
      next = temp;
      time++;
    }

    function render() {
      const data = imageData.data;
      for (let y = 0; y < gridH; y++) {
        for (let x = 0; x < gridW; x++) {
          const i = idx(x, y);
          const p = i * 4;

          if (walls[i]) {
            data[p] = 180;
            data[p + 1] = 190;
            data[p + 2] = 205;
            data[p + 3] = 255;
            continue;
          }

          const h = Math.max(-80, Math.min(80, current[i]));
          const bright = 128 + h * 1.45;
          const interference = Math.abs(h);

          if (h >= 0) {
            data[p] = 15 + interference * 1.2;
            data[p + 1] = 105 + bright * 0.45;
            data[p + 2] = 170 + interference * 0.8;
          } else {
            data[p] = 5;
            data[p + 1] = 35 + bright * 0.35;
            data[p + 2] = 95 + bright * 0.45;
          }
          data[p + 3] = 255;
        }
      }

      renderCtx.putImageData(imageData, 0, 0);
      ctx.imageSmoothingEnabled = false;
      ctx.drawImage(renderCanvas, 0, 0, width, height);

      drawSources();
      updateLabels();
    }

    function drawSources() {
      for (const s of sources) {
        ctx.save();
        ctx.beginPath();
        ctx.arc(s.x, s.y, 10, 0, Math.PI * 2);
        ctx.fillStyle = "#facc15";
        ctx.shadowColor = "#facc15";
        ctx.shadowBlur = 18;
        ctx.fill();
        ctx.beginPath();
        ctx.arc(s.x, s.y, 20 + Math.sin(time * 0.1 + s.phase) * 5, 0, Math.PI * 2);
        ctx.strokeStyle = "rgba(250, 204, 21, 0.45)";
        ctx.lineWidth = 2;
        ctx.stroke();
        ctx.restore();
      }
    }

    function animate() {
      if (running) {
        for (let i = 0; i < 3; i++) stepWave();
      }
      render();
      requestAnimationFrame(animate);
    }

    function updateLabels() {
      speedValue.textContent = parseFloat(waveSpeedSlider.value).toFixed(2);
      dampingValue.textContent = parseFloat(dampingSlider.value).toFixed(3);
      ampValue.textContent = amplitudeSlider.value;
      freqValue.textContent = parseFloat(frequencySlider.value).toFixed(2);
      sourceCountSpan.textContent = sources.length;
      resolutionSpan.textContent = `${gridW}×${gridH}`;
    }

    function pointerPos(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return { x: clientX - rect.left, y: clientY - rect.top };
    }

    function applyTool(event) {
      const p = pointerPos(event);
      const gx = Math.floor(p.x / scale);
      const gy = Math.floor(p.y / scale);

      if (tool === "drop") {
        disturb(gx, gy, 8, parseFloat(amplitudeSlider.value));
      } else if (tool === "source") {
        sources.push({ x: p.x, y: p.y, phase: Math.random() * Math.PI * 2 });
      } else if (tool === "wall") {
        paintWall(gx, gy, 5, 1);
      } else if (tool === "erase") {
        paintWall(gx, gy, 8, 0);
        sources = sources.filter(s => Math.hypot(s.x - p.x, s.y - p.y) > 22);
      } else if (tool === "drag") {
        disturb(gx, gy, 6, 55);
      } else if (tool === "slit") {
        makeDoubleSlit();
      }
    }

    function paintWall(cx, cy, radius, value) {
      for (let dy = -radius; dy <= radius; dy++) {
        for (let dx = -radius; dx <= radius; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (!inside(x, y)) continue;
          if (Math.hypot(dx, dy) <= radius) {
            walls[idx(x, y)] = value;
            current[idx(x, y)] = 0;
            previous[idx(x, y)] = 0;
          }
        }
      }
    }

    function makeDoubleSlit() {
      walls.fill(0);
      sources = [{ x: width * 0.22, y: height * 0.5, phase: 0 }];

      const wallX = Math.floor(gridW * 0.42);
      const slit1 = Math.floor(gridH * 0.42);
      const slit2 = Math.floor(gridH * 0.58);
      const slitHalf = Math.floor(gridH * 0.035);

      for (let y = Math.floor(gridH * 0.15); y < Math.floor(gridH * 0.85); y++) {
        const inSlit1 = Math.abs(y - slit1) < slitHalf;
        const inSlit2 = Math.abs(y - slit2) < slitHalf;
        if (!inSlit1 && !inSlit2) {
          for (let x = wallX - 2; x <= wallX + 2; x++) walls[idx(x, y)] = 1;
        }
      }
    }

    function makeTwoSources() {
      sources = [
        { x: width * 0.36, y: height * 0.48, phase: 0 },
        { x: width * 0.64, y: height * 0.48, phase: 0 },
      ];
    }

    canvas.addEventListener("pointerdown", (event) => {
      drawing = true;
      applyTool(event);
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!drawing) return;
      if (tool === "wall" || tool === "erase" || tool === "drag" || tool === "drop") applyTool(event);
    });

    window.addEventListener("pointerup", () => drawing = false);

    document.querySelectorAll("[data-tool]").forEach(button => {
      button.addEventListener("click", () => {
        tool = button.dataset.tool;
        document.querySelectorAll("[data-tool]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
      });
    });

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    clearButton.addEventListener("click", resetWorld);
    twoSourcesButton.addEventListener("click", makeTwoSources);
    rainButton.addEventListener("click", () => {
      rainMode = !rainMode;
      rainButton.classList.toggle("active", rainMode);
    });

    window.addEventListener("resize", resize);

    resize();
    animate();
  </script>
</body>
