  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");
    const clearButton = document.getElementById("clear");
    const stableOrbitButton = document.getElementById("stableOrbit");
    const solarSystemButton = document.getElementById("solarSystem");
    const starMassSlider = document.getElementById("starMass");
    const timeStepSlider = document.getElementById("timeStep");
    const throwPowerSlider = document.getElementById("throwPower");
    const trailLengthSlider = document.getElementById("trailLength");
    const massValue = document.getElementById("massValue");
    const timeValue = document.getElementById("timeValue");
    const throwValue = document.getElementById("throwValue");
    const trailValue = document.getElementById("trailValue");
    const planetCountSpan = document.getElementById("planetCount");
    const modeNameSpan = document.getElementById("modeName");

    let width = 0;
    let height = 0;
    let planets = [];
    let stars = [];
    let running = true;
    let tool = "throw";
    let dragging = false;
    let dragStart = null;
    let dragNow = null;
    let grabbedStar = null;
    let starField = [];

    const G = 0.12;
    const softening = 80;

    const toolNames = {
      throw: "惑星を投げる",
      star: "太陽を動かす",
      moon: "月つき惑星",
      blackhole: "重い星",
      erase: "消す",
      burst: "大量投入",
    };

    class Star {
      constructor(x, y, mass, radius, label = "太陽", color = "#facc15") {
        this.x = x;
        this.y = y;
        this.mass = mass;
        this.radius = radius;
        this.label = label;
        this.color = color;
      }

      draw() {
        ctx.save();
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
        ctx.fillStyle = this.color;
        ctx.shadowColor = this.color;
        ctx.shadowBlur = this.label === "重い星" ? 30 : 24;
        ctx.fill();

        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius * 2.1, 0, Math.PI * 2);
        ctx.strokeStyle = "rgba(250, 204, 21, 0.12)";
        ctx.lineWidth = 2;
        ctx.stroke();

        ctx.fillStyle = "rgba(255,255,255,0.9)";
        ctx.font = "bold 12px system-ui";
        ctx.textAlign = "center";
        ctx.fillText(this.label, this.x, this.y + this.radius + 18);
        ctx.restore();
      }
    }

    class Planet {
      constructor(x, y, vx, vy, radius = 6, color = null, parent = null) {
        this.x = x;
        this.y = y;
        this.vx = vx;
        this.vy = vy;
        this.radius = radius;
        this.color = color || randomPlanetColor();
        this.trail = [];
        this.dead = false;
        this.parent = parent;
      }

      update(dt) {
        let ax = 0;
        let ay = 0;

        for (const s of stars) {
          const dx = s.x - this.x;
          const dy = s.y - this.y;
          const r2 = dx * dx + dy * dy + softening;
          const r = Math.sqrt(r2);
          const force = G * s.mass / r2;
          ax += force * dx / r;
          ay += force * dy / r;

          if (r < s.radius + this.radius * 0.7) {
            this.dead = true;
          }
        }

        // 惑星同士の重力は弱め。月つき惑星や多体っぽい動きを少し出す。
        for (const other of planets) {
          if (other === this || other.dead) continue;
          const dx = other.x - this.x;
          const dy = other.y - this.y;
          const r2 = dx * dx + dy * dy + 60;
          const r = Math.sqrt(r2);
          if (r < 120) {
            const force = 0.018 * other.radius * other.radius / r2;
            ax += force * dx / r;
            ay += force * dy / r;
          }
        }

        this.vx += ax * dt;
        this.vy += ay * dt;
        this.x += this.vx * dt;
        this.y += this.vy * dt;

        this.trail.push({ x: this.x, y: this.y });
        const maxTrail = parseInt(trailLengthSlider.value, 10);
        if (this.trail.length > maxTrail) this.trail.shift();

        if (this.x < -500 || this.x > width + 500 || this.y < -500 || this.y > height + 500) {
          this.dead = true;
        }
      }

      drawTrail() {
        if (this.trail.length < 2) return;
        ctx.save();
        ctx.beginPath();
        ctx.moveTo(this.trail[0].x, this.trail[0].y);
        for (let i = 1; i < this.trail.length; i++) {
          ctx.lineTo(this.trail[i].x, this.trail[i].y);
        }
        ctx.strokeStyle = hexToRgba(this.color, 0.35);
        ctx.lineWidth = 1.5;
        ctx.stroke();
        ctx.restore();
      }

      draw() {
        ctx.save();
        ctx.beginPath();
        ctx.arc(this.x, this.y, this.radius, 0, Math.PI * 2);
        ctx.fillStyle = this.color;
        ctx.shadowColor = this.color;
        ctx.shadowBlur = 12;
        ctx.fill();

        // 速度ベクトルをうっすら表示。
        ctx.beginPath();
        ctx.moveTo(this.x, this.y);
        ctx.lineTo(this.x + this.vx * 9, this.y + this.vy * 9);
        ctx.strokeStyle = "rgba(255,255,255,0.28)";
        ctx.lineWidth = 1;
        ctx.stroke();
        ctx.restore();
      }
    }

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      resetStars();
      makeStarField();
    }

    function resetStars() {
      stars = [new Star(width * 0.5, height * 0.52, parseFloat(starMassSlider.value), 22, "太陽", "#facc15")];
    }

    function makeStarField() {
      starField = [];
      for (let i = 0; i < 220; i++) {
        starField.push({
          x: Math.random() * width,
          y: Math.random() * height,
          r: Math.random() * 1.4 + 0.2,
          a: Math.random() * 0.65 + 0.15,
        });
      }
    }

    function randomPlanetColor() {
      const colors = ["#60a5fa", "#34d399", "#f97316", "#c084fc", "#f472b6", "#22d3ee", "#a3e635"];
      return colors[Math.floor(Math.random() * colors.length)];
    }

    function hexToRgba(hex, alpha) {
      const h = hex.replace("#", "");
      const r = parseInt(h.substring(0, 2), 16);
      const g = parseInt(h.substring(2, 4), 16);
      const b = parseInt(h.substring(4, 6), 16);
      return `rgba(${r}, ${g}, ${b}, ${alpha})`;
    }

    function animate() {
      drawBackground();

      if (running) {
        const dt = parseFloat(timeStepSlider.value);
        const subSteps = 2;
        for (let s = 0; s < subSteps; s++) {
          for (const p of planets) p.update(dt / subSteps);
          planets = planets.filter(p => !p.dead);
        }
      }

      for (const p of planets) p.drawTrail();
      for (const s of stars) s.draw();
      for (const p of planets) p.draw();
      drawDragPreview();
      updateLabels();

      requestAnimationFrame(animate);
    }

    function drawBackground() {
      ctx.fillStyle = "rgba(2, 6, 23, 0.38)";
      ctx.fillRect(0, 0, width, height);

      for (const s of starField) {
        ctx.beginPath();
        ctx.arc(s.x, s.y, s.r, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(255,255,255,${s.a})`;
        ctx.fill();
      }
    }

    function drawDragPreview() {
      if (!dragging || !dragStart || !dragNow || tool === "star") return;

      ctx.save();
      ctx.beginPath();
      ctx.arc(dragStart.x, dragStart.y, tool === "blackhole" ? 12 : 7, 0, Math.PI * 2);
      ctx.fillStyle = tool === "blackhole" ? "#111827" : "#60a5fa";
      ctx.shadowColor = tool === "blackhole" ? "#e5e7eb" : "#60a5fa";
      ctx.shadowBlur = 16;
      ctx.fill();

      ctx.beginPath();
      ctx.moveTo(dragStart.x, dragStart.y);
      ctx.lineTo(dragNow.x, dragNow.y);
      ctx.strokeStyle = "rgba(255,255,255,0.75)";
      ctx.lineWidth = 2;
      ctx.stroke();

      const angle = Math.atan2(dragNow.y - dragStart.y, dragNow.x - dragStart.x);
      ctx.beginPath();
      ctx.moveTo(dragNow.x, dragNow.y);
      ctx.lineTo(dragNow.x - Math.cos(angle - 0.5) * 12, dragNow.y - Math.sin(angle - 0.5) * 12);
      ctx.lineTo(dragNow.x - Math.cos(angle + 0.5) * 12, dragNow.y - Math.sin(angle + 0.5) * 12);
      ctx.closePath();
      ctx.fillStyle = "rgba(255,255,255,0.75)";
      ctx.fill();
      ctx.restore();
    }

    function updateLabels() {
      if (stars[0] && stars[0].label === "太陽") stars[0].mass = parseFloat(starMassSlider.value);
      massValue.textContent = starMassSlider.value;
      timeValue.textContent = parseFloat(timeStepSlider.value).toFixed(2);
      throwValue.textContent = parseFloat(throwPowerSlider.value).toFixed(2);
      trailValue.textContent = trailLengthSlider.value;
      planetCountSpan.textContent = planets.length;
      modeNameSpan.textContent = toolNames[tool];
    }

    function pointerPos(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return { x: clientX - rect.left, y: clientY - rect.top };
    }

    function addThrownPlanet(start, end) {
      const power = parseFloat(throwPowerSlider.value);
      const vx = (end.x - start.x) * power;
      const vy = (end.y - start.y) * power;

      if (tool === "blackhole") {
        stars.push(new Star(start.x, start.y, 4200, 15, "重い星", "#111827"));
        return;
      }

      if (tool === "moon") {
        const planet = new Planet(start.x, start.y, vx, vy, 8, "#60a5fa");
        planets.push(planet);
        const moonDistance = 28;
        const moonSpeed = 1.65;
        planets.push(new Planet(start.x + moonDistance, start.y, vx, vy + moonSpeed, 3.5, "#e5e7eb", planet));
        return;
      }

      planets.push(new Planet(start.x, start.y, vx, vy, 5 + Math.random() * 3));
    }

    function addStableOrbit() {
      const s = stars[0];
      const r = Math.min(width, height) * (0.18 + Math.random() * 0.22);
      const angle = Math.random() * Math.PI * 2;
      const x = s.x + Math.cos(angle) * r;
      const y = s.y + Math.sin(angle) * r;
      const speed = Math.sqrt(G * s.mass / r);
      const vx = -Math.sin(angle) * speed;
      const vy = Math.cos(angle) * speed;
      planets.push(new Planet(x, y, vx, vy, 5 + Math.random() * 3));
    }

    function makeSolarSystem() {
      planets = [];
      resetStars();
      const s = stars[0];
      const distances = [90, 140, 200, 270, 350];
      const sizes = [4, 5, 6.5, 5, 9];
      const colors = ["#f97316", "#60a5fa", "#34d399", "#f472b6", "#c084fc"];
      for (let i = 0; i < distances.length; i++) {
        const r = distances[i];
        const angle = Math.random() * Math.PI * 2;
        const speed = Math.sqrt(G * s.mass / r);
        planets.push(new Planet(
          s.x + Math.cos(angle) * r,
          s.y + Math.sin(angle) * r,
          -Math.sin(angle) * speed,
          Math.cos(angle) * speed,
          sizes[i],
          colors[i]
        ));
      }
    }

    function addBurst(center) {
      const s = stars[0];
      for (let i = 0; i < 36; i++) {
        const angle = Math.random() * Math.PI * 2;
        const r = 30 + Math.random() * 70;
        const x = center.x + Math.cos(angle) * r;
        const y = center.y + Math.sin(angle) * r;
        const dx = x - s.x;
        const dy = y - s.y;
        const dist = Math.max(40, Math.hypot(dx, dy));
        const orbital = Math.sqrt(G * s.mass / dist);
        const sign = Math.random() < 0.5 ? 1 : -1;
        const vx = -dy / dist * orbital * sign + (Math.random() - 0.5) * 1.2;
        const vy = dx / dist * orbital * sign + (Math.random() - 0.5) * 1.2;
        planets.push(new Planet(x, y, vx, vy, 2.5 + Math.random() * 2.5));
      }
    }

    function eraseAt(p) {
      planets = planets.filter(planet => Math.hypot(planet.x - p.x, planet.y - p.y) > 24);
      stars = stars.filter((s, index) => index === 0 || Math.hypot(s.x - p.x, s.y - p.y) > s.radius + 14);
    }

    canvas.addEventListener("pointerdown", (event) => {
      const p = pointerPos(event);
      dragging = true;
      dragStart = p;
      dragNow = p;

      if (tool === "star") {
        grabbedStar = stars.find(s => Math.hypot(s.x - p.x, s.y - p.y) < s.radius + 20) || null;
      } else if (tool === "erase") {
        eraseAt(p);
      } else if (tool === "burst") {
        addBurst(p);
      }
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!dragging) return;
      const p = pointerPos(event);
      dragNow = p;

      if (tool === "star" && grabbedStar) {
        grabbedStar.x = p.x;
        grabbedStar.y = p.y;
      } else if (tool === "erase") {
        eraseAt(p);
      }
    });

    window.addEventListener("pointerup", () => {
      if (dragging && dragStart && dragNow && ["throw", "moon", "blackhole"].includes(tool)) {
        if (Math.hypot(dragNow.x - dragStart.x, dragNow.y - dragStart.y) > 8) {
          addThrownPlanet(dragStart, dragNow);
        }
      }
      dragging = false;
      dragStart = null;
      dragNow = null;
      grabbedStar = null;
    });

    document.querySelectorAll("[data-tool]").forEach(button => {
      button.addEventListener("click", () => {
        tool = button.dataset.tool;
        document.querySelectorAll("[data-tool]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
      });
    });

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    clearButton.addEventListener("click", () => planets = []);
    stableOrbitButton.addEventListener("click", addStableOrbit);
    solarSystemButton.addEventListener("click", makeSolarSystem);

    window.addEventListener("resize", resize);

    resize();
    makeSolarSystem();
    animate();
  </script>
