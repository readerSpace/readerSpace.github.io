  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const els = {
      toggleBtn: document.getElementById("toggleBtn"),
      resetBtn: document.getElementById("resetBtn"),
      exciteBtn: document.getElementById("exciteBtn"),
      inhibitBtn: document.getElementById("inhibitBtn"),
      inputStrength: document.getElementById("inputStrength"),
      threshold: document.getElementById("threshold"),
      leak: document.getElementById("leak"),
      delay: document.getElementById("delay"),
      rate: document.getElementById("rate"),
      inputValue: document.getElementById("inputValue"),
      thresholdValue: document.getElementById("thresholdValue"),
      leakValue: document.getElementById("leakValue"),
      delayValue: document.getElementById("delayValue"),
      rateValue: document.getElementById("rateValue"),
      vA: document.getElementById("vA"),
      vB: document.getElementById("vB"),
      spikes: document.getElementById("spikes"),
    };

    class Neuron {
      constructor(x, y, name) {
        this.x = x;
        this.y = y;
        this.name = name;
        this.v = 0;
        this.refractory = 0;
        this.flash = 0;
        this.history = [];
      }

      addInput(amount) {
        if (this.refractory <= 0) this.v += amount;
      }

      update(dt, threshold, leak) {
        if (this.refractory > 0) {
          this.refractory -= dt;
          this.v = 0;
          return false;
        }

        this.v -= this.v * leak * dt * 60;
        this.v = Math.max(-1.2, Math.min(2.5, this.v));

        let fired = false;
        if (this.v >= threshold) {
          this.v = 0;
          this.refractory = 0.55;
          this.flash = 1;
          fired = true;
        }

        this.flash = Math.max(0, this.flash - dt * 2.8);
        this.history.push(this.v);
        if (this.history.length > 220) this.history.shift();
        return fired;
      }
    }

    let A, B, particles, paused, lastTime, autoClock, spikeCount;

    function reset() {
      A = new Neuron(250, 225, "ニューロンA");
      B = new Neuron(650, 225, "ニューロンB");
      particles = [];
      paused = false;
      lastTime = performance.now();
      autoClock = 0;
      spikeCount = 0;
      els.toggleBtn.textContent = "一時停止";
    }

    function params() {
      return {
        input: Number(els.inputStrength.value),
        threshold: Number(els.threshold.value),
        leak: Number(els.leak.value),
        delay: Number(els.delay.value),
        rate: Number(els.rate.value),
      };
    }

    function syncLabels() {
      const p = params();
      els.inputValue.textContent = p.input.toFixed(2);
      els.thresholdValue.textContent = p.threshold.toFixed(2);
      els.leakValue.textContent = p.leak.toFixed(3);
      els.delayValue.textContent = p.delay.toFixed(2) + "秒";
      els.rateValue.textContent = p.rate.toFixed(1) + "回/秒";
      els.vA.textContent = A.v.toFixed(2);
      els.vB.textContent = B.v.toFixed(2);
      els.spikes.textContent = spikeCount;
    }

    function addParticle(from, to, kind, strength, duration) {
      particles.push({ from, to, kind, strength, duration, t: 0 });
    }

    function manualExcite() {
      const p = params();
      addParticle({ x: 60, y: A.y - 55 }, A, "excite", p.input, 0.35);
    }

    function manualInhibit() {
      const p = params();
      addParticle({ x: 60, y: A.y + 55 }, A, "inhibit", -p.input * 0.8, 0.35);
    }

    function update(dt) {
      const p = params();

      autoClock += dt;
      const interval = p.rate > 0 ? 1 / p.rate : Infinity;
      while (autoClock > interval) {
        autoClock -= interval;
        const inhibitory = Math.random() < 0.25;
        if (inhibitory) manualInhibit();
        else manualExcite();
      }

      for (const particle of particles) {
        particle.t += dt;
        if (particle.t >= particle.duration && !particle.delivered) {
          particle.to.addInput(particle.strength);
          particle.delivered = true;
        }
      }
      particles = particles.filter(pt => pt.t < pt.duration + 0.05);

      const firedA = A.update(dt, p.threshold, p.leak);
      const firedB = B.update(dt, p.threshold, p.leak);

      if (firedA) {
        spikeCount++;
        addParticle(A, B, "spike", p.input * 0.9, p.delay);
      }
      if (firedB) spikeCount++;
    }

    function drawNeuron(n, threshold) {
      const radius = 58;
      ctx.save();
      ctx.translate(n.x, n.y);

      const glow = n.flash;
      if (glow > 0) {
        ctx.beginPath();
        ctx.arc(0, 0, radius + glow * 38, 0, Math.PI * 2);
        ctx.fillStyle = `rgba(245, 158, 11, ${0.22 * glow})`;
        ctx.fill();
      }

      ctx.beginPath();
      ctx.arc(0, 0, radius, 0, Math.PI * 2);
      ctx.fillStyle = n.refractory > 0 ? "#e5e7eb" : "#ffffff";
      ctx.fill();
      ctx.lineWidth = 4;
      ctx.strokeStyle = n.flash > 0 ? "#f59e0b" : "#374151";
      ctx.stroke();

      const level = Math.max(0, Math.min(1, n.v / threshold));
      ctx.beginPath();
      ctx.arc(0, 0, radius - 10, Math.PI / 2, Math.PI / 2 - level * Math.PI * 2, true);
      ctx.lineWidth = 9;
      ctx.strokeStyle = level > 0.85 ? "#ef4444" : "#2563eb";
      ctx.stroke();

      ctx.fillStyle = "#111827";
      ctx.textAlign = "center";
      ctx.font = "700 17px system-ui";
      ctx.fillText(n.name, 0, 5);

      ctx.font = "13px system-ui";
      if (n.refractory > 0) ctx.fillText("不応期", 0, 28);
      else ctx.fillText("膜電位 " + n.v.toFixed(2), 0, 28);

      ctx.restore();
    }

    function drawSynapse() {
      ctx.save();
      ctx.strokeStyle = "#9ca3af";
      ctx.lineWidth = 5;
      ctx.beginPath();
      ctx.moveTo(A.x + 58, A.y);
      ctx.bezierCurveTo(380, 160, 520, 160, B.x - 58, B.y);
      ctx.stroke();

      ctx.fillStyle = "#6b7280";
      for (let i = 0; i < 5; i++) {
        const x = 430 + i * 18;
        ctx.beginPath();
        ctx.arc(x, 166 + Math.sin(i) * 7, 4, 0, Math.PI * 2);
        ctx.fill();
      }

      ctx.fillStyle = "#374151";
      ctx.font = "14px system-ui";
      ctx.textAlign = "center";
      ctx.fillText("シナプス：発火後、少し遅れて信号が届く", 450, 128);
      ctx.restore();
    }

    function drawParticles() {
      for (const pt of particles) {
        const progress = Math.min(1, pt.t / pt.duration);
        const x = pt.from.x + (pt.to.x - pt.from.x) * progress;
        const y = pt.from.y + (pt.to.y - pt.from.y) * progress;

        ctx.beginPath();
        ctx.arc(x, y, pt.kind === "spike" ? 9 : 7, 0, Math.PI * 2);
        ctx.fillStyle = pt.kind === "inhibit" ? "#3b82f6" : pt.kind === "spike" ? "#f59e0b" : "#ef4444";
        ctx.fill();
      }
    }

    function drawGraph(n, x, y, w, h, color) {
      ctx.save();
      ctx.strokeStyle = "#d1d5db";
      ctx.strokeRect(x, y, w, h);
      ctx.fillStyle = "#6b7280";
      ctx.font = "12px system-ui";
      ctx.fillText(n.name + "の膜電位", x + 10, y - 8);

      ctx.beginPath();
      for (let i = 0; i < n.history.length; i++) {
        const gx = x + (i / 219) * w;
        const gy = y + h - ((n.history[i] + 1.2) / 3.7) * h;
        if (i === 0) ctx.moveTo(gx, gy);
        else ctx.lineTo(gx, gy);
      }
      ctx.strokeStyle = color;
      ctx.lineWidth = 2;
      ctx.stroke();
      ctx.restore();
    }

    function drawThresholdLine(threshold) {
      ctx.save();
      ctx.strokeStyle = "rgba(239, 68, 68, 0.65)";
      ctx.setLineDash([8, 6]);
      ctx.lineWidth = 2;
      ctx.beginPath();
      ctx.moveTo(140, 80);
      ctx.lineTo(760, 80);
      ctx.stroke();
      ctx.setLineDash([]);
      ctx.fillStyle = "#ef4444";
      ctx.font = "13px system-ui";
      ctx.fillText("しきい値を超えると発火", 145, 72);
      ctx.restore();
    }

    function draw() {
      const p = params();
      ctx.clearRect(0, 0, canvas.width, canvas.height);

      drawThresholdLine(p.threshold);
      drawSynapse();
      drawNeuron(A, p.threshold);
      drawNeuron(B, p.threshold);
      drawParticles();
      drawGraph(A, 60, 390, 360, 90, "#2563eb");
      drawGraph(B, 480, 390, 360, 90, "#7c3aed");
    }

    function loop(now) {
      const dt = Math.min(0.04, (now - lastTime) / 1000);
      lastTime = now;
      if (!paused) update(dt);
      syncLabels();
      draw();
      requestAnimationFrame(loop);
    }

    els.toggleBtn.addEventListener("click", () => {
      paused = !paused;
      els.toggleBtn.textContent = paused ? "再開" : "一時停止";
    });
    els.resetBtn.addEventListener("click", reset);
    els.exciteBtn.addEventListener("click", manualExcite);
    els.inhibitBtn.addEventListener("click", manualInhibit);
    document.querySelectorAll("input").forEach(input => input.addEventListener("input", syncLabels));

    reset();
    requestAnimationFrame(loop);
  </script>
