  <script>
    const canvas = document.getElementById('cv');
    const ctx = canvas.getContext('2d');

    const noiseSlider = document.getElementById('noise');
    const seaSlider = document.getElementById('sea');
    const trailSlider = document.getElementById('trail');
    const multipathBox = document.getElementById('multipath');
    const ductingBox = document.getElementById('ducting');
    const showTrueBox = document.getElementById('showTrue');

    const noiseVal = document.getElementById('noiseVal');
    const seaVal = document.getElementById('seaVal');
    const trailVal = document.getElementById('trailVal');

    let W, H, cx, cy, scale;
    let sweep = 0;
    let blips = [];

    const radarRange = 520;
    const seaY = 150;

    let targets = [];

    function resetTargets() {
      targets = [
        { x: -220, y: -120, vx: 0.45, vy: 0.12 },
        { x: 260, y: 70, vx: -0.28, vy: -0.08 },
        { x: 60, y: -260, vx: 0.08, vy: 0.35 }
      ];
      blips = [];
    }

    function resize() {
      canvas.width = canvas.clientWidth * devicePixelRatio;
      canvas.height = canvas.clientHeight * devicePixelRatio;
      ctx.setTransform(devicePixelRatio, 0, 0, devicePixelRatio, 0, 0);
      W = canvas.clientWidth;
      H = canvas.clientHeight;
      cx = W / 2;
      cy = H / 2;
      scale = Math.min(W, H) / 900;
    }

    window.addEventListener('resize', resize);
    document.getElementById('reset').addEventListener('click', resetTargets);
    resize();
    resetTargets();

    function worldToScreen(p) {
      return { x: cx + p.x * scale, y: cy + p.y * scale };
    }

    function length(x, y) {
      return Math.sqrt(x * x + y * y);
    }

    function angleDiff(a, b) {
      let d = Math.atan2(Math.sin(a - b), Math.cos(a - b));
      return Math.abs(d);
    }

    function addBlip(x, y, type, strength = 1) {
      if (length(x, y) > radarRange) return;
      blips.push({ x, y, type, life: 1, strength });
    }

    function updateTargets() {
      for (const t of targets) {
        t.x += t.vx;
        t.y += t.vy;
        if (Math.abs(t.x) > 380) t.vx *= -1;
        if (Math.abs(t.y) > 320) t.vy *= -1;
      }
    }

    function scan() {
      const noise = parseFloat(noiseSlider.value);
      const sea = parseFloat(seaSlider.value);
      const beamWidth = 0.025;

      for (const t of targets) {
        const a = Math.atan2(t.y, t.x);
        const r = length(t.x, t.y);

        if (angleDiff(a, sweep) < beamWidth) {
          addBlip(t.x, t.y, 'realEcho', 1);

          // 海面反射：鏡像位置から戻ったように見える
          if (multipathBox.checked && Math.random() < sea) {
            const mirroredY = 2 * seaY - t.y;
            const delay = 1.15 + Math.random() * 0.35;
            addBlip(t.x * delay, mirroredY * delay, 'ghost', 0.9);
          }

          // 多重反射：余分な経路長のせいで距離が大きく見える
          if (multipathBox.checked && Math.random() < sea * 0.55) {
            const ghostR = r * (1.3 + Math.random() * 0.6);
            const ghostA = a + (Math.random() - 0.5) * 0.25;
            addBlip(Math.cos(ghostA) * ghostR, Math.sin(ghostA) * ghostR, 'ghost', 0.75);
          }
        }
      }

      // 大気ダクト：遠くの反射が折れ曲がって入ってきたような誤検出
      if (ductingBox.checked && Math.random() < noise * 0.25) {
        const a = sweep + (Math.random() - 0.5) * 0.04;
        const r = 300 + Math.random() * 210;
        addBlip(Math.cos(a) * r, Math.sin(a) * r, 'ghost', 0.65);
      }

      // ランダムノイズ：しきい値を超えた点が目標っぽくなる
      if (Math.random() < noise * 0.7) {
        const a = sweep + (Math.random() - 0.5) * 0.08;
        const r = 40 + Math.random() * radarRange;
        addBlip(Math.cos(a) * r, Math.sin(a) * r, 'noise', 0.5);
      }
    }

    function drawGrid() {
      ctx.save();
      ctx.translate(cx, cy);
      ctx.scale(scale, scale);

      ctx.strokeStyle = 'rgba(80, 220, 140, 0.22)';
      ctx.lineWidth = 1 / scale;

      for (let r = 100; r <= radarRange; r += 100) {
        ctx.beginPath();
        ctx.arc(0, 0, r, 0, Math.PI * 2);
        ctx.stroke();
      }

      for (let a = 0; a < Math.PI * 2; a += Math.PI / 6) {
        ctx.beginPath();
        ctx.moveTo(0, 0);
        ctx.lineTo(Math.cos(a) * radarRange, Math.sin(a) * radarRange);
        ctx.stroke();
      }

      // 海面・地表反射面
      ctx.strokeStyle = 'rgba(56, 189, 248, 0.45)';
      ctx.setLineDash([8 / scale, 6 / scale]);
      ctx.beginPath();
      ctx.moveTo(-radarRange, seaY);
      ctx.lineTo(radarRange, seaY);
      ctx.stroke();
      ctx.setLineDash([]);

      ctx.restore();
    }

    function drawSweep() {
      ctx.save();
      ctx.translate(cx, cy);
      ctx.scale(scale, scale);

      const grad = ctx.createRadialGradient(0, 0, 0, 0, 0, radarRange);
      grad.addColorStop(0, 'rgba(80, 255, 130, 0.35)');
      grad.addColorStop(1, 'rgba(80, 255, 130, 0.02)');

      ctx.fillStyle = grad;
      ctx.beginPath();
      ctx.moveTo(0, 0);
      ctx.arc(0, 0, radarRange, sweep - 0.10, sweep, false);
      ctx.closePath();
      ctx.fill();

      ctx.strokeStyle = 'rgba(190, 255, 210, 0.9)';
      ctx.lineWidth = 2 / scale;
      ctx.beginPath();
      ctx.moveTo(0, 0);
      ctx.lineTo(Math.cos(sweep) * radarRange, Math.sin(sweep) * radarRange);
      ctx.stroke();

      ctx.restore();
    }

    function drawTargets() {
      if (!showTrueBox.checked) return;
      ctx.save();
      ctx.translate(cx, cy);
      ctx.scale(scale, scale);
      for (const t of targets) {
        ctx.fillStyle = '#22c55e';
        ctx.beginPath();
        ctx.arc(t.x, t.y, 7 / scale, 0, Math.PI * 2);
        ctx.fill();

        ctx.strokeStyle = 'rgba(34, 197, 94, 0.5)';
        ctx.beginPath();
        ctx.arc(t.x, t.y, 16 / scale, 0, Math.PI * 2);
        ctx.stroke();
      }
      ctx.restore();
    }

    function drawBlips() {
      const decay = parseFloat(trailSlider.value);
      ctx.save();
      ctx.translate(cx, cy);
      ctx.scale(scale, scale);

      for (const b of blips) {
        let color;
        if (b.type === 'realEcho') color = `rgba(56, 189, 248, ${b.life})`;
        else if (b.type === 'ghost') color = `rgba(249, 115, 22, ${b.life})`;
        else color = `rgba(250, 204, 21, ${b.life})`;

        ctx.fillStyle = color;
        ctx.beginPath();
        ctx.arc(b.x, b.y, (5 + 5 * b.strength) / scale, 0, Math.PI * 2);
        ctx.fill();

        b.life *= decay;
      }

      blips = blips.filter(b => b.life > 0.035);
      ctx.restore();
    }

    function drawLabels() {
      ctx.fillStyle = 'rgba(234, 240, 255, 0.8)';
      ctx.font = '13px system-ui';
      ctx.fillText('中央：レーダー / 点：検出された反射', 18, 28);
      ctx.fillText('水色の破線：海面・地表の反射面モデル', 18, 50);
    }

    function updateUI() {
      noiseVal.textContent = Number(noiseSlider.value).toFixed(2);
      seaVal.textContent = Number(seaSlider.value).toFixed(2);
      trailVal.textContent = Number(trailSlider.value).toFixed(2);
    }

    function loop() {
      updateUI();
      updateTargets();
      scan();

      ctx.clearRect(0, 0, W, H);
      drawGrid();
      drawSweep();
      drawBlips();
      drawTargets();
      drawLabels();

      sweep += 0.018;
      if (sweep > Math.PI * 2) sweep -= Math.PI * 2;
      requestAnimationFrame(loop);
    }

    loop();
  </script>
</body>

実在目標の反射
海面反射によるゴースト
多重反射による距離誤認
ノイズのしきい値超え
大気ダクト風の遠距離誤検出
