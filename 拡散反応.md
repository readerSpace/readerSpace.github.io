  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");
    const seedButton = document.getElementById("seed");
    const clearButton = document.getElementById("clear");
    const invertButton = document.getElementById("invert");
    const feedSlider = document.getElementById("feed");
    const killSlider = document.getElementById("kill");
    const brushSlider = document.getElementById("brush");
    const speedSlider = document.getElementById("speed");
    const feedValue = document.getElementById("feedValue");
    const killValue = document.getElementById("killValue");
    const brushValue = document.getElementById("brushValue");
    const speedValue = document.getElementById("speedValue");
    const generationSpan = document.getElementById("generation");
    const resolutionSpan = document.getElementById("resolution");

    let simW = 0;
    let simH = 0;
    let scale = 3;
    let a = [];
    let b = [];
    let nextA = [];
    let nextB = [];
    let imageData;
    let running = true;
    let drawing = false;
    let inverted = false;
    let generation = 0;

    const params = {
      dA: 1.0,
      dB: 0.5,
      feed: 0.035,
      kill: 0.060,
    };

    const presets = {
      zebra:  { feed: 0.035, kill: 0.060, name: "シマウマ" },
      leopard:{ feed: 0.055, kill: 0.062, name: "ヒョウ柄" },
      coral:  { feed: 0.054, kill: 0.063, name: "サンゴ" },
      maze:   { feed: 0.029, kill: 0.057, name: "迷路" },
      spots:  { feed: 0.025, kill: 0.055, name: "水玉" },
      chaos:  { feed: 0.018, kill: 0.050, name: "カオス" },
    };

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      const width = window.innerWidth;
      const height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);

      // 画面が大きすぎると重くなるため、内部解像度を落として描画する。
      scale = width < 700 ? 4 : 3;
      simW = Math.floor(width / scale);
      simH = Math.floor(height / scale);
      imageData = ctx.createImageData(simW, simH);
      resetGrid();
    }

    function index(x, y) {
      return y * simW + x;
    }

    function resetGrid() {
      const n = simW * simH;
      a = new Float32Array(n);
      b = new Float32Array(n);
      nextA = new Float32Array(n);
      nextB = new Float32Array(n);

      for (let i = 0; i < n; i++) {
        a[i] = 1;
        b[i] = 0;
      }

      generation = 0;
      seedPattern();
    }

    function seedPattern() {
      for (let i = 0; i < 18; i++) {
        const x = Math.floor(Math.random() * simW);
        const y = Math.floor(Math.random() * simH);
        addChemicalB(x, y, 8 + Math.random() * 18);
      }

      // 中央に大きめのタネ。最初から変化が見えて楽しい。
      addChemicalB(Math.floor(simW / 2), Math.floor(simH / 2), Math.min(simW, simH) * 0.08);
    }

    function addChemicalB(cx, cy, radius) {
      const r = Math.floor(radius);
      for (let dy = -r; dy <= r; dy++) {
        for (let dx = -r; dx <= r; dx++) {
          const x = cx + dx;
          const y = cy + dy;
          if (x < 1 || x >= simW - 1 || y < 1 || y >= simH - 1) continue;
          const d = Math.hypot(dx, dy);
          if (d <= r) {
            const i = index(x, y);
            b[i] = Math.max(b[i], 1 - d / (r + 1));
            a[i] = Math.min(a[i], 0.45);
          }
        }
      }
    }

    function laplace(arr, x, y) {
      let sum = 0;
      sum += arr[index(x, y)] * -1.0;
      sum += arr[index(x - 1, y)] * 0.2;
      sum += arr[index(x + 1, y)] * 0.2;
      sum += arr[index(x, y - 1)] * 0.2;
      sum += arr[index(x, y + 1)] * 0.2;
      sum += arr[index(x - 1, y - 1)] * 0.05;
      sum += arr[index(x + 1, y - 1)] * 0.05;
      sum += arr[index(x - 1, y + 1)] * 0.05;
      sum += arr[index(x + 1, y + 1)] * 0.05;
      return sum;
    }

    function stepSimulation() {
      const feed = parseFloat(feedSlider.value);
      const kill = parseFloat(killSlider.value);

      for (let y = 1; y < simH - 1; y++) {
        for (let x = 1; x < simW - 1; x++) {
          const i = index(x, y);
          const A = a[i];
          const B = b[i];
          const reaction = A * B * B;

          let newA = A + (params.dA * laplace(a, x, y) - reaction + feed * (1 - A));
          let newB = B + (params.dB * laplace(b, x, y) + reaction - (kill + feed) * B);

          nextA[i] = Math.max(0, Math.min(1, newA));
          nextB[i] = Math.max(0, Math.min(1, newB));
        }
      }

      // 端は固定して暴走を防ぐ。
      for (let x = 0; x < simW; x++) {
        const top = index(x, 0);
        const bottom = index(x, simH - 1);
        nextA[top] = nextA[bottom] = 1;
        nextB[top] = nextB[bottom] = 0;
      }
      for (let y = 0; y < simH; y++) {
        const left = index(0, y);
        const right = index(simW - 1, y);
        nextA[left] = nextA[right] = 1;
        nextB[left] = nextB[right] = 0;
      }

      let temp = a; a = nextA; nextA = temp;
      temp = b; b = nextB; nextB = temp;
      generation++;
    }

    function render() {
      const data = imageData.data;
      for (let y = 0; y < simH; y++) {
        for (let x = 0; x < simW; x++) {
          const i = index(x, y);
          const p = i * 4;
          let v = Math.floor((a[i] - b[i]) * 255);
          v = Math.max(0, Math.min(255, v));
          if (inverted) v = 255 - v;

          // 少し色をつけて「自然模様」っぽく見せる。
          data[p + 0] = inverted ? v : Math.floor(v * 0.75);
          data[p + 1] = inverted ? Math.floor(v * 0.85) : Math.floor(70 + v * 0.72);
          data[p + 2] = inverted ? Math.floor(255 - v * 0.35) : Math.floor(120 + v * 0.50);
          data[p + 3] = 255;
        }
      }

      const off = document.createElement("canvas");
      off.width = simW;
      off.height = simH;
      const offCtx = off.getContext("2d");
      offCtx.putImageData(imageData, 0, 0);
      ctx.imageSmoothingEnabled = false;
      ctx.drawImage(off, 0, 0, window.innerWidth, window.innerHeight);

      updateLabels();
    }

    function animate() {
      if (running) {
        const steps = parseInt(speedSlider.value, 10);
        for (let i = 0; i < steps; i++) stepSimulation();
      }
      render();
      requestAnimationFrame(animate);
    }

    function updateLabels() {
      feedValue.textContent = parseFloat(feedSlider.value).toFixed(3);
      killValue.textContent = parseFloat(killSlider.value).toFixed(3);
      brushValue.textContent = `${brushSlider.value}px`;
      speedValue.textContent = `${speedSlider.value} 回/フレーム`;
      generationSpan.textContent = generation;
      resolutionSpan.textContent = `${simW}×${simH}`;
    }

    function pointerToSim(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return {
        x: Math.floor((clientX - rect.left) / scale),
        y: Math.floor((clientY - rect.top) / scale),
      };
    }

    function drawAtPointer(event) {
      const p = pointerToSim(event);
      addChemicalB(p.x, p.y, parseInt(brushSlider.value, 10));
    }

    canvas.addEventListener("pointerdown", (event) => {
      drawing = true;
      drawAtPointer(event);
    });

    canvas.addEventListener("pointermove", (event) => {
      if (drawing) drawAtPointer(event);
    });

    window.addEventListener("pointerup", () => drawing = false);

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    seedButton.addEventListener("click", seedPattern);
    clearButton.addEventListener("click", resetGrid);
    invertButton.addEventListener("click", () => inverted = !inverted);

    document.querySelectorAll("[data-preset]").forEach(button => {
      button.addEventListener("click", () => {
        const preset = presets[button.dataset.preset];
        feedSlider.value = preset.feed;
        killSlider.value = preset.kill;
        document.querySelectorAll("[data-preset]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
        resetGrid();
      });
    });

    window.addEventListener("resize", resize);

    resize();
    animate();
  </script>
</body>
