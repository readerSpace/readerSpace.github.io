<script>
const sideCanvas = document.getElementById('sideCanvas');
const topCanvas = document.getElementById('topCanvas');
const sctx = sideCanvas.getContext('2d');
const tctx = topCanvas.getContext('2d');

const spinEl = document.getElementById('spin');
const tiltEl = document.getElementById('tilt');
const gravityEl = document.getElementById('gravity');
const frictionEl = document.getElementById('friction');
const speedEl = document.getElementById('speed');

const spinText = document.getElementById('spinText');
const tiltText = document.getElementById('tiltText');
const gravityText = document.getElementById('gravityText');
const frictionText = document.getElementById('frictionText');
const speedText = document.getElementById('speedText');

const precessionStat = document.getElementById('precessionStat');
const spinStat = document.getElementById('spinStat');
const tiltStat = document.getElementById('tiltStat');
const stateStat = document.getElementById('stateStat');

const pauseBtn = document.getElementById('pauseBtn');
const resetBtn = document.getElementById('resetBtn');

let spin = 80;
let tilt = 25 * Math.PI / 180;
let phi = 0;
let selfAngle = 0;
let running = true;
let trail = [];
let last = performance.now();

function clamp(x,a,b){ return Math.max(a, Math.min(b, x)); }
function radToDeg(x){ return x * 180 / Math.PI; }

function reset(){
  spin = Number(spinEl.value);
  tilt = Number(tiltEl.value) * Math.PI / 180;
  phi = 0;
  selfAngle = 0;
  trail = [];
}

function updateLabels(){
  spinText.textContent = spinEl.value;
  tiltText.textContent = `${tiltEl.value}°`;
  gravityText.textContent = Number(gravityEl.value).toFixed(2);
  frictionText.textContent = Number(frictionEl.value).toFixed(2);
  speedText.textContent = Number(speedEl.value).toFixed(1);
}

function step(dt){
  const g = Number(gravityEl.value);
  const friction = Number(frictionEl.value);
  const viewSpeed = Number(speedEl.value);
  dt *= viewSpeed;

  const L = Math.max(8, spin);
  const torque = g * Math.sin(tilt) * 42;
  const omegaP = torque / L;

  phi += omegaP * dt;
  selfAngle += spin * 0.25 * dt;

  spin -= friction * dt * (0.8 + spin * 0.015);
  spin = Math.max(0, spin);

  const stable = spin / (20 + g * 10);
  const fallRate = clamp(0.25 - stable * 0.055, -0.04, 0.35);
  tilt += fallRate * dt;
  tilt = clamp(tilt, 0.04, Math.PI / 2.05);

  const r = Math.sin(tilt);
  trail.push({ x: Math.cos(phi) * r, y: Math.sin(phi) * r });
  if(trail.length > 220) trail.shift();

  precessionStat.textContent = `${omegaP.toFixed(2)} rad/s`;
  spinStat.textContent = spin.toFixed(1);
  tiltStat.textContent = `${radToDeg(tilt).toFixed(1)}°`;
  stateStat.textContent = spin < 8 || tilt > 78 * Math.PI/180 ? '倒れそう' : '歳差運動中';
}

function axisVector(){
  return {
    x: Math.sin(tilt) * Math.cos(phi),
    y: Math.sin(tilt) * Math.sin(phi),
    z: Math.cos(tilt)
  };
}

function drawSide(){
  const w = sideCanvas.width, h = sideCanvas.height;
  const cx = w/2, ground = h - 74;
  sctx.clearRect(0,0,w,h);
  sctx.fillStyle = '#f8fafc';
  sctx.fillRect(0,0,w,h);

  sctx.strokeStyle = '#e2e8f0';
  sctx.lineWidth = 1;
  for(let x=0;x<w;x+=40){ sctx.beginPath(); sctx.moveTo(x,0); sctx.lineTo(x,h); sctx.stroke(); }
  for(let y=0;y<h;y+=40){ sctx.beginPath(); sctx.moveTo(0,y); sctx.lineTo(w,y); sctx.stroke(); }

  sctx.strokeStyle = '#334155';
  sctx.lineWidth = 4;
  sctx.beginPath(); sctx.moveTo(40,ground); sctx.lineTo(w-40,ground); sctx.stroke();

  const v = axisVector();
  const visibleTilt = v.x;
  const axisLen = 250;
  const pivot = { x: cx, y: ground };
  const tip = { x: cx + visibleTilt * axisLen, y: ground - v.z * axisLen };

  sctx.strokeStyle = 'rgba(15,23,42,0.28)';
  sctx.setLineDash([6,6]);
  sctx.beginPath(); sctx.moveTo(cx,ground); sctx.lineTo(cx,ground-axisLen); sctx.stroke();
  sctx.setLineDash([]);
  sctx.fillStyle = '#64748b';
  sctx.font = 'bold 13px system-ui';
  sctx.fillText('鉛直方向', cx+8, ground-axisLen+18);

  const bodyCenter = {
    x: pivot.x + (tip.x - pivot.x) * 0.43,
    y: pivot.y + (tip.y - pivot.y) * 0.43
  };
  const angle = Math.atan2(tip.y-pivot.y, tip.x-pivot.x);

  sctx.save();
  sctx.translate(bodyCenter.x, bodyCenter.y);
  sctx.rotate(angle + Math.PI/2);

  const spinStripe = Math.sin(selfAngle);
  const grad = sctx.createLinearGradient(-70,0,70,0);
  grad.addColorStop(0, '#1d4ed8');
  grad.addColorStop(0.5 + 0.35*spinStripe, '#93c5fd');
  grad.addColorStop(1, '#1e3a8a');
  sctx.fillStyle = grad;
  sctx.beginPath();
  sctx.moveTo(0,-88);
  sctx.bezierCurveTo(82,-35,70,55,0,108);
  sctx.bezierCurveTo(-70,55,-82,-35,0,-88);
  sctx.fill();
  sctx.strokeStyle = '#0f172a';
  sctx.lineWidth = 3;
  sctx.stroke();

  sctx.fillStyle = '#f97316';
  sctx.beginPath();
  sctx.arc(0, -18, 12, 0, Math.PI*2);
  sctx.fill();

  sctx.restore();

  sctx.strokeStyle = '#dc2626';
  sctx.lineWidth = 5;
  sctx.beginPath(); sctx.moveTo(pivot.x,pivot.y); sctx.lineTo(tip.x,tip.y); sctx.stroke();

  sctx.fillStyle = '#dc2626';
  sctx.beginPath(); sctx.arc(tip.x, tip.y, 7, 0, Math.PI*2); sctx.fill();

  sctx.fillStyle = '#0f172a';
  sctx.font = 'bold 16px system-ui';
  sctx.fillText('横から見た傾き', 22, 32);
}

function drawTop(){
  const w = topCanvas.width, h = topCanvas.height;
  const cx = w/2, cy = h/2;
  const R = 190;
  tctx.clearRect(0,0,w,h);
  tctx.fillStyle = '#f8fafc';
  tctx.fillRect(0,0,w,h);

  tctx.strokeStyle = '#e2e8f0';
  tctx.lineWidth = 1;
  for(let r=50;r<=220;r+=50){ tctx.beginPath(); tctx.arc(cx,cy,r,0,Math.PI*2); tctx.stroke(); }
  tctx.beginPath(); tctx.moveTo(cx,30); tctx.lineTo(cx,h-30); tctx.stroke();
  tctx.beginPath(); tctx.moveTo(30,cy); tctx.lineTo(w-30,cy); tctx.stroke();

  tctx.strokeStyle = '#94a3b8';
  tctx.lineWidth = 3;
  tctx.beginPath(); tctx.arc(cx,cy,R,0,Math.PI*2); tctx.stroke();

  tctx.strokeStyle = '#2563eb';
  tctx.lineWidth = 3;
  tctx.beginPath();
  for(let i=0;i<trail.length;i++){
    const p = trail[i];
    const x = cx + p.x * R;
    const y = cy + p.y * R;
    if(i===0) tctx.moveTo(x,y); else tctx.lineTo(x,y);
  }
  tctx.stroke();

  const v = axisVector();
  const tip = { x: cx + v.x * R, y: cy + v.y * R };

  tctx.strokeStyle = '#dc2626';
  tctx.lineWidth = 5;
  tctx.beginPath(); tctx.moveTo(cx,cy); tctx.lineTo(tip.x,tip.y); tctx.stroke();

  tctx.fillStyle = '#dc2626';
  tctx.beginPath(); tctx.arc(tip.x,tip.y,9,0,Math.PI*2); tctx.fill();

  tctx.fillStyle = '#0f172a';
  tctx.beginPath(); tctx.arc(cx,cy,5,0,Math.PI*2); tctx.fill();

  tctx.fillStyle = '#0f172a';
  tctx.font = 'bold 16px system-ui';
  tctx.fillText('上から見た軸の向き', 22, 32);
  tctx.font = '13px system-ui';
  tctx.fillStyle = '#475569';
  tctx.fillText('青い線: 軸の先端が描く軌跡', 22, 54);
}

function loop(now){
  const dt = Math.min(0.04, (now - last)/1000);
  last = now;
  updateLabels();
  if(running) step(dt);
  drawSide();
  drawTop();
  requestAnimationFrame(loop);
}

pauseBtn.addEventListener('click', () => {
  running = !running;
  pauseBtn.textContent = running ? '停止' : '再開';
});
resetBtn.addEventListener('click', reset);
[spinEl, tiltEl].forEach(el => el.addEventListener('input', reset));

reset();
requestAnimationFrame(loop);
</script>
