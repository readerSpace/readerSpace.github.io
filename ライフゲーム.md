    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");

    const eraseModeButton = document.getElementById("eraseMode");
    const speedSlider = document.getElementById("speed");
    const cellSizeSlider = document.getElementById("cellSize");
    const speedValue = document.getElementById("speedValue");
    const cellSizeValue = document.getElementById("cellSizeValue");

    const clearButton = document.getElementById("clear");

    let width = 0;
    let height = 0;
    let cellSize = 10;
    let cols = 0;
    let rows = 0;
    let grid = [];
    let nextGrid = [];

    let drawing = false;
    let drawState = 1;
    let generation = 0;
    let lastUpdate = 0;
    let selectedPattern = null;

    const patterns = {

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      rebuildGrid(true);
    }

    function createGrid(c, r) {
      return Array.from({ length: r }, () => Array(c).fill(0));
    }

    function rebuildGrid(keepOld) {
      const oldGrid = grid;
      const oldRows = rows;
      const oldCols = cols;

      cellSize = parseInt(cellSizeSlider.value, 10);
      cols = Math.ceil(width / cellSize);
      rows = Math.ceil(height / cellSize);
      grid = createGrid(cols, rows);
      nextGrid = createGrid(cols, rows);

      if (keepOld && oldGrid.length) {
        const minRows = Math.min(rows, oldRows);
        const minCols = Math.min(cols, oldCols);
        for (let y = 0; y < minRows; y++) {
          for (let x = 0; x < minCols; x++) {
            grid[y][x] = oldGrid[y][x];
          }
        }
      }
    }

    function randomize() {
      generation = 0;
      for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
          grid[y][x] = Math.random() < 0.18 ? 1 : 0;
        }
      }
    }

    function clearGrid() {
      generation = 0;
      for (let y = 0; y < rows; y++) grid[y].fill(0);
    }

    function countNeighbors(x, y) {
      let sum = 0;
      for (let dy = -1; dy <= 1; dy++) {
        for (let dx = -1; dx <= 1; dx++) {
          if (dx === 0 && dy === 0) continue;
          const nx = (x + dx + cols) % cols;
          const ny = (y + dy + rows) % rows;
          sum += grid[ny][nx];
        }
      }
      return sum;
    }

    function stepSimulation() {
      for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
          const alive = grid[y][x];
          const neighbors = countNeighbors(x, y);
          if (alive && (neighbors === 2 || neighbors === 3)) nextGrid[y][x] = 1;
          else if (!alive && neighbors === 3) nextGrid[y][x] = 1;
          else nextGrid[y][x] = 0;
        }
      }
      const temp = grid;
      grid = nextGrid;
      nextGrid = temp;
      generation++;
    }

    function drawGrid() {
      ctx.clearRect(0, 0, width, height);
      ctx.fillStyle = "#020617";
      ctx.fillRect(0, 0, width, height);

      let alive = 0;
      for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
          if (grid[y][x]) {
            alive++;
            const px = x * cellSize;
            const py = y * cellSize;
            const hueAlpha = 0.55 + 0.35 * Math.sin((x + y + generation) * 0.05);
            ctx.fillStyle = `rgba(94, 234, 212, ${hueAlpha})`;
            ctx.fillRect(px + 1, py + 1, cellSize - 2, cellSize - 2);
          }
        }
      }

      if (cellSize >= 9) {
        ctx.strokeStyle = "rgba(148, 163, 184, 0.12)";
        ctx.lineWidth = 1;
        for (let x = 0; x <= cols; x++) {
          ctx.beginPath();
          ctx.moveTo(x * cellSize, 0);
          ctx.lineTo(x * cellSize, height);
          ctx.stroke();
        }
        for (let y = 0; y <= rows; y++) {
          ctx.beginPath();
          ctx.moveTo(0, y * cellSize);
          ctx.lineTo(width, y * cellSize);
          ctx.stroke();
        }
      }

      if (selectedPattern) drawPatternPreview();

      generationSpan.textContent = generation;
      aliveCountSpan.textContent = alive;
      speedValue.textContent = `${speedSlider.value} 世代/秒`;
      cellSizeValue.textContent = `${cellSize}px`;
    }

    let lastMouseCell = { x: 0, y: 0 };

    function drawPatternPreview() {
      const pattern = patterns[selectedPattern];
      if (!pattern) return;
      ctx.fillStyle = "rgba(250, 204, 21, 0.38)";
      for (let y = 0; y < pattern.length; y++) {
        for (let x = 0; x < pattern[y].length; x++) {
          if (pattern[y][x]) {
            ctx.fillRect((lastMouseCell.x + x) * cellSize + 1, (lastMouseCell.y + y) * cellSize + 1, cellSize - 2, cellSize - 2);
          }
        }
      }
    }

    function animate(timestamp) {
      const fps = parseInt(speedSlider.value, 10);
      const interval = 1000 / fps;
      if (running && timestamp - lastUpdate > interval) {
        stepSimulation();
        lastUpdate = timestamp;
      }
      drawGrid();
      requestAnimationFrame(animate);
    }

    function cellFromEvent(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return {
        x: Math.floor((clientX - rect.left) / cellSize),
        y: Math.floor((clientY - rect.top) / cellSize),
      };
    }

    function setCellFromEvent(event) {
      const c = cellFromEvent(event);
      lastMouseCell = c;
      if (c.x < 0 || c.x >= cols || c.y < 0 || c.y >= rows) return;
      grid[c.y][c.x] = drawState;
    }

    function placePattern(name, x, y) {
      const pattern = patterns[name];
      if (!pattern) return;
      for (let py = 0; py < pattern.length; py++) {
        for (let px = 0; px < pattern[py].length; px++) {
          const gx = x + px;
          const gy = y + py;
          if (gx >= 0 && gx < cols && gy >= 0 && gy < rows) {
            grid[gy][gx] = pattern[py][px];
          }
        }
      }
    }

    canvas.addEventListener("pointerdown", (event) => {
      const c = cellFromEvent(event);
      lastMouseCell = c;
      if (selectedPattern) {
        placePattern(selectedPattern, c.x, c.y);
        selectedPattern = null;
        document.querySelectorAll("[data-pattern]").forEach(btn => btn.classList.remove("active"));
        return;
      }
      drawing = true;
      setCellFromEvent(event);
    });

    canvas.addEventListener("pointermove", (event) => {
      const c = cellFromEvent(event);
      lastMouseCell = c;
      if (drawing) setCellFromEvent(event);
    });

    window.addEventListener("pointerup", () => {
      drawing = false;
    });

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    stepButton.addEventListener("click", () => {
      stepSimulation();
    });

    drawModeButton.addEventListener("click", () => {
      drawState = 1;
      drawModeButton.classList.add("active");
      eraseModeButton.classList.remove("active");
    });

    eraseModeButton.addEventListener("click", () => {
      drawState = 0;
      eraseModeButton.classList.add("active");
      drawModeButton.classList.remove("active");
    });

    document.querySelectorAll("[data-pattern]").forEach(button => {
      button.addEventListener("click", () => {
        selectedPattern = button.dataset.pattern;
        document.querySelectorAll("[data-pattern]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
      });
    });

    randomButton.addEventListener("click", randomize);
    clearButton.addEventListener("click", clearGrid);

    cellSizeSlider.addEventListener("input", () => {
      rebuildGrid(true);
    });

    window.addEventListener("resize", resize);

    resize();
    randomize();
    requestAnimationFrame(animate);
  </script>
