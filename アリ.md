  <script>
    const canvas = document.getElementById("canvas");
    const ctx = canvas.getContext("2d");

    const playPauseButton = document.getElementById("playPause");
    const resetButton = document.getElementById("reset");
    const clearPheromoneButton = document.getElementById("clearPheromone");
    const addAntsButton = document.getElementById("addAnts");
    const antCountSlider = document.getElementById("antCount");
    const evaporationSlider = document.getElementById("evaporation");
    const followSlider = document.getElementById("follow");
    const randomnessSlider = document.getElementById("randomness");
    const antCountValue = document.getElementById("antCountValue");
    const evapValue = document.getElementById("evapValue");
    const followValue = document.getElementById("followValue");
    const randomValue = document.getElementById("randomValue");
    const deliveredSpan = document.getElementById("delivered");
    const searchingSpan = document.getElementById("searching");

    let width = 0;
    let height = 0;
    let scale = 4;
    let gridW = 0;
    let gridH = 0;
    let homePheromone;
    let foodPheromone;
    let obstacles;
    let ants = [];
    let foods = [];
    let nest = { x: 0, y: 0 };
    let running = true;
    let tool = "food";
    let drawing = false;
    let delivered = 0;
    let pheromoneCanvas = document.createElement("canvas");
    let pheromoneCtx = pheromoneCanvas.getContext("2d");

    class Ant {
      constructor() {
        this.resetAtNest();
      }

      resetAtNest() {
        this.x = nest.x + (Math.random() - 0.5) * 12;
        this.y = nest.y + (Math.random() - 0.5) * 12;
        this.angle = Math.random() * Math.PI * 2;
        this.hasFood = false;
        this.speed = 1.6 + Math.random() * 0.7;
        this.turnNoise = 0;
      }

      update() {
        const targetPheromone = this.hasFood ? homePheromone : foodPheromone;
        const follow = parseFloat(followSlider.value);
        const random = parseFloat(randomnessSlider.value);

        const bestAngle = this.sense(targetPheromone);
        this.angle = mixAngle(this.angle, bestAngle, 0.08 * follow);
        this.angle += (Math.random() - 0.5) * random;

        const nx = this.x + Math.cos(this.angle) * this.speed;
        const ny = this.y + Math.sin(this.angle) * this.speed;

        if (this.isBlocked(nx, ny)) {
          this.angle += Math.PI * (0.6 + Math.random() * 0.8);
          return;
        }

        this.x = nx;
        this.y = ny;

        if (this.x < 4 || this.x > width - 4 || this.y < 4 || this.y > height - 4) {
          this.angle += Math.PI;
          this.x = clamp(this.x, 4, width - 4);
          this.y = clamp(this.y, 4, height - 4);
        }

        this.deposit();
        this.checkFoodOrNest();
      }

      sense(field) {
        const sensorDist = 18;
        const sensorAngle = 0.65;
        const angles = [this.angle - sensorAngle, this.angle, this.angle + sensorAngle];
        let best = this.angle;
        let bestValue = -Infinity;

        for (const a of angles) {
          const sx = this.x + Math.cos(a) * sensorDist;
          const sy = this.y + Math.sin(a) * sensorDist;
          const value = sampleField(field, sx, sy);
          if (value > bestValue) {
            bestValue = value;
            best = a;
          }
        }

        // 何も匂いがない時は、少し前へ進む。
        if (bestValue < 0.01) return this.angle + (Math.random() - 0.5) * 0.8;
        return best;
      }

      deposit() {
        const gx = Math.floor(this.x / scale);
        const gy = Math.floor(this.y / scale);
        if (!insideGrid(gx, gy)) return;
        const i = idx(gx, gy);
        if (this.hasFood) foodPheromone[i] = Math.min(255, foodPheromone[i] + 8);
        else homePheromone[i] = Math.min(255, homePheromone[i] + 5);
      }

      checkFoodOrNest() {
        if (!this.hasFood) {
          for (let i = foods.length - 1; i >= 0; i--) {
            const f = foods[i];
            if (Math.hypot(this.x - f.x, this.y - f.y) < f.r) {
              this.hasFood = true;
              this.angle += Math.PI;
              f.amount--;
              if (f.amount <= 0) foods.splice(i, 1);
              break;
            }
          }
        } else {
          if (Math.hypot(this.x - nest.x, this.y - nest.y) < 24) {
            this.hasFood = false;
            delivered++;
            this.angle += Math.PI + (Math.random() - 0.5);
          }
        }
      }

      isBlocked(x, y) {
        const gx = Math.floor(x / scale);
        const gy = Math.floor(y / scale);
        if (!insideGrid(gx, gy)) return true;
        return obstacles[idx(gx, gy)] === 1;
      }

      draw() {
        ctx.save();
        ctx.translate(this.x, this.y);
        ctx.rotate(this.angle);
        ctx.fillStyle = this.hasFood ? "#facc15" : "#e5e7eb";
        ctx.shadowColor = this.hasFood ? "#facc15" : "rgba(255,255,255,0.45)";
        ctx.shadowBlur = this.hasFood ? 8 : 3;
        ctx.beginPath();
        ctx.ellipse(0, 0, 4.2, 2.1, 0, 0, Math.PI * 2);
        ctx.fill();
        ctx.beginPath();
        ctx.arc(4, 0, 2.2, 0, Math.PI * 2);
        ctx.fill();
        ctx.restore();
      }
    }

    function resize() {
      const dpr = window.devicePixelRatio || 1;
      width = window.innerWidth;
      height = window.innerHeight;
      canvas.width = Math.floor(width * dpr);
      canvas.height = Math.floor(height * dpr);
      canvas.style.width = width + "px";
      canvas.style.height = height + "px";
      ctx.setTransform(dpr, 0, 0, dpr, 0, 0);
      scale = width < 700 ? 5 : 4;
      gridW = Math.ceil(width / scale);
      gridH = Math.ceil(height / scale);
      pheromoneCanvas.width = gridW;
      pheromoneCanvas.height = gridH;
      resetWorld();
    }

    function resetWorld() {
      homePheromone = new Float32Array(gridW * gridH);
      foodPheromone = new Float32Array(gridW * gridH);
      obstacles = new Uint8Array(gridW * gridH);
      foods = [{ x: width * 0.78, y: height * 0.48, r: 22, amount: 900 }];
      nest = { x: width * 0.22, y: height * 0.52 };
      delivered = 0;
      ants = [];
      for (let i = 0; i < parseInt(antCountSlider.value, 10); i++) ants.push(new Ant());
    }

    function idx(x, y) { return y * gridW + x; }
    function insideGrid(x, y) { return x >= 0 && x < gridW && y >= 0 && y < gridH; }
    function clamp(v, min, max) { return Math.max(min, Math.min(max, v)); }

    function sampleField(field, x, y) {
      const gx = Math.floor(x / scale);
      const gy = Math.floor(y / scale);
      if (!insideGrid(gx, gy)) return 0;
      if (obstacles[idx(gx, gy)]) return 0;
      return field[idx(gx, gy)];
    }

    function mixAngle(a, b, t) {
      let diff = ((b - a + Math.PI * 3) % (Math.PI * 2)) - Math.PI;
      return a + diff * t;
    }

    function evaporateAndDiffuse() {
      const evap = parseFloat(evaporationSlider.value);
      for (let i = 0; i < homePheromone.length; i++) {
        homePheromone[i] *= evap;
        foodPheromone[i] *= evap;
      }

      // 簡易拡散：近くに少しにじませる。
      for (let y = 1; y < gridH - 1; y += 2) {
        for (let x = 1; x < gridW - 1; x += 2) {
          const i = idx(x, y);
          if (obstacles[i]) continue;
          const h = homePheromone[i] * 0.015;
          const f = foodPheromone[i] * 0.015;
          homePheromone[idx(x - 1, y)] += h;
          homePheromone[idx(x + 1, y)] += h;
          homePheromone[idx(x, y - 1)] += h;
          homePheromone[idx(x, y + 1)] += h;
          foodPheromone[idx(x - 1, y)] += f;
          foodPheromone[idx(x + 1, y)] += f;
          foodPheromone[idx(x, y - 1)] += f;
          foodPheromone[idx(x, y + 1)] += f;
        }
      }
    }

    function drawBackground() {
      ctx.fillStyle = "rgba(7, 11, 20, 0.35)";
      ctx.fillRect(0, 0, width, height);
    }

    function drawPheromones() {
      const imageData = pheromoneCtx.createImageData(gridW, gridH);
      const data = imageData.data;
      for (let i = 0; i < gridW * gridH; i++) {
        const h = Math.min(255, homePheromone[i]);
        const f = Math.min(255, foodPheromone[i]);
        const p = i * 4;
        data[p] = f * 1.0;
        data[p + 1] = h * 0.75 + f * 0.55;
        data[p + 2] = h * 1.0;
        data[p + 3] = Math.min(210, Math.max(h, f) * 1.3);
      }
      pheromoneCtx.putImageData(imageData, 0, 0);
      ctx.imageSmoothingEnabled = false;
      ctx.drawImage(pheromoneCanvas, 0, 0, width, height);
    }

    function drawObstacles() {
      ctx.fillStyle = "rgba(100, 116, 139, 0.95)";
      for (let y = 0; y < gridH; y++) {
        for (let x = 0; x < gridW; x++) {
          if (obstacles[idx(x, y)]) {
            ctx.fillRect(x * scale, y * scale, scale + 0.5, scale + 0.5);
          }
        }
      }
    }

    function drawNestAndFood() {
      ctx.save();
      ctx.beginPath();
      ctx.arc(nest.x, nest.y, 25, 0, Math.PI * 2);
      ctx.fillStyle = "#a78bfa";
      ctx.shadowColor = "#a78bfa";
      ctx.shadowBlur = 18;
      ctx.fill();
      ctx.fillStyle = "white";
      ctx.font = "bold 13px system-ui";
      ctx.textAlign = "center";
      ctx.textBaseline = "middle";
      ctx.fillText("巣", nest.x, nest.y);
      ctx.restore();

      for (const f of foods) {
        ctx.save();
        ctx.beginPath();
        ctx.arc(f.x, f.y, f.r, 0, Math.PI * 2);
        ctx.fillStyle = "#22c55e";
        ctx.shadowColor = "#22c55e";
        ctx.shadowBlur = 18;
        ctx.fill();
        ctx.fillStyle = "white";
        ctx.font = "bold 12px system-ui";
        ctx.textAlign = "center";
        ctx.textBaseline = "middle";
        ctx.fillText("エサ", f.x, f.y);
        ctx.restore();
      }
    }

    function animate() {
      if (running) {
        evaporateAndDiffuse();
        syncAntCount();
        for (const ant of ants) ant.update();
      }

      drawBackground();
      drawPheromones();
      drawObstacles();
      drawNestAndFood();
      for (const ant of ants) ant.draw();
      updateLabels();

      requestAnimationFrame(animate);
    }

    function syncAntCount() {
      const target = parseInt(antCountSlider.value, 10);
      while (ants.length < target) ants.push(new Ant());
      while (ants.length > target) ants.pop();
    }

    function updateLabels() {
      antCountValue.textContent = antCountSlider.value;
      evapValue.textContent = parseFloat(evaporationSlider.value).toFixed(3);
      followValue.textContent = parseFloat(followSlider.value).toFixed(2);
      randomValue.textContent = parseFloat(randomnessSlider.value).toFixed(2);
      deliveredSpan.textContent = delivered;
      searchingSpan.textContent = ants.filter(a => !a.hasFood).length;
    }

    function pointerPos(event) {
      const rect = canvas.getBoundingClientRect();
      const clientX = event.touches ? event.touches[0].clientX : event.clientX;
      const clientY = event.touches ? event.touches[0].clientY : event.clientY;
      return { x: clientX - rect.left, y: clientY - rect.top };
    }

    function applyTool(event) {
      const p = pointerPos(event);
      if (tool === "food") {
        foods.push({ x: p.x, y: p.y, r: 22, amount: 700 });
      } else if (tool === "nest") {
        nest.x = p.x;
        nest.y = p.y;
        for (const ant of ants) ant.resetAtNest();
        clearPheromones();
      } else if (tool === "obstacle") {
        paintObstacle(p.x, p.y, 18, 1);
      } else if (tool === "erase") {
        paintObstacle(p.x, p.y, 26, 0);
        foods = foods.filter(f => Math.hypot(f.x - p.x, f.y - p.y) > f.r + 18);
      } else if (tool === "shortcut") {
        makeShortcutExperiment();
      } else if (tool === "maze") {
        makeMaze();
      }
    }

    function paintObstacle(x, y, radius, value) {
      const gx = Math.floor(x / scale);
      const gy = Math.floor(y / scale);
      const gr = Math.ceil(radius / scale);
      for (let dy = -gr; dy <= gr; dy++) {
        for (let dx = -gr; dx <= gr; dx++) {
          const nx = gx + dx;
          const ny = gy + dy;
          if (!insideGrid(nx, ny)) continue;
          if (Math.hypot(dx, dy) <= gr) obstacles[idx(nx, ny)] = value;
        }
      }
    }

    function clearPheromones() {
      homePheromone.fill(0);
      foodPheromone.fill(0);
    }

    function makeShortcutExperiment() {
      obstacles.fill(0);
      clearPheromones();
      foods = [{ x: width * 0.82, y: height * 0.50, r: 22, amount: 1200 }];
      nest = { x: width * 0.18, y: height * 0.50 };

      // 中央に壁を置き、上下どちらかを迂回する状況を作る。
      for (let y = Math.floor(gridH * 0.18); y < Math.floor(gridH * 0.82); y++) {
        for (let x = Math.floor(gridW * 0.48); x < Math.floor(gridW * 0.52); x++) {
          if (y > gridH * 0.42 && y < gridH * 0.52) continue;
          obstacles[idx(x, y)] = 1;
        }
      }
      for (const ant of ants) ant.resetAtNest();
    }

    function makeMaze() {
      obstacles.fill(0);
      clearPheromones();
      foods = [{ x: width * 0.82, y: height * 0.25, r: 22, amount: 1200 }];
      nest = { x: width * 0.18, y: height * 0.75 };

      const walls = [
        {x1:0.15,y1:0.35,x2:0.72,y2:0.39},
        {x1:0.28,y1:0.60,x2:0.88,y2:0.64},
        {x1:0.48,y1:0.15,x2:0.52,y2:0.60},
        {x1:0.68,y1:0.39,x2:0.72,y2:0.88},
      ];
      for (const w of walls) {
        const x1 = Math.floor(gridW * w.x1), x2 = Math.floor(gridW * w.x2);
        const y1 = Math.floor(gridH * w.y1), y2 = Math.floor(gridH * w.y2);
        for (let y = y1; y <= y2; y++) {
          for (let x = x1; x <= x2; x++) {
            if (insideGrid(x, y)) obstacles[idx(x, y)] = 1;
          }
        }
      }
      for (const ant of ants) ant.resetAtNest();
    }

    canvas.addEventListener("pointerdown", (event) => {
      drawing = true;
      applyTool(event);
    });

    canvas.addEventListener("pointermove", (event) => {
      if (!drawing) return;
      if (tool === "obstacle" || tool === "erase") applyTool(event);
    });

    window.addEventListener("pointerup", () => drawing = false);

    document.querySelectorAll("[data-tool]").forEach(button => {
      button.addEventListener("click", () => {
        tool = button.dataset.tool;
        document.querySelectorAll("[data-tool]").forEach(btn => btn.classList.remove("active"));
        button.classList.add("active");
      });
    });

    playPauseButton.addEventListener("click", () => {
      running = !running;
      playPauseButton.textContent = running ? "一時停止" : "再生";
      playPauseButton.classList.toggle("green", running);
    });

    resetButton.addEventListener("click", resetWorld);
    clearPheromoneButton.addEventListener("click", clearPheromones);
    addAntsButton.addEventListener("click", () => {
      antCountSlider.value = Math.min(500, parseInt(antCountSlider.value, 10) + 60);
      syncAntCount();
    });

    window.addEventListener("resize", resize);

    resize();
    animate();
